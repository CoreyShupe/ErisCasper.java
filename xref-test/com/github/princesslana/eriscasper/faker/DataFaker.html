12  public class DataFaker {
13    private DataFaker() {}
14  
15    public static String discriminator() {
16      return RandomStringUtils.randomNumeric(4);
17    }
18  
19    public static Message message() {
20      return ImmutableMessage.builder()
21          .id(DiscordFaker.snowflake())
22          .author(user())
23          .channelId(DiscordFaker.snowflake())
24          .content(Faker.instance().chuckNorris().fact())
25          .build();
26    }
27  
28    public static ReadyEventData ready() {
29      return ImmutableReadyEventData.builder()
30          .v(6L)
31          .user(user())
32          .sessionId(DiscordFaker.sessionId().unwrap())
33          .build();
34    }
35  
36    public static User user() {
37      return ImmutableUser.builder()
38          .id(DiscordFaker.snowflake())
39          .username(username())
40          .discriminator(discriminator())
41          .build();
42    }
43  
44    public static String username() {
45      return Faker.instance().name().name();
46    }
47  }
