1   package com.github.princesslana.eriscasper.util;

3   import com.github.princesslana.eriscasper.ErisCasperInfo;
4   import okhttp3.OkHttpClient;
5   
6   public class OkHttp {
7     private OkHttp() {}
8   
9     /**
10     * Creates a OkHttpClient for use within ErisCasper. Ensures we set the user agent correctly on
11     * all requests.
12     *
13     * @see <a href="https://discordapp.com/developers/docs/reference#user-agent">
14     *     https://discordapp.com/developers/docs/reference#user-agent</a>
15     */
16    public static OkHttpClient newHttpClient() {
17      ErisCasperInfo info = ErisCasperInfo.load();
18  
19      String userAgent = String.format("DiscordBot (%s, %s)", info.getUrl(), info.getVersion());
20  
21      new OkHttpClient.Builder()
22          .addInterceptor(
23              c -> c.proceed(c.request().newBuilder().header("User-Agent", userAgent).build()))
24          .build();
25  
26      return new OkHttpClient();
27    }
28  }
