26  public class Routes {
27  
28    private static final Logger LOG = LoggerFactory.getLogger(Routes.class);
29  
30    private static final MediaType MEDIA_TYPE_JSON = MediaType.parse("application/json");
31  
32    private final RateLimiterRegistry rateLimiterRegistry = RateLimiterRegistry.ofDefaults();
33  
34    private final BotToken token;
35  
36    private final OkHttpClient client;
37    private final ObjectMapper jackson;
38  
39    public Routes(BotToken token, OkHttpClient client, ObjectMapper jackson) {
40      this.token = token;
41      this.client = client;
42      this.jackson = jackson;
43    }
44  
45    public <O> Single<O> execute(Route<Void, O> route) {
46      return execute(route, null);
47    }
48  
49    public <I, O> Single<O> execute(Route<I, O> route, I data) {
50      Function<Optional<RequestBody>, Request> buildRequest =
51          body ->
52              new Request.Builder()
53                  .method(route.getMethod().get(), body.orElse(null))
54                  .url(route.getUrl())
55                  .header("Authorization", "Bot " + token.unwrap())
56                  .build();
57  
58      return Maybes.fromNullable(data)
59          .map(d -> RequestBody.create(MEDIA_TYPE_JSON, jackson.writeValueAsString(data)))
60          .map(Optional::of)
61          .toSingle(Optional.empty())
62          .map(buildRequest)
63          .flatMap(rq -> executeRequest(route, rq))
64          .lift(RateLimiterOperator.of(getRateLimiter(route)));
65    }
66  
67    private <O> Single<O> executeRequest(Route<?, O> route, Request rq) {
68      Callable<Response> execute =
69          () -> {
70            LOG.debug("Executing: {}...", route);
71            return client.newCall(rq).execute();
72          };
73  
74      Consumer<Response> close =
75          r -> {
76            r.close();
77            LOG.debug("Closed: {}.", r);
78          };
79  
80      Consumer<Response> updateRateLimit =
81          r -> {
82            String remainingHeader = r.header("X-RateLimit-Remaining");
83            String resetHeader = r.header("X-RateLimit-Reset");
84  
85            try {
86              int remaining = Integer.parseInt(remainingHeader, 10);
87              Instant until = Instant.ofEpochSecond(Long.parseLong(resetHeader));
88  
89              RateLimiter rl = getRateLimiter(route);
90  
91              rl.changeLimitForPeriod(remaining);
92              rl.changeTimeoutDuration(Duration.between(Instant.now(), until));
93            } catch (IllegalArgumentException e) {
94              // we use a little EAFP here (https://docs.python.org/3/glossary.html#term-eafp)
95              // If the headers are absent or not numeric or the values passed into the rate limiter
96              // are invalid we may end up here
97              LOG.debug(
98                  "Could not update rate limit to {}/{} ({})",
99                  remainingHeader,
100                 resetHeader,
101                 e.getMessage());
102           }
103         };
104 
105     // Single#using does not work here, as it performs the close operation immediately upon emitting
106     // the response, because once it's received the response it knows it can receive no more.
107     //
108     // Observable#using closes the response at a more appropriate time. My feeling is that this is
109     // wrong - we're tricking rxjava into thinking there is something coming next so as to delay the
110     // close.
111     return Observable.using(execute, Observable::just, close)
112         .subscribeOn(Schedulers.io())
113         .doOnNext(r -> LOG.debug("Done: {} -> {}.", route, r))
114         .doOnNext(updateRateLimit)
115         .flatMapSingle(
116             r ->
117                 r.isSuccessful()
118                     ? Single.just(r)
119                     : Single.error(new IllegalStateException("Unexpected response: ")))
120         .doOnError(e -> LOG.warn("Error: {} - {}.", route, e))
121         .map(rs -> jackson.readValue(rs.body().byteStream(), route.getResponseClass()))
122         .firstOrError();
123   }
124 
125   private RateLimiter getRateLimiter(Route<?, ?> r) {
126     return rateLimiterRegistry.rateLimiter(r.getUrl());
127   }
128 }
