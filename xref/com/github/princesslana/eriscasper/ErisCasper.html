3   import com.fasterxml.jackson.databind.ObjectMapper;
4   import com.github.princesslana.eriscasper.data.event.Event;
5   import com.github.princesslana.eriscasper.gateway.Gateway;
6   import com.github.princesslana.eriscasper.gateway.Payloads;
7   import com.github.princesslana.eriscasper.repository.RepositoryManager;
8   import com.github.princesslana.eriscasper.rest.RouteCatalog;
9   import com.github.princesslana.eriscasper.rest.Routes;
10  import com.github.princesslana.eriscasper.util.Jackson;
11  import com.github.princesslana.eriscasper.util.OkHttp;
12  import io.reactivex.Observable;
13  import io.reactivex.Single;
14  import io.reactivex.schedulers.Schedulers;
15  import okhttp3.OkHttpClient;
16  import org.slf4j.Logger;
17  import org.slf4j.LoggerFactory;
18  
19  public class ErisCasper {
20  
21    private static final Logger LOG = LoggerFactory.getLogger(ErisCasper.class);
22  
23    private final BotToken token;
24  
25    private final OkHttpClient httpClient = OkHttp.newHttpClient();
26    private final ObjectMapper jackson = Jackson.newObjectMapper();
27    private final Payloads payloads = new Payloads(jackson);
28  
29    private final Routes routes;
30  
31    private ErisCasper(BotToken token) {
32      this.token = token;
33  
34      routes = new Routes(token, httpClient, jackson);
35    }
36  
37    private Observable<Event> getEvents() {
38      Gateway gateway = Gateway.create(httpClient, payloads);
39  
40      return Single.just(RouteCatalog.getGateway())
41          .observeOn(Schedulers.io())
42          .flatMap(routes::execute)
43          .toObservable()
44          .flatMap(gr -> gateway.connect(gr.getUrl(), token))
45          .observeOn(Schedulers.computation())
46          .share();
47    }
48  
49    public void run(Bot bot) {
50      Observable<Event> events = getEvents();
51  
52      RepositoryManager rm = RepositoryManager.create(events);
53  
54      bot.apply(new BotContext(events, routes, rm))
55          .doOnError(t -> LOG.warn("Exception thrown by Bot", t))
56          .blockingAwait();
57    }
58  
59    public static ErisCasper create() {
60      return create(System.getenv("EC_TOKEN"));
61    }
62  
63    public static ErisCasper create(String token) {
64      return new ErisCasper(BotToken.of(token));
65    }
66  }
