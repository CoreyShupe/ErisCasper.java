23    public Payloads(ObjectMapper jackson) {
24      this.jackson = jackson;
25    }
26  
27    public <T> Single<T> dataAs(Payload p, Class<T> clazz) {
28      return Single.fromCallable(() -> jackson.readerFor(clazz).readValue(p.d().get()));
29    }
30  
31    public Payload heartbeat(SequenceNumber s) {
32      return ImmutablePayload.builder().op(OpCode.HEARTBEAT).s(s).build();
33    }
34  
35    public Payload identify(BotToken token) {
36      return identify(ImmutableIdentify.builder().token(token).build());
37    }
38  
39    public Payload identify(Identify id) {
40      return ImmutablePayload.builder().op(OpCode.IDENTIFY).d(jackson.valueToTree(id)).build();
41    }
42  
43    public Single<Payload> read(String text) {
44      return Single.fromCallable(() -> jackson.readValue(text, Payload.class));
45    }
46  
47    public Maybe<Event<?>> toEvent(Payload payload) {
48      return Single.just(payload)
49          .filter(Payload.isOp(OpCode.DISPATCH))
50          .flatMap(p -> Maybes.fromOptional(p.t()))
51          .map(EventType::getFactory)
52          .flatMap(f -> dataAs(payload, f.getDataClass()).map(f::apply).toMaybe())
53          .doOnError(t -> LOG.warn("Unable to convert payload to event: {}", payload, t))
54          .onErrorComplete();
55    }
56  
57    public Single<String> writeToString(Payload p) {
58      return Single.fromCallable(() -> jackson.writeValueAsString(p));
59    }
