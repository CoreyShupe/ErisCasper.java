14  /**
15   * @see <a href="https://discordapp.com/developers/docs/topics/gateway#payloads">
16   *     https://discordapp.com/developers/docs/topics/gateway#payloads</a>
17   */
18  @Value.Immutable
19  @JsonDeserialize(as = ImmutablePayload.class)
20  public abstract class Payload {
21    public abstract OpCode op();
22  
23    protected abstract Optional<JsonNode> d();
24  
25    public abstract Optional<SequenceNumber> s();
26  
27    public abstract Optional<EventType> t();
28  
29    public <T> Single<T> d(ObjectMapper jackson, Class<T> clazz) {
30      return Single.fromCallable(() -> jackson.readerFor(clazz).readValue(d().get()));
31    }
32  
33    public static Predicate<Payload> isOp(OpCode op) {
34      return p -> p.op() == op;
35    }
36  
37    @Value.Immutable
38    @Wrapped
39    public static interface SequenceNumberWrapper extends Wrapper<Integer> {}
40  }
