11  import io.github.resilience4j.ratelimiter.RateLimiter;
12  import io.github.resilience4j.ratelimiter.RateLimiterConfig;
13  import io.github.resilience4j.ratelimiter.operator.RateLimiterOperator;

22  import org.slf4j.LoggerFactory;

24  public class Gateway {
25  
26    private static final Logger LOG = LoggerFactory.getLogger(Gateway.class);
27  
28    /**
29     * @see <a href="https://discordapp.com/developers/docs/topics/gateway#sending-payloads">
30     *     https://discordapp.com/developers/docs/topics/gateway#sending-payloads</a>
31     */
32    private static final int MAX_MESSAGE_SIZE = 4096;
33  
34    private static final String VERSION = "6";
35    private static final String ENCODING = "json";
36  
37    private final RxWebSocket ws;
38    private final Payloads payloads;
39  
40    private Optional<SequenceNumber> lastSeenSequenceNumber = Optional.empty();
41  
42    private Optional<SessionId> sessionId = Optional.empty();
43  
44    /**
45     * @see <a href="https://discordapp.com/developers/docs/topics/gateway#rate-limiting">
46     *     https://discordapp.com/developers/docs/topics/gateway#rate-limiting</a>
47     */
48    private final RateLimiter sendLimit =
49        RateLimiter.of(
50            "Gateway#sendLimit",
51            RateLimiterConfig.custom()
52                .limitRefreshPeriod(Duration.ofSeconds(60))
53                .limitForPeriod(120)
54                .timeoutDuration(Duration.ofSeconds(60))
55                .build());
56  
57    /**
58     * @see <a href="https://discordapp.com/developers/docs/topics/gateway#identifying">
59     *     https://discordapp.com/developers/docs/topics/gateway#identifying</a>
60     */
61    private final RateLimiter identifyLimit =
62        RateLimiter.of(
63            "Gateway#identifyLimit",
64            RateLimiterConfig.custom()
65                .limitRefreshPeriod(Duration.ofSeconds(5))
66                .limitForPeriod(1)
67                .timeoutDuration(Duration.ofSeconds(60))
68                .build());
69  
70    public Gateway(RxWebSocket ws, Payloads payloads) {
71      this.ws = ws;
72      this.payloads = payloads;
73    }
74  
75    private boolean isResumable() {
76      return sessionId.isPresent() && lastSeenSequenceNumber.isPresent();
77    }
78  
79    private void setSessionId(SessionId sid) {
80      this.sessionId = Optional.of(sid);
81    }
82  
83    private void sequenceNumberSeen(Optional<SequenceNumber> seq) {
84      seq.ifPresent(sid -> lastSeenSequenceNumber = Optional.of(sid));
85    }
86  
87    @SuppressWarnings("unchecked")
88    public Observable<Event> connect(String url, BotToken token) {
89      Observable<Payload> ps =
90          ws.connect(String.format("%s?v=%s&encoding=%s", url, VERSION, ENCODING))
91              .ofType(RxWebSocketEvent.StringMessage.class)
92              .map(RxWebSocketEvent.StringMessage::getText)
93              .flatMapMaybe(
94                  Singles.toMaybeAnd(
95                      payloads::read, (s, t) -> LOG.warn("Error reading payload: {}", s, t)))
96              .doOnNext(p -> sequenceNumberSeen(p.s()))
97              .share();
98  
99      Completable heartbeat =
100         ps.filter(Payload.isOp(OpCode.HELLO)).flatMapCompletable(p -> heartbeat(ws, p));
101 
102     Completable identify =
103         ps.filter(Payload.isOp(OpCode.HELLO))
104             .flatMapCompletable(p -> isResumable() ? resume(ws, token) : identify(ws, token));
105 
106     Observable<Event> events = ps.flatMapMaybe(payloads::toEvent).share();
107 
108     Completable setSessionId =
109         events
110             .ofType(ReadyEvent.class)
111             .map(r -> r.unwrap().getSessionId())
112             .doOnNext(s -> setSessionId(SessionId.of(s)))
113             .ignoreElements();
114 
115     return Observable.mergeArray(
116             events, setSessionId.toObservable(), heartbeat.toObservable(), identify.toObservable())
117         .doOnNext(e -> LOG.debug("Event: {}.", e));
118   }
119 
120   private Completable send(RxWebSocket ws, Payload payload) {
121     return payloads
122         .writeToString(payload)
123         .lift(RateLimiterOperator.of(sendLimit))
124         .filter(s -> s.getBytes().length <= MAX_MESSAGE_SIZE)
125         .doOnComplete(() -> LOG.warn("Payload rejected as too long: {}.", payload))
126         .flatMapCompletable(ws::send)
127         .doOnComplete(() -> LOG.debug("Sent: {}.", payload));
128   }
129 
130   private Completable identify(RxWebSocket ws, BotToken token) {
131     return Single.just(payloads.identify(token))
132         .lift(RateLimiterOperator.of(identifyLimit))
133         .flatMapCompletable(p -> send(ws, p));
134   }
135 
136   private Completable heartbeat(RxWebSocket ws, Payload hello) {
137     return payloads
138         .dataAs(hello, HelloEventData.class)
139         .flatMapObservable(
140             h -> Observable.interval(h.getHeartbeatInterval(), TimeUnit.MILLISECONDS))
141         .flatMapCompletable(l -> send(ws, payloads.heartbeat(lastSeenSequenceNumber)));
142   }
143 
144   private Completable resume(RxWebSocket ws, BotToken token) {
145     Preconditions.checkState(sessionId.isPresent(), "Can not resume without a session id");
146     Preconditions.checkState(
147         lastSeenSequenceNumber.isPresent(), "Can not resume without a sequence number");
148 
149     return Single.just(
150             ImmutableResume.builder()
151                 .token(token)
152                 .sessionId(sessionId.get())
153                 .seq(lastSeenSequenceNumber.get())
154                 .build())
155         .map(payloads::resume)
156         .flatMapCompletable(p -> send(ws, p));
157   }
158 
159   public static Gateway create(OkHttpClient client, Payloads payloads) {
160     return new Gateway(new RxWebSocket(client), payloads);
161   }
162 }
