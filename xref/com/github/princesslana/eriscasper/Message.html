3   import io.reactivex.Completable;
4   import net.dv8tion.jda.core.MessageBuilder;
5   import net.dv8tion.jda.core.entities.TextChannel;
6   import org.immutables.value.Value;
7   
8   @Value.Immutable
9   public interface Message {
10  
11    String getContent();
12  
13    public static Completable send(Message m, TextChannel to) {
14      return Completable.fromAction(
15          () -> to.sendMessage(new MessageBuilder().append(m.getContent()).build()).complete());
16    }
17  
18    public static Message from(String content) {
19      return ImmutableMessage.builder().content(content).build();
20    }
21  
22    public static Message from(net.dv8tion.jda.core.entities.Message jdaMsg) {
23      return from(jdaMsg.getContentRaw());
24    }
25  }
