1   package com.github.princesslana.eriscasper;
2   
3   import io.reactivex.Completable;
4   import io.reactivex.Observable;
5   import net.dv8tion.jda.core.entities.TextChannel;
6   import net.dv8tion.jda.core.entities.User;
7   import net.dv8tion.jda.core.events.Event;
8   import net.dv8tion.jda.core.events.message.MessageReceivedEvent;
9   import org.immutables.value.Value;
10  
11  @Value.Immutable
12  public interface ReceivedMessage {
13  
14    User getAuthor();
15  
16    Message getMessage();
17  
18    TextChannel getSource();
19  
20    public static Completable sendReply(ReceivedMessage to, String with) {
21      return sendReply(to, Message.from(with));
22    }
23  
24    public static Completable sendReply(ReceivedMessage to, Message with) {
25      return Message.send(with, to.getSource());
26    }
27  
28    public static ReceivedMessage from(MessageReceivedEvent evt) {
29      return ImmutableReceivedMessage.builder()
30          .author(evt.getAuthor())
31          .message(Message.from(evt.getMessage()))
32          .source(evt.getTextChannel())
33          .build();
34    }
35  
36    public static Observable<ReceivedMessage> from(ErisCasper ec) {
37      return from(ec.events());
38    }
39  
40    public static Observable<ReceivedMessage> from(Observable<Event> evts) {
41      return evts.ofType(MessageReceivedEvent.class).map(ReceivedMessage::from);
42    }
43  }
