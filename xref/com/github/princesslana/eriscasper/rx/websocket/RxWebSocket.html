1   package com.github.princesslana.eriscasper.rx.websocket;
2   
3   import io.reactivex.BackpressureStrategy;
4   import io.reactivex.Completable;
5   import io.reactivex.Flowable;
6   import io.reactivex.FlowableEmitter;
7   import java.io.Closeable;
8   import okhttp3.OkHttpClient;
9   import okhttp3.Request;
10  import okhttp3.Response;
11  import okhttp3.WebSocket;
12  import okhttp3.WebSocketListener;
13  import org.slf4j.Logger;
14  import org.slf4j.LoggerFactory;
15  
16  public class RxWebSocket implements Closeable {
17  
18    private static final Logger LOG = LoggerFactory.getLogger(RxWebSocket.class);
19  
20    private final OkHttpClient http;
21  
22    private WebSocket ws;
23  
24    public RxWebSocket(OkHttpClient http) {
25      this.http = http;
26    }
27  
28    public Flowable<RxWebSocketEvent> connect(String url) {
29      return Flowable.<RxWebSocketEvent>create(
30              em -> {
31                Request rq = new Request.Builder().url(url).build();
32  
33                ws = http.newWebSocket(rq, new Listener(em));
34              },
35              BackpressureStrategy.BUFFER)
36          .doOnNext(e -> LOG.debug("Received: {}.", e))
37          .doOnError(e -> LOG.warn("Error: {}.", e));
38    }
39  
40    public Completable send(String text) {
41      return Completable.fromAction(() -> ws.send(text))
42          .doOnComplete(() -> LOG.debug("Sent: {}.", text));
43    }
44  
45    @Override
46    public void close() {
47      if (ws != null) {
48        ws.close(1000, "Bye.");
49      }
50    }
51  
52    private static class Listener extends WebSocketListener {
53      private final FlowableEmitter<RxWebSocketEvent> em;
54  
55      public Listener(FlowableEmitter<RxWebSocketEvent> em) {
56        this.em = em;
57      }
58  
59      @Override
60      public void onClosed(WebSocket ws, int code, String reason) {
61        em.onNext(ClosedTuple.of(ws, code, reason));
62      }
63  
64      @Override
65      public void onClosing(WebSocket ws, int code, String reason) {
66        em.onNext(ClosingTuple.of(ws, code, reason));
67      }
68  
69      @Override
70      public void onFailure(WebSocket ws, Throwable t, Response response) {
71        em.onNext(FailureTuple.of(ws, t, response));
72        em.onError(t);
73      }
74  
75      @Override
76      public void onMessage(WebSocket ws, okio.ByteString bytes) {
77        em.onNext(ByteStringMessageTuple.of(ws, bytes));
78      }
79  
80      @Override
81      public void onMessage(WebSocket ws, String text) {
82        em.onNext(StringMessageTuple.of(ws, text));
83      }
84  
85      @Override
86      public void onOpen(WebSocket ws, Response response) {
87        em.onNext(OpenTuple.of(ws, response));
88      }
89    }
90  }
