1   package com.github.princesslana.eriscasper.rx.websocket;

3   import io.reactivex.Completable;
4   import io.reactivex.Observable;
5   import io.reactivex.ObservableEmitter;
6   import java.util.Optional;
7   import okhttp3.OkHttpClient;
8   import okhttp3.Request;
9   import okhttp3.Response;
10  import okhttp3.WebSocket;
11  import okhttp3.WebSocketListener;
12  import org.slf4j.Logger;
13  import org.slf4j.LoggerFactory;
14  
15  public class RxWebSocket {
16  
17    private static final Logger LOG = LoggerFactory.getLogger(RxWebSocket.class);
18  
19    private final OkHttpClient http;
20  
21    private WebSocket ws;
22  
23    public RxWebSocket(OkHttpClient http) {
24      this.http = http;
25    }
26  
27    public Observable<RxWebSocketEvent> connect(String url) {
28      return Observable.<RxWebSocketEvent>create(
29              em -> {
30                Request rq = new Request.Builder().url(url).build();
31  
32                ws = http.newWebSocket(rq, new Listener(em));
33              })
34          .doOnNext(e -> LOG.trace("Received: {}.", e))
35          .doOnError(e -> LOG.warn("Error: {}.", e));
36    }
37  
38    public Completable send(String text) {
39      return Completable.fromAction(() -> ws.send(text))
40          .doOnComplete(() -> LOG.trace("Sent: {}.", text));
41    }
42  
43    private static class Listener extends WebSocketListener {
44      private final ObservableEmitter<RxWebSocketEvent> em;
45  
46      public Listener(ObservableEmitter<RxWebSocketEvent> em) {
47        this.em = em;
48      }
49  
50      @Override
51      public void onClosed(WebSocket ws, int code, String reason) {
52        em.onNext(ClosedTuple.of(ws, code, reason));
53      }
54  
55      @Override
56      public void onClosing(WebSocket ws, int code, String reason) {
57        em.onNext(ClosingTuple.of(ws, code, reason));
58      }
59  
60      @Override
61      public void onFailure(WebSocket ws, Throwable t, Response response) {
62        em.onNext(FailureTuple.of(ws, t, Optional.ofNullable(response)));
63        em.onError(t);
64      }
65  
66      @Override
67      public void onMessage(WebSocket ws, okio.ByteString bytes) {
68        em.onNext(ByteStringMessageTuple.of(ws, bytes));
69      }
70  
71      @Override
72      public void onMessage(WebSocket ws, String text) {
73        em.onNext(StringMessageTuple.of(ws, text));
74      }
75  
76      @Override
77      public void onOpen(WebSocket ws, Response response) {
78        em.onNext(OpenTuple.of(ws, response));
79      }
80    }
81  }
