1   package com.github.princesslana.eriscasper.examples;
2   
3   import com.github.princesslana.eriscasper.ErisCasper;
4   import com.github.princesslana.eriscasper.event.Event;
5   import com.github.princesslana.eriscasper.event.Events;
6   import com.github.princesslana.eriscasper.rest.ImmutableSendMessageRequest;
7   import com.github.princesslana.eriscasper.rest.RouteCatalog;
8   
9   public class PingBot {
10  
11    public static void main(String[] args) {
12      ErisCasper.create()
13          .run(
14              ctx ->
15                  ctx.getEvents()
16  
17                      // only message create events
18                      .ofType(Events.MessageCreate.class)
19                      .map(Event::getData)
20  
21                      // only with content "+ping"
22                      .filter(d -> d.getContent().equals("+ping"))
23  
24                      // send create message request
25                      .map(d -> RouteCatalog.createMessage(d.getChannelId()))
26                      .flatMapCompletable(
27                          r ->
28                              ctx.execute(
29                                      r,
30                                      ImmutableSendMessageRequest.builder().content("pong").build())
31                                  .toCompletable()));
32    }
33  }
