14  /**
15   * Possible event types that we can receive from Discord.
16   *
17   * <p>Each EventType is associated with an {@link EventFactory} so that it is known how to construct
18   * an {@link Event} from a {@link com.github.princesslana.eriscasper.gateway.Payload}.
19   *
20   * <p>It is tempting to try and merge EventType and {@link EventFactory} into a single class, but we
21   * can not match the dataClass and factory types using generics within the EventType enum.
22   *
23   * @see <a
24   *     href="https://discordapp.com/developers/docs/topics/gateway#commands-and-events-gateway-events">
25   *     https://discordapp.com/developers/docs/topics/gateway#commands-and-events-gateway-events</a>
26   */
27  public enum EventType {
28    GUILD_CREATE(GuildCreateData.class, GuildCreate::of),
29    MESSAGE_CREATE(Message.class, MessageCreate::of),
30    READY(ReadyEventData.class, ReadyEvent::of),
31    RESUMED(ResumedEventData.class, ResumedEvent::of),
32    TYPING_START(TypingStartEventData.class, TypingStartEvent::of);
33  
34    private final EventFactory<?> factory;
35  
36    private EventType(EventFactory<?> factory) {
37      this.factory = factory;
38    }
39  
40    private <T> EventType(Class<T> dataClass, Function<T, Event> factory) {
41      this(EventFactoryTuple.of(dataClass, factory));
42    }
43  
44    public Class<?> getDataClass() {
45      return factory.getDataClass();
46    }
47  
48    public Event newEvent(Object data) {
49      return factory.apply(data);
50    }
51  }
