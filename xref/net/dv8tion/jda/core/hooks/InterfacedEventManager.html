17  package net.dv8tion.jda.core.hooks;
18  
19  import java.util.Collections;
20  import java.util.LinkedList;
21  import java.util.List;
22  import java.util.concurrent.CopyOnWriteArrayList;
23  import net.dv8tion.jda.core.entities.impl.JDAImpl;
24  import net.dv8tion.jda.core.events.Event;
25  
26  /**
27   * An {@link net.dv8tion.jda.core.hooks.IEventManager IEventManager} implementation that uses the
28   * {@link net.dv8tion.jda.core.hooks.EventListener EventListener} interface for event listeners.
29   *
30   * <p>This only accepts listeners that implement {@link net.dv8tion.jda.core.hooks.EventListener
31   * EventListener} <br>
32   * An adapter implementation is {@link net.dv8tion.jda.core.hooks.ListenerAdapter ListenerAdapter}
33   * which provides methods for each individual {@link net.dv8tion.jda.core.events.Event}.
34   *
35   * <p><b>This is the default IEventManager used by JDA</b>
36   *
37   * @see net.dv8tion.jda.core.hooks.AnnotatedEventManager
38   * @see net.dv8tion.jda.core.hooks.IEventManager
39   */
40  public class InterfacedEventManager implements IEventManager {
41    private final CopyOnWriteArrayList<EventListener> listeners = new CopyOnWriteArrayList<>();
42  
43    public InterfacedEventManager() {}
44  
45    /**
46     * {@inheritDoc}
47     *
48     * @throws IllegalArgumentException If the provided listener does not implement {@link
49     *     net.dv8tion.jda.core.hooks.EventListener EventListener}
50     */
51    @Override
52    public void register(Object listener) {
53      if (!(listener instanceof EventListener)) {
54        throw new IllegalArgumentException("Listener must implement EventListener");
55      }
56      listeners.add(((EventListener) listener));
57    }
58  
59    @Override
60    public void unregister(Object listener) {
61      listeners.remove(listener);
62    }
63  
64    @Override
65    public List<Object> getRegisteredListeners() {
66      return Collections.unmodifiableList(new LinkedList<>(listeners));
67    }
68  
69    @Override
70    public void handle(Event event) {
71      for (EventListener listener : listeners) {
72        try {
73          listener.onEvent(event);
74        } catch (Throwable throwable) {
75          JDAImpl.LOG.error("One of the EventListeners had an uncaught exception", throwable);
76        }
77      }
78    }
79  }
