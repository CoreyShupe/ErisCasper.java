17  package net.dv8tion.jda.core.hooks;
18  
19  import java.util.List;
20  import net.dv8tion.jda.core.events.Event;
21  
22  /**
23   * An interface for JDA's EventManager system. <br>
24   * This should be registered in the {@link net.dv8tion.jda.core.JDABuilder JDABuilder}
25   *
26   * <p>JDA provides 2 implementations:
27   *
28   * <ul>
29   *   <li>{@link net.dv8tion.jda.core.hooks.InterfacedEventManager InterfacedEventManager} <br>
30   *       Simple implementation that allows {@link net.dv8tion.jda.core.hooks.EventListener
31   *       EventListener} instances as listeners.
32   *   <li>{@link net.dv8tion.jda.core.hooks.AnnotatedEventManager AnnotatedEventManager} <br>
33   *       An implementation that accepts any object and uses the {@link
34   *       net.dv8tion.jda.core.hooks.SubscribeEvent SubscribeEvent} annotation to handle events.
35   * </ul>
36   *
37   * <p>The default event manager is {@link net.dv8tion.jda.core.hooks.InterfacedEventManager
38   * InterfacedEventManager} <br>
39   * Use {@link net.dv8tion.jda.core.JDABuilder#setEventManager(IEventManager)
40   * JDABuilder.setEventManager(IEventManager)} to set the preferred event manager implementation.
41   * <br>
42   * You can only use one implementation per JDA instance!
43   *
44   * @see net.dv8tion.jda.core.hooks.InterfacedEventManager
45   * @see net.dv8tion.jda.core.hooks.AnnotatedEventManager
46   */
47  public interface IEventManager {
48  
49    /**
50     * Registers the specified listener <br>
51     * Accepted types may be specified by implementations
52     *
53     * @param listener A listener object
54     */
55    void register(Object listener);
56  
57    /**
58     * Removes the specified listener
59     *
60     * @param listener The listener object to remove
61     */
62    void unregister(Object listener);
63  
64    /**
65     * Handles the provided {@link net.dv8tion.jda.core.events.Event Event}. How this is handled is
66     * specified by the implementation.
67     *
68     * @param event The event to handle
69     */
70    void handle(Event event);
71  
72    /**
73     * The currently registered listeners
74     *
75     * @return An immutable list of listeners that have already been registered
76     */
77    List<Object> getRegisteredListeners();
78  }
