88  /**
89   * An abstract implementation of {@link net.dv8tion.jda.core.hooks.EventListener EventListener}
90   * which divides {@link net.dv8tion.jda.core.events.Event Events} for you.
91   *
92   * <p><b>Example:</b> <br>
93   *
94   * <pre><code>
95   * public class MyReadyListener extends ListenerAdapter
96   * {
97   *    {@literal @Override}
98   *     public void onReady(ReadyEvent event)
99   *     {
100  *         System.out.println("I am ready to go!");
101  *     }
102  *
103  *    {@literal @Override}
104  *     public void onMessageReceived(MessageReceivedEvent event)
105  *     {
106  *         System.out.printf("[%s]: %s\n", event.getAuthor().getName(), event.getMessage().getContentDisplay());
107  *     }
108  * }
109  * </code></pre>
110  *
111  * @see net.dv8tion.jda.core.hooks.EventListener EventListener
112  * @see net.dv8tion.jda.core.hooks.InterfacedEventManager InterfacedEventManager
113  */

503     else if (event instanceof DisconnectEvent) onDisconnect((DisconnectEvent) event);
504     else if (event instanceof ShutdownEvent) onShutdown((ShutdownEvent) event);
505     else if (event instanceof StatusChangeEvent) onStatusChange((StatusChangeEvent) event);
