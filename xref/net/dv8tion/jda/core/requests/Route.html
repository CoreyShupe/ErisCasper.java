310   private final String route;
311   private final String ratelimitRoute;
312   private final String compilableRoute;
313   private final int paramCount;
314   private final Method method;
315   private final List<Integer> majorParamIndexes = new ArrayList<>();
316   private final RateLimit ratelimit;
317 
318   private Route(Method method, String route, String... majorParameters) {
319     this(method, null, route, majorParameters);
320   }
321 
322   private Route(Method method, RateLimit rateLimit, String route, String... majorParameters) {
323     this.method = method;
324     this.ratelimit = rateLimit;
325     this.route = route;
326     this.paramCount = Helpers.countMatches(route, '{'); // All parameters start with {
327 
328     if (paramCount != Helpers.countMatches(route, '}'))
329       throw new IllegalArgumentException(
330           "An argument does not have both {}'s for route: " + method + "  " + route);
331 
332     // Create a String.format compilable route for parameter compiling.
333     compilableRoute = route.replaceAll("\\{.*?\\}", "%s");
334 
335     // If this route has major parameters that are unique markers for the ratelimit route, then we
336     // need to
337     // create a ratelimit compilable route. This goes through and replaces the parameters specified
338     // by majorParameters
339     // and records the parameter index so that when we compile it later we can select the proper
340     // parameters
341     // from the ones provided to make sure we inject in the proper indexes.
342     if (majorParameters.length != 0) {
343       int paramIndex = 0;
344       String replaceRoute = route;
345       Pattern keyP = Pattern.compile("\\{(.*?)\\}");
346       Matcher keyM = keyP.matcher(route);
347       // Search the route for all parameters
348       while (keyM.find()) {
349         String param = keyM.group(1);
350         // Attempt to match the found parameter with any of our majorParameters
351         for (String majorParam : majorParameters) {
352           // If the parameter is a major parameter, replace it with a string token and record its
353           // parameter index for later ratelimitRoute compiling.
354           if (param.equals(majorParam)) {
355             replaceRoute = replaceRoute.replace(keyM.group(0), "%s");
356             majorParamIndexes.add(paramIndex);
357           }
358         }
359         paramIndex++;
360       }
361       ratelimitRoute = replaceRoute;
362     } else ratelimitRoute = route;
363   }

389   public CompiledRoute compile(String... params) {
390     if (params.length != paramCount) {
391       throw new IllegalArgumentException(
392           "Error Compiling Route: ["
393               + route
394               + "], incorrect amount of parameters provided."
395               + "Expected: "
396               + paramCount
397               + ", Provided: "
398               + params.length);
399     }
400 
401     // Compile the route for interfacing with discord.
402     String compiledRoute = String.format(compilableRoute, (Object[]) params);
403     String compiledRatelimitRoute = ratelimitRoute;
404 
405     // If this route has major parameters which help to uniquely distinguish it from others of this
406     // route type then
407     // compile it using the major parameter indexes we discovered in the constructor.
408     if (!majorParamIndexes.isEmpty()) {
409 
410       String[] majorParams = new String[majorParamIndexes.size()];
411       for (int i = 0; i < majorParams.length; i++) {
412         majorParams[i] = params[majorParamIndexes.get(i)];
413       }
414       compiledRatelimitRoute = String.format(compiledRatelimitRoute, (Object[]) majorParams);
415     }
416 
417     return new CompiledRoute(this, compiledRatelimitRoute, compiledRoute);
418   }

438   public class CompiledRoute {
439     private final Route baseRoute;
440     private final String ratelimitRoute;
441     private final String compiledRoute;
442     private final boolean hasQueryParams;
443 
444     private CompiledRoute(
445         Route baseRoute, String ratelimitRoute, String compiledRoute, boolean hasQueryParams) {
446       this.baseRoute = baseRoute;
447       this.ratelimitRoute = ratelimitRoute;
448       this.compiledRoute = compiledRoute;
449       this.hasQueryParams = hasQueryParams;
450     }
451 
452     private CompiledRoute(Route baseRoute, String ratelimitRoute, String compiledRoute) {
453       this(baseRoute, ratelimitRoute, compiledRoute, false);
454     }
455 
456     public CompiledRoute withQueryParams(String... params) {
457       Checks.check(params.length >= 2, "params length must be at least 2");
458       Checks.check(params.length % 2 == 0, "params length must be a multiple of 2");
459 
460       StringBuilder newRoute = new StringBuilder(compiledRoute);
461 
462       for (int i = 0; i < params.length; i++)
463         newRoute
464             .append(!hasQueryParams && i == 0 ? '?' : '&')
465             .append(params[i])
466             .append('=')
467             .append(params[++i]);
468 
469       return new CompiledRoute(baseRoute, ratelimitRoute, newRoute.toString(), true);
470     }
471 
472     public String getRatelimitRoute() {
473       return ratelimitRoute;
474     }
475 
476     public String getCompiledRoute() {
477       return compiledRoute;
478     }
479 
480     public Route getBaseRoute() {
481       return baseRoute;
482     }
483 
484     public Method getMethod() {
485       return baseRoute.method;
486     }
487 
488     @Override
489     public int hashCode() {
490       return (compiledRoute + method.toString()).hashCode();
491     }
492 
493     @Override
494     public boolean equals(Object o) {
495       if (!(o instanceof CompiledRoute)) return false;
496 
497       CompiledRoute oCompiled = (CompiledRoute) o;
498 
499       return baseRoute.equals(oCompiled.getBaseRoute())
500           && compiledRoute.equals(oCompiled.compiledRoute);
501     }
502 
503     @Override
504     public String toString() {
505       return "CompiledRoute(" + method + ": " + compiledRoute + ")";
506     }
507   }
