73    SERVER_ERROR(0, "Discord encountered an internal server error! Not good!");
74  
75    private final int code;
76    private final String meaning;
77  
78    ErrorResponse(int code, String meaning) {
79      this.code = code;
80      this.meaning = meaning;
81    }
82  
83    public int getCode() {
84      return code;
85    }
86  
87    public String getMeaning() {
88      return meaning;
89    }
90  
91    public static ErrorResponse fromCode(int code) {
92      for (ErrorResponse error : values()) {
93        if (code == error.getCode()) return error;
94      }
95      return SERVER_ERROR;
96    }
97  
98    public static ErrorResponse fromJSON(JSONObject obj) {
99      if (obj == null || obj.isNull("code")) return SERVER_ERROR;
100     else {
101       return ErrorResponse.fromCode(obj.getInt("code"));
102     }
103   }
