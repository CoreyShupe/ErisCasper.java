110   public WebSocketClient(JDAImpl api) {
111     this.api = api;
112     this.shardInfo = api.getShardInfo();
113     this.shouldReconnect = api.isAutoReconnect();
114     this.connectNode = new StartingNode();
115     try {
116       api.getSessionController().appendSession(connectNode);
117     } catch (RuntimeException | Error e) {
118       LOG.error("Failed to append new session to session controller queue. Shutting down!", e);
119       this.api.setStatus(JDA.Status.SHUTDOWN);
120       this.api.getEventManager().handle(new ShutdownEvent(api, OffsetDateTime.now(), 1006));
121       if (e instanceof RuntimeException) throw (RuntimeException) e;
122       else throw (Error) e;
123     }
124   }

161   public void ready() {
162     if (initiating) {
163       initiating = false;
164       processingReady = false;
165       if (firstInit) {
166         firstInit = false;
167         if (api.getGuilds().size() >= 2000) // Show large warning when connected to >=2000 guilds
168         {
169           JDAImpl.LOG.warn(" __      __ _    ___  _  _  ___  _  _   ___  _ ");
170           JDAImpl.LOG.warn(" \\ \\    / //_\\  | _ \\| \\| ||_ _|| \\| | / __|| |");
171           JDAImpl.LOG.warn("  \\ \\/\\/ // _ \\ |   /| .` | | | | .` || (_ ||_|");
172           JDAImpl.LOG.warn("   \\_/\\_//_/ \\_\\|_|_\\|_|\\_||___||_|\\_| \\___|(_)");
173           JDAImpl.LOG.warn("You're running a session with over 2000 connected");
174           JDAImpl.LOG.warn("guilds. You should shard the connection in order");
175           JDAImpl.LOG.warn("to split the load or things like resuming");
176           JDAImpl.LOG.warn("connection might not work as expected.");
177           JDAImpl.LOG.warn("For more info see https://git.io/vrFWP");
178         }
179         JDAImpl.LOG.info("Finished Loading!");
180         api.getEventManager().handle(new ReadyEvent(api, api.getResponseTotal()));
181       } else {
182         updateAudioManagerReferences();
183         JDAImpl.LOG.info("Finished (Re)Loading!");
184         api.getEventManager().handle(new ReconnectedEvent(api, api.getResponseTotal()));
185       }
186     } else {
187       JDAImpl.LOG.info("Successfully resumed Session!");
188       api.getEventManager().handle(new ResumedEvent(api, api.getResponseTotal()));
189     }
190     api.setStatus(JDA.Status.CONNECTED);
191     LOG.debug("Resending {} cached events...", cachedEvents.size());
192     handle(cachedEvents);
193     LOG.debug("Sending of cached events finished.");
194     cachedEvents.clear();
195   }

213   private boolean send(String message, boolean skipQueue) {
214     if (!connected) return false;
215 
216     long now = System.currentTimeMillis();
217 
218     if (this.ratelimitResetTime <= now) {
219       this.messagesSent = 0;
220       this.ratelimitResetTime = now + 60000; // 60 seconds
221       this.printedRateLimitMessage = false;
222     }
223 
224     // Allows 115 messages to be sent before limiting.
225     if (this.messagesSent <= 115
226         || (skipQueue
227             && this.messagesSent
228                 <= 119)) // technically we could go to 120, but we aren't going to chance it
229     {
230       LOG.trace("<- {}", message);
231       socket.sendText(message);
232       this.messagesSent++;
233       return true;
234     } else {
235       if (!printedRateLimitMessage) {
236         LOG.warn(
237             "Hit the WebSocket RateLimit! If you see this message a lot then you might need to talk to DV8FromTheWorld.");
238         printedRateLimitMessage = true;
239       }
240       return false;
241     }
242   }

244   private void setupSendingThread() {
245     ratelimitThread =
246         new Thread(
247             () -> {
248               if (api.getContextMap() != null) MDC.setContextMap(api.getContextMap());
249               boolean needRatelimit;
250               boolean attemptedToSend;
251               while (!Thread.currentThread().isInterrupted()) {
252                 try {
253                   // Make sure that we don't send any packets before sending auth info.
254                   if (!sentAuthInfo) {
255                     Thread.sleep(500);
256                     continue;
257                   }
258                   attemptedToSend = false;
259                   needRatelimit = false;
260                   audioQueueLock.lockInterruptibly();
261 
262                   ConnectionRequest audioRequest = getNextAudioConnectRequest();
263                   String chunkOrSyncRequest = chunkSyncQueue.peekFirst();
264 
265                   // if lock isn't needed we already unlock here
266                   if (audioRequest == null || chunkOrSyncRequest != null) maybeUnlock();
267                   if (chunkOrSyncRequest != null) {
268                     needRatelimit = !send(chunkOrSyncRequest, false);
269                     if (!needRatelimit) chunkSyncQueue.removeFirst();
270 
271                     attemptedToSend = true;
272                   } else if (audioRequest != null) {
273                     VoiceChannel channel = audioRequest.getChannel();
274                     Guild guild = api.getGuildById(audioRequest.getGuildIdLong());
275                     if (guild == null) {
276                       // race condition on guild delete, avoid NPE on DISCONNECT requests
277                       queuedAudioConnections.remove(audioRequest.getGuildIdLong());
278 
279                       maybeUnlock();
280                       continue;
281                     }
282                     ConnectionStage stage = audioRequest.getStage();
283                     AudioManager audioManager = guild.getAudioManager();
284                     JSONObject packet;
285                     switch (stage) {
286                       case RECONNECT:
287                       case DISCONNECT:
288                         packet = newVoiceClose(audioRequest.getGuildIdLong());
289                         break;
290                       default:
291                       case CONNECT:
292                         packet = newVoiceOpen(audioManager, channel);
293                     }
294                     needRatelimit = !send(packet.toString(), false);
295                     if (!needRatelimit) {
296                       // If we didn't get RateLimited, Next request attempt will be 2 seconds from
297                       // now
298                       // we remove it in VoiceStateUpdateHandler once we hear that it has updated
299                       // our status
300                       // in 2 seconds we will attempt again in case we did not receive an update
301                       audioRequest.setNextAttemptEpoch(System.currentTimeMillis() + 2000);
302                       // If we are already in the correct state according to voice state
303                       // we will not receive a VOICE_STATE_UPDATE that would remove it
304                       // thus we update it here
305                       final GuildVoiceState voiceState = guild.getSelfMember().getVoiceState();
306                       updateAudioConnection0(guild.getIdLong(), voiceState.getChannel());
307                     }
308                     maybeUnlock();
309                     attemptedToSend = true;
310                   } else {
311                     String message = ratelimitQueue.peekFirst();
312                     if (message != null) {
313                       needRatelimit = !send(message, false);
314                       if (!needRatelimit) ratelimitQueue.removeFirst();
315                       attemptedToSend = true;
316                     }
317                   }
318 
319                   if (needRatelimit || !attemptedToSend) Thread.sleep(1000);
320                 } catch (InterruptedException ignored) {
321                   LOG.debug(
322                       "Main WS send thread interrupted. Most likely JDA is disconnecting the websocket.");
323                   break;
324                 } finally {
325                   // on any exception that might cause this lock to not release
326                   maybeUnlock();
327                 }
328               }
329             });
330     ratelimitThread.setUncaughtExceptionHandler(
331         (thread, throwable) -> {
332           handleCallbackError(socket, throwable);
333           setupSendingThread();
334         });
335     ratelimitThread.setName(api.getIdentifierString() + " MainWS-Sending Thread");
336     ratelimitThread.start();
337   }

386   protected synchronized void connect() {
387     if (api.getStatus() != JDA.Status.ATTEMPTING_TO_RECONNECT)
388       api.setStatus(JDA.Status.CONNECTING_TO_WEBSOCKET);
389     if (shutdown) throw new RejectedExecutionException("JDA is shutdown!");
390     initiating = true;
391 
392     String url =
393         api.getGatewayUrl() + "?encoding=json&compress=zlib-stream&v=" + DISCORD_GATEWAY_VERSION;
394 
395     try {
396       socket =
397           api.getWebSocketFactory()
398               .createSocket(url)
399               .addHeader("Accept-Encoding", "gzip")
400               .addListener(this);
401       socket.connect();
402     } catch (IOException | WebSocketException e) {
403       // Completely fail here. We couldn't make the connection.
404       throw new IllegalStateException(e);
405     }
406   }

418   @Override
419   public void onConnected(WebSocket websocket, Map<String, List<String>> headers) {
420     // writing thread
421     if (api.getContextMap() != null) MDC.setContextMap(api.getContextMap());
422     api.setStatus(JDA.Status.IDENTIFYING_SESSION);
423     LOG.info("Connected to WebSocket");
424     if (headers.containsKey("cf-ray")) {
425       List<String> values = headers.get("cf-ray");
426       if (!values.isEmpty()) {
427         String ray = values.get(0);
428         cfRays.add(ray);
429         LOG.debug("Received new CF-RAY: {}", ray);
430       }
431     }
432     connected = true;
433     reconnectTimeoutS = 2;
434     messagesSent = 0;
435     ratelimitResetTime = System.currentTimeMillis() + 60000;
436     if (sessionId == null) sendIdentify();
437     else sendResume();
438   }

440   @Override
441   public void onDisconnected(
442       WebSocket websocket,
443       WebSocketFrame serverCloseFrame,
444       WebSocketFrame clientCloseFrame,
445       boolean closedByServer) {
446     sentAuthInfo = false;
447     connected = false;
448     api.setStatus(JDA.Status.DISCONNECTED);
449 
450     CloseCode closeCode = null;
451     int rawCloseCode = 1000;
452     // When we get 1000 from remote close we will try to resume
453     // as apparently discord doesn't understand what "graceful disconnect" means
454     boolean isInvalidate = false;
455 
456     if (keepAliveThread != null) {
457       keepAliveThread.interrupt();
458       keepAliveThread = null;
459     }
460     if (serverCloseFrame != null) {
461       rawCloseCode = serverCloseFrame.getCloseCode();
462       closeCode = CloseCode.from(rawCloseCode);
463       if (closeCode == CloseCode.RATE_LIMITED)
464         LOG.error(
465             "WebSocket connection closed due to ratelimit! Sent more than 120 websocket messages in under 60 seconds!");
466       else if (closeCode != null) LOG.debug("WebSocket connection closed with code {}", closeCode);
467       else
468         LOG.warn(
469             "WebSocket connection closed with unknown meaning for close-code {}", rawCloseCode);
470     }
471     if (clientCloseFrame != null
472         && clientCloseFrame.getCloseCode() == 1000
473         && Objects.equals(clientCloseFrame.getCloseReason(), INVALIDATE_REASON)) {
474       // When we close with 1000 we properly dropped our session due to invalidation
475       // in that case we can be sure that resume will not work and instead we invalidate and
476       // reconnect here
477       isInvalidate = true;
478     }
479 
480     // null is considered -reconnectable- as we do not know the close-code meaning
481     boolean closeCodeIsReconnect = closeCode == null || closeCode.isReconnect();
482     if (!shouldReconnect || !closeCodeIsReconnect) // we should not reconnect
483     {
484       if (ratelimitThread != null) ratelimitThread.interrupt();
485 
486       if (!closeCodeIsReconnect) {
487         // it is possible that a token can be invalidated due to too many reconnect attempts
488         // or that a bot reached a new shard minimum and cannot connect with the current settings
489         // if that is the case we have to drop our connection and inform the user with a fatal error
490         // message
491         LOG.error(
492             "WebSocket connection was closed and cannot be recovered due to identification issues\n{}",
493             closeCode);
494       }
495 
496       api.setStatus(JDA.Status.SHUTDOWN);
497       api.getEventManager().handle(new ShutdownEvent(api, OffsetDateTime.now(), rawCloseCode));
498     } else {
499       // reset our zlib decompression tools
500       zlibContext = new Inflater();
501       readBuffer = null;
502       if (isInvalidate) invalidate(); // 1000 means our session is dropped so we cannot resume
503       api.getEventManager()
504           .handle(
505               new DisconnectEvent(
506                   api, serverCloseFrame, clientCloseFrame, closedByServer, OffsetDateTime.now()));
507       try {
508         if (sessionId == null) queueReconnect();
509         else // if resume is possible
510         reconnect();
511       } catch (InterruptedException e) {
512         LOG.error("Failed to resume due to interrupted thread", e);
513         invalidate();
514         queueReconnect();
515       }
516     }
517   }

519   protected void queueReconnect() {
520     if (!handleIdentifyRateLimit)
521       LOG.warn(
522           "Got disconnected from WebSocket (Internet?!)... Appending session to reconnect queue");
523     try {
524       this.api.setStatus(JDA.Status.RECONNECT_QUEUED);
525       this.connectNode = new ReconnectNode();
526       this.api.getSessionController().appendSession(connectNode);
527     } catch (IllegalStateException ex) {
528       LOG.error("Reconnect queue rejected session. Shutting down...");
529       this.api.setStatus(JDA.Status.SHUTDOWN);
530       this.api.getEventManager().handle(new ShutdownEvent(api, OffsetDateTime.now(), 1006));
531     }
532   }

538   /**
539    * This method is used to start the reconnect of the JDA instance. It is public for access from
540    * SessionReconnectQueue extensions.
541    *
542    * @param callFromQueue whether this was in SessionReconnectQueue and got polled
543    * @param shouldHandleIdentify whether SessionReconnectQueue already handled an IDENTIFY rate
544    *     limit for this session
545    */
546   public void reconnect(boolean callFromQueue, boolean shouldHandleIdentify)
547       throws InterruptedException {
548     if (callFromQueue && api.getContextMap() != null) api.getContextMap().forEach(MDC::put);
549     if (shutdown) {
550       api.setStatus(JDA.Status.SHUTDOWN);
551       api.getEventManager().handle(new ShutdownEvent(api, OffsetDateTime.now(), 1000));
552       return;
553     }
554     if (!handleIdentifyRateLimit) {
555       if (callFromQueue)
556         LOG.warn(
557             "Queue is attempting to reconnect a shard...{}",
558             JDALogger.getLazyString(
559                 () -> shardInfo != null ? " Shard: " + shardInfo.getShardString() : ""));
560       else LOG.warn("Got disconnected from WebSocket (Internet?!)...");
561       LOG.warn("Attempting to reconnect in {}s", reconnectTimeoutS);
562     }
563     while (shouldReconnect) {
564       api.setStatus(JDA.Status.WAITING_TO_RECONNECT);
565       int delay = IDENTIFY_DELAY;
566       if (handleIdentifyRateLimit && shouldHandleIdentify)
567         LOG.error(
568             "Encountered IDENTIFY (OP {}) Rate Limit! Waiting {} seconds before trying again!",
569             WebSocketCode.IDENTIFY,
570             IDENTIFY_DELAY);
571       else delay = reconnectTimeoutS;
572       Thread.sleep(delay * 1000);
573       handleIdentifyRateLimit = false;
574       api.setStatus(JDA.Status.ATTEMPTING_TO_RECONNECT);
575       LOG.warn("Attempting to reconnect!");
576       try {
577         connect();
578         break;
579       } catch (RejectedExecutionException ex) {
580         // JDA has already been shutdown so we can stop here
581         api.setStatus(JDA.Status.SHUTDOWN);
582         api.getEventManager().handle(new ShutdownEvent(api, OffsetDateTime.now(), 1000));
583         return;
584       } catch (RuntimeException ex) {
585         reconnectTimeoutS = Math.min(reconnectTimeoutS << 1, api.getMaxReconnectDelay());
586         LOG.warn("Reconnect failed! Next attempt in {}s", reconnectTimeoutS);
587       }
588     }
589   }

591   @Override
592   public void onTextMessage(WebSocket websocket, String message) {
593     // reading thread
594     if (api.getContextMap() != null) MDC.setContextMap(api.getContextMap());
595     JSONObject content = new JSONObject(message);
596     try {
597       int opCode = content.getInt("op");
598 
599       if (!content.isNull("s")) {
600         api.setResponseTotal(content.getInt("s"));
601       }
602 
603       switch (opCode) {
604         case WebSocketCode.DISPATCH:
605           handleEvent(content);
606           break;
607         case WebSocketCode.HEARTBEAT:
608           LOG.debug("Got Keep-Alive request (OP 1). Sending response...");
609           sendKeepAlive();
610           break;
611         case WebSocketCode.RECONNECT:
612           LOG.debug("Got Reconnect request (OP 7). Closing connection now...");
613           close(4000, "OP 7: RECONNECT");
614           break;
615         case WebSocketCode.INVALIDATE_SESSION:
616           LOG.debug("Got Invalidate request (OP 9). Invalidating...");
617           sentAuthInfo = false;
618           final boolean isResume = content.getBoolean("d");
619           // When d: true we can wait a bit and then try to resume again
620           // sending 4000 to not drop session
621           int closeCode = isResume ? 4000 : 1000;
622           if (isResume)
623             LOG.debug("Session can be recovered... Closing and sending new RESUME request");
624           else invalidate();
625 
626           close(closeCode, INVALIDATE_REASON);
627           break;
628         case WebSocketCode.HELLO:
629           LOG.debug("Got HELLO packet (OP 10). Initializing keep-alive.");
630           final JSONObject data = content.getJSONObject("d");
631           setupKeepAlive(data.getLong("heartbeat_interval"));
632           if (!data.isNull("_trace"))
633             updateTraces(data.getJSONArray("_trace"), "HELLO", WebSocketCode.HELLO);
634           break;
635         case WebSocketCode.HEARTBEAT_ACK:
636           LOG.trace("Got Heartbeat Ack (OP 11).");
637           api.setPing(System.currentTimeMillis() - heartbeatStartTime);
638           break;
639         default:

763   protected void updateAudioManagerReferences() {
764     final TLongObjectMap<AudioManager> managerMap = api.getAudioManagerMap();
765     if (managerMap.size() > 0) LOG.trace("Updating AudioManager references");
766 
767     synchronized (managerMap) {

772         ConnectionListener listener = mng.getConnectionListener();
773 
774         GuildImpl guild = (GuildImpl) api.getGuildById(guildId);
775         if (guild == null) {

777           // null.
778           queuedAudioConnections.remove(guildId);
779           if (listener != null)
780             listener.onStatusChange(ConnectionStatus.DISCONNECTED_REMOVED_FROM_GUILD);
781           it.remove();
782         } else {

784           newMng.setSelfMuted(mng.isSelfMuted());
785           newMng.setSelfDeafened(mng.isSelfDeafened());
786           newMng.setQueueTimeout(mng.getConnectTimeout());
787           newMng.setSendingHandler(mng.getSendingHandler());
788           newMng.setReceivingHandler(mng.getReceiveHandler());
789           newMng.setConnectionListener(listener);
790           newMng.setAutoReconnect(mng.isAutoReconnect());
791 
792           if (mng.isConnected() || mng.isAttemptingToConnect()) {
793             final long channelId =
794                 mng.isConnected()
795                     ? mng.getConnectedChannel().getIdLong()
796                     : mng.getQueuedAudioConnection().getIdLong();
797 
798             final VoiceChannel channel = api.getVoiceChannelById(channelId);
799             if (channel != null) {
800               if (mng.isConnected()) mng.closeAudioConnection(ConnectionStatus.ERROR_CANNOT_RESUME);
801               // closing old connection in order to reconnect later
802               newMng.setQueuedAudioConnection(channel);
803             } else {
804               // The voice channel is not cached. It was probably deleted.
805               queuedAudioConnections.remove(guildId);
806               if (listener != null)
807                 listener.onStatusChange(ConnectionStatus.DISCONNECTED_CHANNEL_DELETED);
808             }
809           }
810           it.setValue(newMng);
811         }
812       }
813     }

837   protected void handleEvent(JSONObject raw) {
838     String type = raw.getString("t");
839     long responseTotal = api.getResponseTotal();
840 
841     if (type.equals("GUILD_MEMBER_ADD"))
842       ((GuildMembersChunkHandler) getHandler("GUILD_MEMBERS_CHUNK"))
843           .modifyExpectedGuildMember(raw.getJSONObject("d").getLong("guild_id"), 1);
844     if (type.equals("GUILD_MEMBER_REMOVE"))
845       ((GuildMembersChunkHandler) getHandler("GUILD_MEMBERS_CHUNK"))
846           .modifyExpectedGuildMember(raw.getJSONObject("d").getLong("guild_id"), -1);
847 
848     boolean isJSON = raw.opt("d") instanceof JSONObject;
849     // If initiating, only allows READY, RESUMED, GUILD_MEMBERS_CHUNK, GUILD_SYNC, and GUILD_CREATE
850     // through.
851     // If we are currently chunking, we don't allow GUILD_CREATE through anymore.
852     if (initiating
853         && !(type.equals("READY")
854             || type.equals("GUILD_MEMBERS_CHUNK")
855             || type.equals("RESUMED")
856             || type.equals("GUILD_SYNC")
857             || (!chunkingAndSyncing && type.equals("GUILD_CREATE")))) {
858       if (!isJSON) {
859         if (type.equals("PRESENCES_REPLACE")) {
860           List<JSONObject> converted =
861               convertPresencesReplace(responseTotal, raw.getJSONArray("d"));
862           LOG.debug("Caching PRESENCES_REPLACE event during init as PRESENCE_UPDATE dispatches!");
863           cachedEvents.addAll(converted);
864         } else {
865           LOG.debug("Received event with unhandled body type during init JSON: {}", raw);
866         }
867         return;
868       }
869       // If we are currently GuildStreaming, and we get a GUILD_DELETE informing us that a Guild is
870       // unavailable
871       // convert it to a GUILD_CREATE for handling.
872       JSONObject content = raw.getJSONObject("d");
873       if (!chunkingAndSyncing
874           && type.equals("GUILD_DELETE")
875           && content.has("unavailable")
876           && content.getBoolean("unavailable")) {
877         type = "GUILD_CREATE";
878         raw.put("t", "GUILD_CREATE")
879             .put(
880                 "jda-field",
881                 "This event was originally a GUILD_DELETE but was converted to GUILD_CREATE for WS init Guild streaming");
882       } else {
883         LOG.debug("Caching {} event during init!", type);
884         cachedEvents.add(raw);
885         return;
886       }
887     }
888 
889     if (!isJSON) {
890       // Needs special handling due to content of "d" being an array
891       if (type.equals("PRESENCES_REPLACE")) {
892         final JSONArray payload = raw.getJSONArray("d");
893         final List<JSONObject> converted = convertPresencesReplace(responseTotal, payload);
894         final PresenceUpdateHandler handler = getHandler("PRESENCE_UPDATE");
895         LOG.trace(String.format("%s -> %s", type, payload.toString()));
896         for (JSONObject o : converted) handler.handle(responseTotal, o);
897       } else {
898         LOG.debug("Received event with unhandled body type JSON: {}", raw);
899       }
900       return;
901     }
902 
903     JSONObject content = raw.getJSONObject("d");
904     LOG.trace("{} -> {}", type, content);
905 
906     try {
907       switch (type) {
908           // INIT types
909         case "READY":
910           api.setStatus(JDA.Status.LOADING_SUBSYSTEMS);
911           processingReady = true;
912           handleIdentifyRateLimit = false;
913           sessionId = content.getString("session_id");
914           if (!content.isNull("_trace"))
915             updateTraces(content.getJSONArray("_trace"), "READY", WebSocketCode.DISPATCH);
916           handlers.get("READY").handle(responseTotal, raw);
917           break;
918         case "RESUMED":
919           sentAuthInfo = true;
920           if (!processingReady) {
921             api.setStatus(JDA.Status.LOADING_SUBSYSTEMS);
922             initiating = false;
923             ready();
924           }
925           if (!content.isNull("_trace"))
926             updateTraces(content.getJSONArray("_trace"), "RESUMED", WebSocketCode.DISPATCH);
927           break;
928         default:
929           SocketHandler handler = handlers.get(type);
930           if (handler != null) handler.handle(responseTotal, raw);
931           else LOG.debug("Unrecognized event:\n{}", raw);
932       }
933     } catch (JSONException ex) {
934       LOG.warn(
935           "Got an unexpected Json-parse error. Please redirect following message to the devs:\n\t{}\n\t{} -> {}",
936           ex.getMessage(),
937           type,
938           content,
939           ex);
940     } catch (Exception ex) {
941       LOG.error(
942           "Got an unexpected error. Please redirect following message to the devs:\n\t{} -> {}",
943           type,
944           content,
945           ex);
946     }
947   }

949   protected boolean onBufferMessage(byte[] binary) throws IOException {
950     if (binary.length >= 4 && getInt(binary, binary.length - 4) == ZLIB_SUFFIX) {
951       extendBuffer(binary);
952       return true;
953     }
954 
955     if (readBuffer != null) extendBuffer(binary);
956     else allocateBuffer(binary);
957 
958     return false;
959   }

961   @Override
962   public void onBinaryMessage(WebSocket websocket, byte[] binary)
963       throws IOException, DataFormatException {
964     if (!onBufferMessage(binary)) return;
965     // Thanks to ShadowLordAlpha and Shredder121 for code and debugging.
966     // Get the compressed message and inflate it
967     final int size = readBuffer != null ? readBuffer.size() : binary.length;
968     ByteArrayOutputStream out = new ByteArrayOutputStream(size * 2);
969     try (InflaterOutputStream decompressor = new InflaterOutputStream(out, zlibContext)) {
970       if (readBuffer != null) readBuffer.writeTo(decompressor);
971       else decompressor.write(binary);
972       // send the inflated message to the TextMessage method
973       onTextMessage(websocket, out.toString("UTF-8"));
974     } catch (IOException e) {
975       throw (DataFormatException) new DataFormatException("Malformed").initCause(e);
976     } finally {
977       readBuffer = null;
978     }
979   }

1125   public ConnectionRequest updateAudioConnection0(long guildId, VoiceChannel connectedChannel) {
1126     // Called by VoiceStateUpdateHandler when we receive a response from discord
1127     // about our request to CONNECT or DISCONNECT.
1128     // "stage" should never be RECONNECT here thus we don't check for that case
1129     ConnectionRequest request = queuedAudioConnections.get(guildId);
1130 
1131     if (request == null) return null;
1132     ConnectionStage requestStage = request.getStage();
1133     if (connectedChannel == null) {
1134       // If we got an update that DISCONNECT happened
1135       // -> If it was on RECONNECT we now switch to CONNECT
1136       // -> If it was on DISCONNECT we can now remove it
1137       // -> Otherwise we ignore it
1138       switch (requestStage) {
1139         case DISCONNECT:
1140           return queuedAudioConnections.remove(guildId);
1141         case RECONNECT:
1142           request.setStage(ConnectionStage.CONNECT);
1143           request.setNextAttemptEpoch(System.currentTimeMillis());
1144         default:
1145           return null;
1146       }
1147     } else if (requestStage == ConnectionStage.CONNECT) {
1148       // If the removeRequest was related to a channel that isn't the currently queued
1149       // request, then don't remove it.
1150       if (request.getChannel().getIdLong() == connectedChannel.getIdLong())
1151         return queuedAudioConnections.remove(guildId);
1152     }
1153     // If the channel is not the one we are looking for!
1154     return null;
1155   }

1162   protected ConnectionRequest getNextAudioConnectRequest() {
1163     // Don't try to setup audio connections before JDA has finished loading.
1164     if (!isReady()) return null;
1165 
1166     long now = System.currentTimeMillis();
1167     TLongObjectIterator<ConnectionRequest> it = queuedAudioConnections.iterator();
1168     while (it.hasNext()) {
1169       it.advance();
1170       ConnectionRequest audioRequest = it.value();
1171       if (audioRequest.getNextAttemptEpoch() < now) {
1172         Guild guild = api.getGuildById(audioRequest.getGuildIdLong());
1173         if (guild == null) {
1174           it.remove();
1175           // if (listener != null)
1176           //    listener.onStatusChange(ConnectionStatus.DISCONNECTED_REMOVED_FROM_GUILD);
1177           // already handled by event handling
1178           continue;
1179         }
1180 
1181         ConnectionListener listener = guild.getAudioManager().getConnectionListener();
1182         if (audioRequest.getStage() != ConnectionStage.DISCONNECT) {
1183           VoiceChannel channel = guild.getVoiceChannelById(audioRequest.getChannel().getIdLong());
1184           if (channel == null) {
1185             it.remove();
1186             if (listener != null)
1187               listener.onStatusChange(ConnectionStatus.DISCONNECTED_CHANNEL_DELETED);
1188             continue;
1189           }
1190 
1191           if (!guild.getSelfMember().hasPermission(channel, Permission.VOICE_CONNECT)) {
1192             it.remove();
1193             if (listener != null)
1194               listener.onStatusChange(ConnectionStatus.DISCONNECTED_LOST_PERMISSION);
1195             continue;
1196           }
1197         }
1198 
1199         return audioRequest;
1200       }
1201     }
1202 
1203     return null;
1204   }
