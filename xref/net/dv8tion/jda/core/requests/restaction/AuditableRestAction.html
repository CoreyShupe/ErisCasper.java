19  import java.util.function.BooleanSupplier;
20  import java.util.function.Consumer;
21  import net.dv8tion.jda.core.JDA;
22  import net.dv8tion.jda.core.requests.*;
23  import net.dv8tion.jda.core.utils.MiscUtil;

25  import org.apache.commons.collections4.map.CaseInsensitiveMap;

28  /**
29   * Extension of RestAction to allow setting a reason, only available to accounts of {@link
30   * net.dv8tion.jda.core.AccountType#BOT AccounType.BOT}
31   *
32   * @param  <T> The return type
33   * @since 3.3.0
34   */
35  public abstract class AuditableRestAction<T> extends RestAction<T> {
36  
37    protected String reason = null;
38  
39    public AuditableRestAction(JDA api, Route.CompiledRoute route) {
40      super(api, route);
41    }
42  
43    public AuditableRestAction(JDA api, Route.CompiledRoute route, RequestBody data) {
44      super(api, route, data);
45    }
46  
47    public AuditableRestAction(JDA api, Route.CompiledRoute route, JSONObject data) {
48      super(api, route, data);
49    }
50  
51    @Override
52    public AuditableRestAction<T> setCheck(BooleanSupplier checks) {
53      return (AuditableRestAction) super.setCheck(checks);
54    }
55  
56    /**
57     * Applies the specified reason as audit-log reason field. <br>
58     * When the provided reason is empty or {@code null} it will be treated as not set.
59     *
60     * <p>Reasons for any AuditableRestAction may be retrieved via {@link
61     * net.dv8tion.jda.core.audit.AuditLogEntry#getReason() AuditLogEntry.getReason()} in iterable
62     * {@link net.dv8tion.jda.core.requests.restaction.pagination.AuditLogPaginationAction
63     * AuditLogPaginationActions} from {@link net.dv8tion.jda.core.entities.Guild#getAuditLogs()
64     * Guild.getAuditLogs()}!
65     *
66     * <p>This will specify the reason via the {@code X-Audit-Log-Reason} Request Header. <br>
67     * <b>Note: This may not be available to accounts for {@link
68     * net.dv8tion.jda.core.AccountType#CLIENT AccountType.CLIENT}. <br>
69     * Using methods with a reason parameter will always work and <u>override</u> this header.</b>
70     * (ct. {@link
71     * net.dv8tion.jda.core.managers.GuildController#ban(net.dv8tion.jda.core.entities.User, int,
72     * String) GuildController.ban(User, int, String)})
73     *
74     * @param reason The reason for this action which should be logged in the Guild's AuditLogs
75     * @return The current AuditableRestAction instance for chaining convenience
76     */
77    public AuditableRestAction<T> reason(String reason) {
78      this.reason = reason;
79      return this;
80    }
81  
82    @Override
83    protected CaseInsensitiveMap<String, String> finalizeHeaders() {
84      CaseInsensitiveMap<String, String> headers = super.finalizeHeaders();
85  
86      if (reason == null || reason.isEmpty()) return headers;
87  
88      if (headers == null) headers = new CaseInsensitiveMap<>();
89  
90      headers.put("X-Audit-Log-Reason", uriEncode(reason));
91  
92      return headers;
93    }
94  
95    private String uriEncode(String input) {
96      String formEncode = MiscUtil.encodeUTF8(input);
97      return formEncode.replace('+', ' ');
98    }

100   /**
101    * Specialized form of {@link net.dv8tion.jda.core.requests.restaction.AuditableRestAction
102    * AuditableRestAction} that is used to provide information that has already been retrieved or
103    * generated so that another request does not need to be made to Discord. <br>
104    * Basically: Allows you to provide a value directly to the success returns.
105    *
106    * @param <T> The generic response type for this RestAction
107    */
108   public static class EmptyRestAction<T> extends AuditableRestAction<T> {
109     protected final T content;
110 
111     public EmptyRestAction(JDA api) {
112       this(api, null);
113     }
114 
115     public EmptyRestAction(JDA api, T content) {
116       super(api, null);
117       this.content = content;
118     }
119 
120     @Override
121     public void queue(Consumer<T> success, Consumer<Throwable> failure) {
122       if (success != null) success.accept(content);
123     }
124 
125     @Override
126     public RequestFuture<T> submit(boolean shouldQueue) {
127       return new RestFuture<>(content);
128     }
129 
130     @Override
131     protected void handleResponse(Response response, Request<T> request) {}
132   }
133 }
