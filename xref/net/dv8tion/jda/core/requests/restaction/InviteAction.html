17  package net.dv8tion.jda.core.requests.restaction;
18  
19  import java.util.concurrent.TimeUnit;
20  import java.util.function.BooleanSupplier;
21  import net.dv8tion.jda.core.JDA;
22  import net.dv8tion.jda.core.entities.Invite;
23  import net.dv8tion.jda.core.requests.Request;
24  import net.dv8tion.jda.core.requests.Response;
25  import net.dv8tion.jda.core.requests.Route;
26  import net.dv8tion.jda.core.utils.Checks;
27  import okhttp3.RequestBody;
28  import org.json.JSONObject;
29  
30  /**
31   * {@link net.dv8tion.jda.core.entities.Invite Invite} Builder system created as an extension of
32   * {@link net.dv8tion.jda.core.requests.RestAction} <br>
33   * Provides an easy way to gather and deliver information to Discord to create {@link
34   * net.dv8tion.jda.core.entities.Invite Invites}.
35   */
36  public class InviteAction extends AuditableRestAction<Invite> {
37    private Integer maxAge = null;
38    private Integer maxUses = null;
39    private Boolean temporary = null;
40    private Boolean unique = null;
41  
42    public InviteAction(final JDA api, final String channelId) {
43      super(api, Route.Invites.CREATE_INVITE.compile(channelId));
44    }
45  
46    @Override
47    public InviteAction setCheck(BooleanSupplier checks) {
48      return (InviteAction) super.setCheck(checks);
49    }
50  
51    @Override
52    protected RequestBody finalizeData() {
53      final JSONObject object = new JSONObject();
54  
55      if (this.maxAge != null) object.put("max_age", (int) this.maxAge);
56      if (this.maxUses != null) object.put("max_uses", (int) this.maxUses);
57      if (this.temporary != null) object.put("temporary", (boolean) this.temporary);
58      if (this.unique != null) object.put("unique", (boolean) this.unique);
59  
60      return getRequestBody(object);
61    }
62  
63    @Override
64    protected void handleResponse(final Response response, final Request<Invite> request) {
65      if (response.isOk())
66        request.onSuccess(this.api.getEntityBuilder().createInvite(response.getObject()));
67      else request.onFailure(response);
68    }
69  
70    /**
71     * Sets the max age in seconds for the invite. Set this to {@code 0} if the invite should never
72     * expire. Default is {@code 86400} (24 hours). {@code null} will reset this to the default value.
73     *
74     * @param maxAge The max age for this invite or {@code null} to use the default value.
75     * @throws IllegalArgumentException If maxAge is negative.
76     * @return The current InviteAction for chaining.
77     */
78    public final InviteAction setMaxAge(final Integer maxAge) {
79      if (maxAge != null) Checks.notNegative(maxAge, "maxAge");
80  
81      this.maxAge = maxAge;
82      return this;
83    }
84  
85    /**
86     * Sets the max age for the invite. Set this to {@code 0} if the invite should never expire.
87     * Default is {@code 86400} (24 hours). {@code null} will reset this to the default value.
88     *
89     * @param maxAge The max age for this invite or {@code null} to use the default value.
90     * @param timeUnit The {@link java.util.concurrent.TimeUnit TimeUnit} type of {@code maxAge}.
91     * @throws IllegalArgumentException If maxAge is negative or maxAge is positive and timeUnit is
92     *     null.
93     * @return The current InviteAction for chaining.
94     */
95    public final InviteAction setMaxAge(final Long maxAge, final TimeUnit timeUnit) {
96      if (maxAge == null) return this.setMaxAge(null);
97  
98      Checks.notNegative(maxAge, "maxAge");
99      Checks.notNull(timeUnit, "timeUnit");
100 
101     return this.setMaxAge(Math.toIntExact(timeUnit.toSeconds(maxAge)));
102   }
103 
104   /**
105    * Sets the max uses for the invite. Set this to {@code 0} if the invite should have unlimited
106    * uses. Default is {@code 0}. {@code null} will reset this to the default value.
107    *
108    * @param maxUses The max uses for this invite or {@code null} to use the default value.
109    * @throws IllegalArgumentException If maxUses is negative.
110    * @return The current InviteAction for chaining.
111    */
112   public final InviteAction setMaxUses(final Integer maxUses) {
113     if (maxUses != null) Checks.notNegative(maxUses, "maxUses");
114 
115     this.maxUses = maxUses;
116     return this;
117   }
118 
119   /**
120    * Sets whether the invite should only grant temporary membership. Default is {@code false}.
121    *
122    * @param temporary Whether the invite should only grant temporary membership or {@code null} to
123    *     use the default value.
124    * @return The current InviteAction for chaining.
125    */
126   public final InviteAction setTemporary(final Boolean temporary) {
127     this.temporary = temporary;
128     return this;
129   }
130 
131   /**
132    * Sets whether discord should reuse a similar invite. Default is {@code false}.
133    *
134    * @param unique Whether discord should reuse a similar invite or {@code null} to use the default
135    *     value.
136    * @return The current InviteAction for chaining.
137    */
138   public final InviteAction setUnique(final Boolean unique) {
139     this.unique = unique;
140     return this;
141   }
142 }
