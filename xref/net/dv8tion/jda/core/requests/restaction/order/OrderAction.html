17  package net.dv8tion.jda.core.requests.restaction.order;

19  import java.util.ArrayList;
20  import java.util.Collections;
21  import java.util.Comparator;
22  import java.util.List;
23  import java.util.function.BooleanSupplier;
24  import net.dv8tion.jda.core.JDA;
25  import net.dv8tion.jda.core.requests.Request;
26  import net.dv8tion.jda.core.requests.Response;
27  import net.dv8tion.jda.core.requests.RestAction;
28  import net.dv8tion.jda.core.requests.Route;
29  import net.dv8tion.jda.core.utils.Checks;

31  /**
32   * Extension of {@link net.dv8tion.jda.core.requests.RestAction RestAction} - Type: Void that allows
33   * to modify the order of entities provided as an {@link java.util.ArrayList ArrayList}. <br>
34   * This action contains a List or entities for the specified type {@code T} which can be moved
35   * within the bounds but not removed, nor can any new entities be added.
36   *
37   * @param <T> The entity type for the {@link java.util.List List} of entities contained in the
38   *     OrderAction's orderList
39   * @param <M> The extension implementing the abstract operations of this OrderAction, this will be
40   *     important for chaining convenience as it returns the specific implementation rather than a
41   *     mask of this class. It allows us to implement chaining operations in this class instead of
42   *     having to implement it in every inheriting class!
43   * @since 3.0
44   */
45  public abstract class OrderAction<T, M extends OrderAction<T, M>> extends RestAction<Void> {
46    protected final List<T> orderList;
47    protected final boolean ascendingOrder;
48    protected int selectedPosition = -1;
49  
50    /**
51     * Creates a new OrderAction instance
52     *
53     * @param api JDA instance which is associated with the entities contained in the order list
54     * @param route The {@link net.dv8tion.jda.core.requests.Route.CompiledRoute CompiledRoute} which
55     *     is provided to the {@link RestAction#RestAction(JDA, Route.CompiledRoute,
56     *     okhttp3.RequestBody) RestAction Constructor}
57     */
58    public OrderAction(JDA api, Route.CompiledRoute route) {
59      this(api, true, route);
60    }
61  
62    /**
63     * Creates a new OrderAction instance
64     *
65     * @param api JDA instance which is associated with the entities contained in the order list
66     * @param ascendingOrder Whether or not the order of items should be ascending
67     * @param route The {@link net.dv8tion.jda.core.requests.Route.CompiledRoute CompiledRoute} which
68     *     is provided to the {@link RestAction#RestAction(JDA, Route.CompiledRoute,
69     *     okhttp3.RequestBody) RestAction Constructor}
70     */
71    public OrderAction(JDA api, boolean ascendingOrder, Route.CompiledRoute route) {
72      super(api, route);
73      this.orderList = new ArrayList<>();
74      this.ascendingOrder = ascendingOrder;
75    }
76  
77    @Override
78    public M setCheck(BooleanSupplier checks) {
79      return (M) super.setCheck(checks);
80    }
81  
82    /**
83     * Immutable List representing the currently selected order of entities in this OrderAction
84     * instance
85     *
86     * @return Immutable List representing the current order
87     */
88    public List<T> getCurrentOrder() {
89      return Collections.unmodifiableList(orderList);
90    }
91  
92    /**
93     * Selects a new current entity at the specified index <br>
94     * This index is in correlation to the {@link #getCurrentOrder() current order}
95     *
96     * @param selectedPosition The index for the new position that will be in focus for all
97     *     modification operations
98     * @throws java.lang.IllegalArgumentException If the provided position is out-of-bounds
99     * @return The current OrderAction sub-implementation instance
100    * @see #getSelectedPosition()
101    * @see #getSelectedEntity()
102    */
103   public M selectPosition(int selectedPosition) {
104     Checks.notNegative(selectedPosition, "Provided selectedPosition");
105     Checks.check(
106         selectedPosition < orderList.size(),
107         "Provided selectedPosition is too big and is out of bounds. selectedPosition: "
108             + selectedPosition);
109 
110     this.selectedPosition = selectedPosition;
111 
112     return (M) this;
113   }
114 
115   /**
116    * Selects a new current entity based on the index of the specified entity in the {@link
117    * #getCurrentOrder() current order} <br>
118    * This is a convenience function that uses {@link #selectPosition(int)} internally
119    *
120    * @param selectedEntity The entity for the new position that will be in focus for all
121    *     modification operations
122    * @return The current OrderAction sub-implementation instance
123    * @see #selectPosition(int)
124    * @see #getSelectedPosition()
125    * @see #getSelectedEntity()
126    */
127   public M selectPosition(T selectedEntity) {
128     Checks.notNull(selectedEntity, "Channel");
129     validateInput(selectedEntity);
130 
131     return selectPosition(orderList.indexOf(selectedEntity));
132   }
133 
134   /**
135    * The currently selected position that is in focus for all modification operations of this
136    * OrderAction instance
137    *
138    * @return The currently selected index, or -1 if no position has been selected yet
139    */
140   public int getSelectedPosition() {
141     return selectedPosition;
142   }
143 
144   /**
145    * The entity which is currently at the {@link #getSelectedPosition() selected position}
146    *
147    * @throws java.lang.IllegalStateException If no entity has been selected yet
148    * @return The currently selected entity
149    */
150   public T getSelectedEntity() {
151     if (selectedPosition == -1)
152       throw new IllegalStateException("No position has been selected yet");
153 
154     return orderList.get(selectedPosition);
155   }
156 
157   /**
158    * Moves the currently selected entity {@code amount} positions <b>UP</b> in order by pushing all
159    * entities down by one position.
160    *
161    * @param amount The amount of positions that should be moved
162    * @throws java.lang.IllegalStateException If no entity has been selected yet
163    * @throws java.lang.IllegalArgumentException If the specified amount would cause the entity to go
164    *     out-of-bounds
165    * @return The current OrderAction sub-implementation instance
166    * @see #moveTo(int)
167    */
168   public M moveUp(int amount) {
169     Checks.notNegative(amount, "Provided amount");
170     if (selectedPosition == -1)
171       throw new IllegalStateException(
172           "Cannot move until an item has been selected. Use #selectPosition first.");
173     if (ascendingOrder) {
174       Checks.check(
175           selectedPosition - amount >= 0,
176           "Amount provided to move up is too large and would be out of bounds."
177               + "Selected position: "
178               + selectedPosition
179               + " Amount: "
180               + amount
181               + " Largest Position: "
182               + orderList.size());
183     } else {
184       Checks.check(
185           selectedPosition + amount < orderList.size(),
186           "Amount provided to move up is too large and would be out of bounds."
187               + "Selected position: "
188               + selectedPosition
189               + " Amount: "
190               + amount
191               + " Largest Position: "
192               + orderList.size());
193     }
194 
195     if (ascendingOrder) return moveTo(selectedPosition - amount);
196     else return moveTo(selectedPosition + amount);
197   }
198 
199   /**
200    * Moves the currently selected entity {@code amount} positions <b>DOWN</b> in order by pushing
201    * all entities up by one position.
202    *
203    * @param amount The amount of positions that should be moved
204    * @throws java.lang.IllegalStateException If no entity has been selected yet
205    * @throws java.lang.IllegalArgumentException If the specified amount would cause the entity to go
206    *     out-of-bounds
207    * @return The current OrderAction sub-implementation instance
208    * @see #moveTo(int)
209    */
210   public M moveDown(int amount) {
211     Checks.notNegative(amount, "Provided amount");
212     if (selectedPosition == -1)
213       throw new IllegalStateException(
214           "Cannot move until an item has been selected. Use #selectPosition first.");
215 
216     if (ascendingOrder) {
217       Checks.check(
218           selectedPosition + amount < orderList.size(),
219           "Amount provided to move down is too large and would be out of bounds."
220               + "Selected position: "
221               + selectedPosition
222               + " Amount: "
223               + amount
224               + " Largest Position: "
225               + orderList.size());
226     } else {
227       Checks.check(
228           selectedPosition - amount >= orderList.size(),
229           "Amount provided to move down is too large and would be out of bounds."
230               + "Selected position: "
231               + selectedPosition
232               + " Amount: "
233               + amount
234               + " Largest Position: "
235               + orderList.size());
236     }
237 
238     if (ascendingOrder) return moveTo(selectedPosition + amount);
239     else return moveTo(selectedPosition - amount);
240   }
241 
242   /**
243    * Moves the currently selected entity to the specified position (0 based index). All entities are
244    * moved in the direction of the left <i>hole</i> to fill the gap.
245    *
246    * @param position The new not-negative position for the currently selected entity
247    * @throws java.lang.IllegalStateException If no entity has been selected yet
248    * @throws java.lang.IllegalArgumentException If the specified position is out-of-bounds
249    * @return The current OrderAction sub-implementation instance
250    * @see #moveDown(int)
251    * @see #moveUp(int)
252    */
253   public M moveTo(int position) {
254     Checks.notNegative(position, "Provided position");
255     Checks.check(position < orderList.size(), "Provided position is too big and is out of bounds.");
256 
257     T selectedItem = orderList.remove(selectedPosition);
258     orderList.add(position, selectedItem);
259 
260     return (M) this;
261   }
262 
263   /**
264    * Swaps the currently selected entity with the entity located at the specified position. No other
265    * entities are affected by this operation.
266    *
267    * @param swapPosition 0 based index of target position
268    * @throws java.lang.IllegalStateException If no entity has been selected yet
269    * @throws java.lang.IllegalArgumentException If the specified position is out-of-bounds
270    * @return The current OrderAction sub-implementation instance
271    */
272   public M swapPosition(int swapPosition) {
273     Checks.notNegative(swapPosition, "Provided swapPosition");
274     Checks.check(
275         swapPosition < orderList.size(),
276         "Provided swapPosition is too big and is out of bounds. swapPosition: " + swapPosition);
277 
278     T selectedItem = orderList.get(selectedPosition);
279     T swapItem = orderList.get(swapPosition);
280     orderList.set(swapPosition, selectedItem);
281     orderList.set(selectedPosition, swapItem);
282 
283     return (M) this;
284   }
285 
286   /**
287    * Swaps the currently selected entity with the specified entity. No other entities are affected
288    * by this operation.
289    *
290    * @param swapEntity Target entity to switch positions with
291    * @throws java.lang.IllegalStateException If no entity has been selected yet
292    * @throws java.lang.IllegalArgumentException If the specified position is out-of-bounds, or if
293    *     the target entity is {@code null} or not available in this order action implementation
294    * @return The current OrderAction sub-implementation instance
295    * @see #swapPosition(int)
296    */
297   public M swapPosition(T swapEntity) {
298     Checks.notNull(swapEntity, "Provided swapEntity");
299     validateInput(swapEntity);
300 
301     return swapPosition(orderList.indexOf(swapEntity));
302   }
303 
304   /**
305    * Reverses the {@link #getCurrentOrder() current order} by using {@link
306    * java.util.Collections#reverse(java.util.List) Collections.reverse(orderList)}
307    *
308    * @return The current OrderAction sub-implementation instance
309    * @see java.util.Collections#reverse(java.util.List)
310    */
311   public M reverseOrder() {
312     Collections.reverse(this.orderList);
313     return (M) this;
314   }
315 
316   /**
317    * Shuffles the {@link #getCurrentOrder() current order} by using {@link
318    * java.util.Collections#shuffle(java.util.List) Collections.shuffle(orderList)}
319    *
320    * @return The current OrderAction sub-implementation instance
321    * @see java.util.Collections#shuffle(java.util.List)
322    */
323   public M shuffleOrder() {
324     Collections.shuffle(this.orderList);
325     return (M) this;
326   }
327 
328   /**
329    * Sorts the {@link #getCurrentOrder() current order} based on the specified {@link
330    * java.util.Comparator Comparator}. <br>
331    * Using {@link java.util.ArrayList#sort(java.util.Comparator) ArrayList.sort(comparator)}
332    *
333    * @param comparator Comparator used to sort the current order
334    * @throws java.lang.IllegalArgumentException If the specified comparator is {@code null}
335    * @return The current OrderAction sub-implementation instance
336    * @see java.util.ArrayList#sort(java.util.Comparator)
337    */
338   public M sortOrder(final Comparator<T> comparator) {
339     Checks.notNull(comparator, "Provided comparator");
340 
341     this.orderList.sort(comparator);
342     return (M) this;
343   }
344 
345   @Override
346   protected void handleResponse(Response response, Request<Void> request) {
347     if (response.isOk()) request.onSuccess(null);
348     else request.onFailure(response);
349   }
350 
351   protected abstract void validateInput(T entity);
352 }
