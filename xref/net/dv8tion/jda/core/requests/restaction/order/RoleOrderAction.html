17  package net.dv8tion.jda.core.requests.restaction.order;
18  
19  import java.util.ArrayList;
20  import java.util.Collections;
21  import java.util.List;
22  import net.dv8tion.jda.core.Permission;
23  import net.dv8tion.jda.core.entities.Guild;
24  import net.dv8tion.jda.core.entities.Member;
25  import net.dv8tion.jda.core.entities.Role;
26  import net.dv8tion.jda.core.exceptions.InsufficientPermissionException;
27  import net.dv8tion.jda.core.requests.Route;
28  import net.dv8tion.jda.core.utils.Checks;
29  import okhttp3.RequestBody;
30  import org.json.JSONArray;
31  import org.json.JSONObject;
32  
33  /**
34   * Implementation of {@link net.dv8tion.jda.core.requests.restaction.order.OrderAction OrderAction}
35   * designed to modify the order of {@link net.dv8tion.jda.core.entities.Role Roles} of the specified
36   * {@link net.dv8tion.jda.core.entities.Guild Guild}. <br>
37   * To apply the changes you must finish the {@link net.dv8tion.jda.core.requests.RestAction
38   * RestAction}
39   *
40   * <p>Before you can use any of the {@code move} methods you must use either {@link
41   * #selectPosition(Object) selectPosition(Role)} or {@link #selectPosition(int)}!
42   *
43   * <p><b>This uses descending order!</b>
44   *
45   * @since 3.0
46   */
47  public class RoleOrderAction extends OrderAction<Role, RoleOrderAction> {
48    protected final Guild guild;
49  
50    /**
51     * Creates a new RoleOrderAction instance
52     *
53     * @param guild The target {@link net.dv8tion.jda.core.entities.Guild Guild} of which to change
54     *     the {@link net.dv8tion.jda.core.entities.Role Role} order
55     * @param useDiscordOrder Defines the ordering of the OrderAction. If {@code true}, the
56     *     OrderAction will be in the ordering defined by Discord for roles, which is Descending. This
57     *     means that the highest role appears at index {@code 0} and the lowest role at index {@code
58     *     n - 1}. Providing {@code false} will result in the ordering being in ascending order, with
59     *     the lower role at index {@code 0} and the highest at index {@code n - 1}. <br>
60     *     As a note: {@link net.dv8tion.jda.core.entities.Member#getRoles() Member.getRoles()} and
61     *     {@link net.dv8tion.jda.core.entities.Guild#getRoles() Guild.getRoles()} are both in
62     *     descending order.
63     */
64    public RoleOrderAction(Guild guild, boolean useDiscordOrder) {
65      super(guild.getJDA(), !useDiscordOrder, Route.Guilds.MODIFY_ROLES.compile(guild.getId()));
66      this.guild = guild;
67  
68      List<Role> roles = guild.getRoles();
69      roles = roles.subList(0, roles.size() - 1); // Don't include the @everyone role.
70  
71      if (useDiscordOrder) {
72        // Add roles to orderList in reverse due to role position ordering being descending
73        // Top role starts at roles.size() - 1, bottom is 0.
74        for (int i = roles.size() - 1; i >= 0; i--) this.orderList.add(roles.get(i));
75      } else {
76        // If not using discord ordering, we are ascending, so we add from first to last.
77        // We add first to last because the roles provided from getRoles() are in ascending order
78        // already
79        // with the highest role at index 0.
80        this.orderList.addAll(roles);
81      }
82    }
83  
84    /**
85     * The {@link net.dv8tion.jda.core.entities.Guild Guild} which holds the roles from {@link
86     * #getCurrentOrder()}
87     *
88     * @return The corresponding {@link net.dv8tion.jda.core.entities.Guild Guild}
89     */
90    public Guild getGuild() {
91      return guild;
92    }
93  
94    @Override
95    protected RequestBody finalizeData() {
96      final Member self = guild.getSelfMember();
97      final boolean isOwner = self.isOwner();
98  
99      if (!isOwner) {
100       if (self.getRoles().isEmpty())
101         throw new IllegalStateException(
102             "Cannot move roles above your highest role unless you are the guild owner");
103       if (!self.hasPermission(Permission.MANAGE_ROLES))
104         throw new InsufficientPermissionException(Permission.MANAGE_ROLES);
105     }
106 
107     JSONArray array = new JSONArray();
108     List<Role> ordering = new ArrayList<>(orderList);
109 
110     // If not in normal discord order, reverse.
111     // Normal order is descending, not ascending.
112     if (ascendingOrder) Collections.reverse(ordering);
113 
114     for (int i = 0; i < ordering.size(); i++) {
115       Role role = ordering.get(i);
116       final int initialPos = role.getPosition();
117       if (initialPos != i && !isOwner && !self.canInteract(role))
118         // If the current role was moved, we are not owner and we can't interact with the role then
119         // throw a PermissionException
120         throw new IllegalStateException(
121             "Cannot change order: One of the roles could not be moved due to hierarchical power!");
122 
123       array.put(
124           new JSONObject()
125               .put("id", role.getId())
126               .put("position", i + 1)); // plus 1 because position 0 is the @everyone position.
127     }
128 
129     return getRequestBody(array);
130   }
131 
132   @Override
133   protected void validateInput(Role entity) {
134     Checks.check(entity.getGuild().equals(guild), "Provided selected role is not from this Guild!");
135     Checks.check(
136         orderList.contains(entity), "Provided role is not in the list of orderable roles!");
137   }
138 }
