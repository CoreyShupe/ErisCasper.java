17  package net.dv8tion.jda.core;
18  
19  /**
20   * Represents the Regions used for Audio connections. <br>
21   * This is used by {@link net.dv8tion.jda.core.entities.Guild Guild} to where the server that hosts
22   * the {@link net.dv8tion.jda.core.entities.Guild Guild} is located.
23   */
24  public enum Region {
25    AMSTERDAM("amsterdam", "Amsterdam", false),
26    BRAZIL("brazil", "Brazil", false),
27    EU_WEST("eu-west", "EU West", false),
28    EU_CENTRAL("eu-central", "EU Central", false),
29    FRANKFURT("frankfurt", "Frankfurt", false),
30    HONG_KONG("hongkong", "Hong Kong", false),
31    LONDON("london", "London", false),
32    RUSSIA("russia", "Russia", false),
33    SINGAPORE("singapore", "Singapore", false),
34    SYDNEY("sydney", "Sydney", false),
35    US_EAST("us-east", "US East", false),
36    US_WEST("us-west", "US West", false),
37    US_CENTRAL("us-central", "US Central", false),
38    US_SOUTH("us-south", "US South", false),
39  
40    VIP_AMSTERDAM("vip-amsterdam", "Amsterdam (VIP)", true),
41    VIP_BRAZIL("vip-brazil", "Brazil (VIP)", true),
42    VIP_EU_WEST("vip-eu-west", "EU West (VIP)", true),
43    VIP_EU_CENTRAL("vip-eu-central", "EU Central (VIP)", true),
44    VIP_FRANKFURT("vip-frankfurt", "Frankfurt (VIP)", true),
45    VIP_LONDON("vip-london", "London (VIP)", true),
46    VIP_SINGAPORE("vip-singapore", "Singapore (VIP)", true),
47    VIP_SYDNEY("vip-sydney", "Sydney (VIP)", true),
48    VIP_US_EAST("vip-us-east", "US East (VIP)", true),
49    VIP_US_WEST("vip-us-west", "US West (VIP)", true),
50    VIP_US_CENTRAL("vip-us-central", "US Central (VIP)", true),
51    VIP_US_SOUTH("vip-us-south", "US South (VIP)", true),
52  
53    UNKNOWN("", "Unknown Region", false);
54  
55    private final String key;
56    private final String name;
57    private final boolean vip;
58  
59    Region(String key, String name, boolean vip) {
60      this.key = key;
61      this.name = name;
62      this.vip = vip;
63    }
64  
65    /**
66     * The human readable region name.
67     *
68     * @return The name of this region
69     */
70    public String getName() {
71      return name;
72    }
73  
74    /**
75     * The Region key as defined by Discord.
76     *
77     * @return The key (internal name) of this region
78     */
79    public String getKey() {
80      return key;
81    }
82  
83    /**
84     * Whether or not this Region is a VIP region. <br>
85     * VIP regions have special perks like higher bitrate in VoiceChannels and priority during times
86     * of high Discord usage.
87     *
88     * @return True if this region is a VIP audio region.
89     */
90    public boolean isVip() {
91      return vip;
92    }
93  
94    /**
95     * Retrieves the {@link net.dv8tion.jda.core.Region Region} based on the provided key.
96     *
97     * @param key The key relating to the {@link net.dv8tion.jda.core.Region Region} we wish to
98     *     retrieve.
99     * @return The {@link net.dv8tion.jda.core.Region Region} matching the key. If there is no match,
100    *     returns {@link net.dv8tion.jda.core.Region#UNKNOWN UNKNOWN}.
101    */
102   public static Region fromKey(String key) {
103     for (Region region : values()) {
104       if (region.getKey().equals(key)) {
105         return region;
106       }
107     }
108     return UNKNOWN;
109   }
110 
111   @Override
112   public String toString() {
113     return getName();
114   }
115 }
