17  package net.dv8tion.jda.core.exceptions;
18  
19  import net.dv8tion.jda.core.requests.ErrorResponse;
20  import net.dv8tion.jda.core.requests.Response;
21  import org.json.JSONObject;
22  
23  /**
24   * Indicates an unhandled error that is returned by Discord API Request using {@link
25   * net.dv8tion.jda.core.requests.RestAction RestAction} <br>
26   * It holds an {@link net.dv8tion.jda.core.requests.ErrorResponse ErrorResponse}
27   */
28  public class ErrorResponseException extends RuntimeException {
29    private final ErrorResponse errorResponse;
30    private final Response response;
31    private final String meaning;
32    private final int code;
33  
34    /**
35     * Creates a new ErrorResponseException instance
36     *
37     * @param errorResponse The {@link net.dv8tion.jda.core.requests.ErrorResponse ErrorResponse}
38     *     corresponding for the received error response from Discord
39     * @param response The Discord Response causing the ErrorResponse
40     */
41    private ErrorResponseException(
42        ErrorResponse errorResponse, Response response, int code, String meaning) {
43      super(code + ": " + meaning);
44  
45      this.response = response;
46      this.errorResponse = errorResponse;
47      this.code = code;
48      this.meaning = meaning;
49    }
50  
51    /**
52     * Whether this is an internal server error from discord (status 500)
53     *
54     * @return True, if this is an internal server error {@link
55     *     net.dv8tion.jda.core.requests.ErrorResponse#SERVER_ERROR ErrorResponse.SERVER_ERROR}
56     */
57    public boolean isServerError() {
58      return errorResponse == ErrorResponse.SERVER_ERROR;
59    }
60  
61    /**
62     * The meaning for this error. <br>
63     * It is possible that the value from this method is different for {@link #isServerError() server
64     * errors}
65     *
66     * @return Never-null meaning of this error.
67     */
68    public String getMeaning() {
69      return meaning;
70    }
71  
72    /**
73     * The discord error code for this error response.
74     *
75     * @return The discord error code.
76     * @see <a href="https://discordapp.com/developers/docs/topics/response-codes#json-error-response"
77     *     target="_blank">Discord Error Codes</a>
78     */
79    public int getErrorCode() {
80      return code;
81    }
82  
83    /**
84     * The {@link net.dv8tion.jda.core.requests.ErrorResponse ErrorResponse} corresponding for the
85     * received error response from Discord
86     *
87     * @return {@link net.dv8tion.jda.core.requests.ErrorResponse ErrorResponse}
88     */
89    public ErrorResponse getErrorResponse() {
90      return errorResponse;
91    }
92  
93    /**
94     * The Discord Response causing the ErrorResponse
95     *
96     * @return {@link net.dv8tion.jda.core.requests.Response Response}
97     */
98    public Response getResponse() {
99      return response;
100   }
101 
102   public static ErrorResponseException create(ErrorResponse errorResponse, Response response) {
103     JSONObject obj = response.getObject();
104     String meaning = errorResponse.getMeaning();
105     int code = errorResponse.getCode();
106     if (response.isError() && response.getException() != null) {
107       // this generally means that an exception occurred trying to
108       // make an http request. e.g.:
109       // SocketTimeoutException/ UnknownHostException
110       code = response.code;
111       meaning = response.getException().getClass().getName();
112     } else if (obj != null) {
113       if (!obj.isNull("code") || !obj.isNull("message")) {
114         if (!obj.isNull("code")) code = obj.getInt("code");
115         if (!obj.isNull("message")) meaning = obj.getString("message");
116       } else {
117         // This means that neither code or message is provided
118         // In that case we simply put the raw response in place!
119         code = response.code;
120         meaning = response.getString();
121       }
122     } else {
123       // error response body is not JSON
124       code = response.code;
125       meaning = response.getString();
126     }
127 
128     return new ErrorResponseException(errorResponse, response, code, meaning);
129   }
130 }
