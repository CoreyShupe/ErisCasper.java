17  package net.dv8tion.jda.core.events.message.react;
18  
19  import net.dv8tion.jda.client.entities.Group;
20  import net.dv8tion.jda.core.JDA;
21  import net.dv8tion.jda.core.entities.*;
22  import net.dv8tion.jda.core.events.message.GenericMessageEvent;
23  
24  public class GenericMessageReactionEvent extends GenericMessageEvent {
25    protected User issuer;
26    protected MessageReaction reaction;
27  
28    public GenericMessageReactionEvent(
29        JDA api, long responseNumber, User user, MessageReaction reaction) {
30      super(api, responseNumber, reaction.getMessageIdLong(), reaction.getChannel());
31      this.issuer = user;
32      this.reaction = reaction;
33    }
34  
35    public Guild getGuild() {
36      TextChannel channel = getTextChannel();
37      return channel != null ? channel.getGuild() : null;
38    }
39  
40    public TextChannel getTextChannel() {
41      return isFromType(ChannelType.TEXT) ? (TextChannel) getChannel() : null;
42    }
43  
44    public PrivateChannel getPrivateChannel() {
45      return isFromType(ChannelType.PRIVATE) ? (PrivateChannel) getChannel() : null;
46    }
47  
48    public Group getGroup() {
49      return isFromType(ChannelType.GROUP) ? (Group) getChannel() : null;
50    }
51  
52    public User getUser() {
53      return issuer;
54    }
55  
56    public Member getMember() {
57      Guild guild = getGuild();
58      return guild != null ? guild.getMember(getUser()) : null;
59    }
60  
61    public MessageReaction getReaction() {
62      return reaction;
63    }
64  
65    public MessageReaction.ReactionEmote getReactionEmote() {
66      return reaction.getReactionEmote();
67    }
68  }
