98    private final int offset;
99    private final long raw;
100   private final boolean isGuild, isChannel;
101   private final String name;

103   Permission(int offset, boolean isGuild, boolean isChannel, String name) {
104     this.offset = offset;
105     this.raw = 1 << offset;
106     this.isGuild = isGuild;
107     this.isChannel = isChannel;
108     this.name = name;
109   }

162   /**
163    * Whether this permission is specifically for {@link net.dv8tion.jda.core.entities.TextChannel
164    * TextChannels}
165    *
166    * @return True, if and only if this permission can be applied to only text channels
167    */
168   public boolean isText() {
169     return offset > 9 && offset < 20;
170   }
171 
172   /**
173    * Whether this permission is specifically for {@link net.dv8tion.jda.core.entities.VoiceChannel
174    * VoiceChannels}
175    *
176    * @return True, if and only if this permission can be applied to only voice channels
177    */
178   public boolean isVoice() {
179     return offset == 10 || offset > 19 && offset < 26;
180   }

182   /**
183    * Gets the {@link net.dv8tion.jda.core.Permission Permission} relating to the provided offset.
184    * <br>
185    * If there is no {@link net.dv8tion.jda.core.Permission Permssions} that matches the provided
186    * offset, {@link net.dv8tion.jda.core.Permission#UNKNOWN Permission.UNKNOWN} is returned.
187    *
188    * @param offset The offset to match a {@link net.dv8tion.jda.core.Permission Permission} to.
189    * @return {@link net.dv8tion.jda.core.Permission Permission} relating to the provided offset.
190    */
191   public static Permission getFromOffset(int offset) {
192     for (Permission perm : values()) {
193       if (perm.offset == offset) return perm;
194     }
195     return UNKNOWN;
196   }

198   /**
199    * A list of all {@link net.dv8tion.jda.core.Permission Permissions} that are specified by this
200    * raw long representation of permissions. The is best used with the getRaw methods in {@link
201    * net.dv8tion.jda.core.entities.Role Role}, {@link
202    * net.dv8tion.jda.core.entities.PermissionOverride PermissionOverride} or {@link
203    * net.dv8tion.jda.core.utils.PermissionUtil}.
204    *
205    * <p>Examples: <br>
206    * {@link net.dv8tion.jda.core.entities.Role#getPermissionsRaw() Role.getPermissionsRaw()} <br>
207    * {@link
208    * net.dv8tion.jda.core.utils.PermissionUtil#getEffectivePermission(net.dv8tion.jda.core.entities.Channel,
209    * net.dv8tion.jda.core.entities.Member) PermissionUtil.getEffectivePermission(Channel, Member)}
210    *
211    * @param permissions The raw {@code long} representation of permissions.
212    * @return Possibly-empty list of {@link net.dv8tion.jda.core.Permission Permissions}.
213    */
214   public static List<Permission> getPermissions(long permissions) {
215     List<Permission> perms = new LinkedList<>();
216     for (Permission perm : Permission.values()) {
217       if (perm.equals(Permission.UNKNOWN)) continue;
218       if (((permissions >> perm.getOffset()) & 1) == 1) perms.add(perm);
219     }
220     return perms;
221   }
222 
223   /**
224    * This is effectively the opposite of {@link #getPermissions(long)}, this takes 1 or more {@link
225    * net.dv8tion.jda.core.Permission Permissions} and returns the raw offset {@code long}
226    * representation of the permissions.
227    *
228    * @param permissions The array of permissions of which to form into the raw long representation.
229    * @return Unsigned long representing the provided permissions.
230    */
231   public static long getRaw(Permission... permissions) {
232     long raw = 0;
233     for (Permission perm : permissions) {
234       if (perm != UNKNOWN) raw |= perm.raw;
235     }
236 
237     return raw;
238   }
239 
240   /**
241    * This is effectively the opposite of {@link #getPermissions(long)}, this takes a Collection of
242    * {@link net.dv8tion.jda.core.Permission Permissions} and returns the raw offset {@code long}
243    * representation of the permissions.
244    *
245    * @param permissions The Collection of permissions of which to form into the raw long
246    *     representation.
247    * @return Unsigned long representing the provided permissions.
248    */
249   public static long getRaw(Collection<Permission> permissions) {
250     Checks.notNull(permissions, "Permission Collection");
251 
252     return getRaw(permissions.toArray(new Permission[permissions.size()]));
253   }
