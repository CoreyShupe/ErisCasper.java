17  package net.dv8tion.jda.core.utils.tuple;
18  
19  import java.io.Serializable;
20  import java.util.Objects;
21  
22  /**
23   * This class has been copied from <a href="https://commons.apache.org/proper/commons-lang/"
24   * target="_blank">Lang 3</a>
25   *
26   * <p>A pair consisting of two elements.
27   *
28   * <p>This class is an abstract implementation defining the basic API. It refers to the elements as
29   * 'left' and 'right'.
30   *
31   * <p>Subclass implementations may be mutable or immutable. However, there is no restriction on the
32   * type of the stored objects that may be stored. If mutable objects are stored in the pair, then
33   * the pair itself effectively becomes mutable.
34   *
35   * @param <L> the left element type
36   * @param <R> the right element type
37   * @since Lang 3.0
38   */
39  public abstract class Pair<L, R>
40      implements /*Map.Entry<L, R>, Comparable<Pair<L, R>>,*/ Serializable {
41  
42    /**
43     * Obtains an immutable pair of from two objects inferring the generic types.
44     *
45     * <p>This factory allows the pair to be created using inference to obtain the generic types.
46     *
47     * @param <L> the left element type
48     * @param <R> the right element type
49     * @param left the left element, may be null
50     * @param right the right element, may be null
51     * @return a pair formed from the two parameters, not null
52     */
53    public static <L, R> Pair<L, R> of(final L left, final R right) {
54      return new ImmutablePair<>(left, right);
55    }
56  
57    // -----------------------------------------------------------------------
58    /**
59     * Gets the left element from this pair.
60     *
61     * <p>When treated as a key-value pair, this is the key.
62     *
63     * @return the left element, may be null
64     */
65    public abstract L getLeft();
66  
67    /**
68     * Gets the right element from this pair.
69     *
70     * <p>When treated as a key-value pair, this is the value.
71     *
72     * @return the right element, may be null
73     */
74    public abstract R getRight();
75  
76    /**
77     * Compares this pair to another based on the two elements.
78     *
79     * @param obj the object to compare to, null returns false
80     * @return true if the elements of the pair are equal
81     */
82    @Override
83    public boolean equals(final Object obj) {
84      if (obj == this) {
85        return true;
86      }
87      if (obj instanceof Pair<?, ?>) {
88        final Pair<?, ?> other = (Pair<?, ?>) obj;
89        return Objects.equals(getLeft(), other.getLeft())
90            && Objects.equals(getRight(), other.getRight());
91      }
92      return false;
93    }
94  
95    /**
96     * Returns a suitable hash code. The hash code follows the definition in {@code Map.Entry}.
97     *
98     * @return the hash code
99     */
100   @Override
101   public int hashCode() {
102     // see Map.Entry API specification
103     return Objects.hashCode(getLeft()) ^ Objects.hashCode(getRight());
104   }
105 
106   /**
107    * Returns a String representation of this pair using the format {@code ($left,$right)}.
108    *
109    * @return a string describing this object, not null
110    */
111   @Override
112   public String toString() {
113     return "(" + getLeft() + ',' + getRight() + ')';
114   }
115 }
