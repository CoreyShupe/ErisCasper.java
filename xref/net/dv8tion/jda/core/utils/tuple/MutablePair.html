17  package net.dv8tion.jda.core.utils.tuple;
18  
19  /**
20   * This class has been copied from <a href="https://commons.apache.org/proper/commons-lang/"
21   * target="_blank">Lang 3</a>
22   *
23   * <p>A mutable pair consisting of two {@code Object} elements.
24   *
25   * <p>Not #ThreadSafe#
26   *
27   * @param <L> the left element type
28   * @param <R> the right element type
29   * @since Lang 3.0
30   */
31  public class MutablePair<L, R> extends Pair<L, R> {
32    /** Left object */
33    public L left;
34    /** Right object */
35    public R right;
36  
37    /**
38     * Obtains an immutable pair of from two objects inferring the generic types.
39     *
40     * <p>This factory allows the pair to be created using inference to obtain the generic types.
41     *
42     * @param <L> the left element type
43     * @param <R> the right element type
44     * @param left the left element, may be null
45     * @param right the right element, may be null
46     * @return a pair formed from the two parameters, not null
47     */
48    public static <L, R> MutablePair<L, R> of(final L left, final R right) {
49      return new MutablePair<>(left, right);
50    }
51  
52    /** Create a new pair instance of two nulls. */
53    public MutablePair() {
54      super();
55    }
56  
57    /**
58     * Create a new pair instance.
59     *
60     * @param left the left value, may be null
61     * @param right the right value, may be null
62     */
63    public MutablePair(final L left, final R right) {
64      super();
65      this.left = left;
66      this.right = right;
67    }
68  
69    @Override
70    public L getLeft() {
71      return left;
72    }
73  
74    /**
75     * Sets the left element of the pair.
76     *
77     * @param left the new value of the left element, may be null
78     */
79    public void setLeft(final L left) {
80      this.left = left;
81    }
82  
83    @Override
84    public R getRight() {
85      return right;
86    }
87  
88    /**
89     * Sets the right element of the pair.
90     *
91     * @param right the new value of the right element, may be null
92     */
93    public void setRight(final R right) {
94      this.right = right;
95    }
96  }
