114       boolean autoReconnect,

130     this.bulkDeleteSplittingEnabled = bulkDeleteSplittingEnabled;

142     this.jdaBot = accountType == AccountType.BOT ? new JDABotImpl(this) : null;

149   public int login(String gatewayUrl, ShardInfo shardInfo) throws LoginException {
150     this.gatewayUrl = gatewayUrl;
151     this.shardInfo = shardInfo;
152 
153     setStatus(Status.LOGGING_IN);
154     if (token == null || token.isEmpty())
155       throw new LoginException("Provided token was null or empty!");
156 
157     Map<String, String> previousContext = null;
158     if (contextMap != null) {
159       if (shardInfo != null) {
160         contextMap.put("jda.shard", shardInfo.getShardString());
161         contextMap.put("jda.shard.id", String.valueOf(shardInfo.getShardId()));
162         contextMap.put("jda.shard.total", String.valueOf(shardInfo.getShardTotal()));
163       }
164     }
165     verifyToken();
166     LOG.info("Login Successful!");
167 
168     client = new WebSocketClient(this);
169 
170     if (shutdownHook != null) Runtime.getRuntime().addShutdownHook(shutdownHook);
171 
172     return shardInfo == null ? -1 : shardInfo.getShardTotal();
173   }

208   // @param alreadyFailed If has already been a failed attempt with the current configuration
209   public void verifyToken(boolean alreadyFailed) throws LoginException {
210 
211     RestAction<JSONObject> login =
212         new RestAction<JSONObject>(this, Route.Self.GET_SELF.compile()) {
213           @Override
214           protected void handleResponse(Response response, Request<JSONObject> request) {
215             if (response.isOk()) request.onSuccess(response.getObject());
216             else if (response.isRateLimit())
217               request.onFailure(new RateLimitedException(request.getRoute(), response.retryAfter));
218             else if (response.code == 401) request.onSuccess(null);
219             else
220               request.onFailure(
221                   new LoginException(
222                       "When verifying the authenticity of the provided token, Discord returned an unknown response:\n"
223                           + response.toString()));
224           }
225         };
226 
227     JSONObject userResponse;
228 
229     if (!alreadyFailed) {
230       userResponse = checkToken(login);
231       if (userResponse != null) {
232         verifyAccountType(userResponse);
233         return;
234       }
235     }
236 
237     // If we received a null return for userResponse, then that means we hit a 401.
238     // 401 occurs when we attempt to access the users/@me endpoint with the wrong token prefix.
239     // e.g: If we use a Client token and prefix it with "Bot ", or use a bot token and don't prefix
240     // it.
241     // It also occurs when we attempt to access the endpoint with an invalid token.
242     // The code below already knows that something is wrong with the token. We want to determine if
243     // it is invalid
244     // or if the developer attempted to login with a token using the wrong AccountType.
245 
246     // If we attempted to login as a Bot, remove the "Bot " prefix and set the Requester to be a
247     // client.
248     if (getAccountType() == AccountType.BOT) {
249       token = token.substring("Bot ".length());
250       requester = new Requester(this, AccountType.CLIENT);
251     } else // If we attempted to login as a Client, prepend the "Bot " prefix and set the Requester
252     // to be a Bot
253     {
254       token = "Bot " + token;
255       requester = new Requester(this, AccountType.BOT);
256     }
257 
258     userResponse = checkToken(login);
259 
260     // If the response isn't null (thus it didn't 401) send it to the secondary verify method to
261     // determine
262     // which account type the developer wrongly attempted to login as
263     if (userResponse != null) verifyAccountType(userResponse);
264     else // We 401'd again. This is an invalid token
265     throw new LoginException("The provided token is invalid!");
266   }
267 
268   private void verifyAccountType(JSONObject userResponse) {
269     if (getAccountType() == AccountType.BOT) {
270       if (!userResponse.has("bot") || !userResponse.getBoolean("bot"))
271         throw new AccountTypeException(
272             AccountType.BOT, "Attempted to login as a BOT with a CLIENT token!");
273     } else {
274       if (userResponse.has("bot") && userResponse.getBoolean("bot"))
275         throw new AccountTypeException(
276             AccountType.CLIENT, "Attempted to login as a CLIENT with a BOT token!");
277     }
278   }
279 
280   private JSONObject checkToken(RestAction<JSONObject> login) throws LoginException {
281     JSONObject userResponse;
282     try {
283       userResponse = login.complete();
284     } catch (RuntimeException e) {
285       // We check if the LoginException is masked inside of a ExecutionException which is masked
286       // inside of the RuntimeException
287       Throwable ex = e.getCause() instanceof ExecutionException ? e.getCause().getCause() : null;
288       if (ex instanceof LoginException) throw new LoginException(ex.getMessage());
289       else throw e;
290     }
291     return userResponse;
292   }

367   @Override
368   public RestAction<User> retrieveUserById(long id) {
369     AccountTypeException.check(accountType, AccountType.BOT);
370 
371     // check cache
372     User user = this.getUserById(id);
373     if (user != null) return new RestAction.EmptyRestAction<>(this, user);
374 
375     Route.CompiledRoute route = Route.Users.GET_USER.compile(Long.toUnsignedString(id));
376     return new RestAction<User>(this, route) {
377       @Override
378       protected void handleResponse(Response response, Request<User> request) {
379         if (!response.isOk()) {
380           request.onFailure(response);
381           return;
382         }
383         JSONObject user = response.getObject();
384         request.onSuccess(getEntityBuilder().createFakeUser(user, false));
385       }
386     };
387   }

446   @Override
447   public void shutdown() {
448     if (status == Status.SHUTDOWN || status == Status.SHUTTING_DOWN) return;
449 
450     setStatus(Status.SHUTTING_DOWN);
451     audioManagers.forEach(AudioManager::closeAudioConnection);
452     audioManagers.clear();
453 
454     if (audioKeepAlivePool != null) audioKeepAlivePool.shutdownNow();
455 
456     getClient().shutdown();
457 
458     final long time = 5L;
459     final TimeUnit unit = TimeUnit.SECONDS;
460     getRequester().shutdown(time, unit);
461     pool.setKeepAliveTime(time, unit);
462     pool.allowCoreThreadTimeOut(true);
463 
464     if (shutdownHook != null) {
465       try {
466         Runtime.getRuntime().removeShutdownHook(shutdownHook);
467       } catch (Exception ignored) {
468       }
469     }
470 
471     setStatus(Status.SHUTDOWN);
472   }

543   @Override
544   public GuildAction createGuild(String name) {
545     switch (accountType) {
546       case BOT:
547         if (guildCache.size() >= 10)
548           throw new IllegalStateException(
549               "Cannot create a Guild with a Bot in more than 10 guilds!");
550         break;
551       case CLIENT:
552         if (guildCache.size() >= 100)
553           throw new IllegalStateException(
554               "Cannot be in more than 100 guilds with AccountType.CLIENT!");
555     }
556     return new GuildAction(this, name);
557   }

644   private class JDAThreadFactory implements ThreadFactory {
645     @Override
646     public Thread newThread(Runnable r) {
647       final Thread thread =
648           new Thread(
649               () -> {
650                 r.run();
651               },
652               "JDA-Thread " + getIdentifierString());
653       thread.setDaemon(true);
654       return thread;
655     }
656   }

658   public ScheduledThreadPoolExecutor getAudioKeepAlivePool() {
659     ScheduledThreadPoolExecutor akap = audioKeepAlivePool;
660     if (akap == null) {
661       synchronized (akapLock) {
662         akap = audioKeepAlivePool;
663         if (akap == null)
664           akap =
665               audioKeepAlivePool =
666                   new ScheduledThreadPoolExecutor(
667                       1, new AudioWebSocket.KeepAliveThreadFactory(this));
668       }
669     }
670     return akap;
671   }
