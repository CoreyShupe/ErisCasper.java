57    @Override
58    public RestAction<List<Webhook>> getWebhooks() {
59      checkPermission(Permission.MANAGE_WEBHOOKS);
60  
61      Route.CompiledRoute route = Route.Channels.GET_WEBHOOKS.compile(getId());
62      return new RestAction<List<Webhook>>(getJDA(), route) {
63        @Override
64        protected void handleResponse(Response response, Request<List<Webhook>> request) {
65          if (!response.isOk()) {
66            request.onFailure(response);
67            return;
68          }
69  
70          JSONArray array = response.getArray();
71          List<Webhook> webhooks = new ArrayList<>(array.length());
72          EntityBuilder builder = api.getEntityBuilder();
73  
74          for (Object object : array) {
75            try {
76              webhooks.add(builder.createWebhook((JSONObject) object));
77            } catch (JSONException | NullPointerException e) {
78              JDAImpl.LOG.error("Error while creating websocket from json", e);
79            }
80          }
81  
82          request.onSuccess(Collections.unmodifiableList(webhooks));
83        }
84      };
85    }

89      Checks.notBlank(name, "Webhook name");

97      return new WebhookAction(getJDA(), route, name);

108   @Override
109   public RestAction<Void> deleteMessagesByIds(Collection<String> messageIds) {
110     checkPermission(
111         Permission.MESSAGE_MANAGE,
112         "Must have MESSAGE_MANAGE in order to bulk delete messages in this channel regardless of author.");
113     if (messageIds.size() < 2 || messageIds.size() > 100)
114       throw new IllegalArgumentException(
115           "Must provide at least 2 or at most 100 messages to be deleted.");
116 
117     long twoWeeksAgo =
118         ((System.currentTimeMillis() - (14 * 24 * 60 * 60 * 1000)) - MiscUtil.DISCORD_EPOCH)
119             << MiscUtil.TIMESTAMP_OFFSET;
120     for (String id : messageIds) {
121       Checks.notEmpty(id, "Message id in messageIds");
122       Checks.check(
123           MiscUtil.parseSnowflake(id) > twoWeeksAgo,
124           "Message Id provided was older than 2 weeks. Id: " + id);
125     }
126 
127     JSONObject body = new JSONObject().put("messages", messageIds);
128     Route.CompiledRoute route = Route.Messages.DELETE_MESSAGES.compile(getId());
129     return new RestAction<Void>(getJDA(), route, body) {
130       @Override
131       protected void handleResponse(Response response, Request<Void> request) {
132         if (response.isOk()) request.onSuccess(null);
133         else request.onFailure(response);
134       }
135     };
136   }

207   @Override
208   public int getPosition() {
209     // We call getTextChannels instead of directly accessing the GuildImpl.getTextChannelMap because
210     // getTextChannels does the sorting logic.
211     List<TextChannel> channels = guild.getTextChannels();
212     for (int i = 0; i < channels.size(); i++) {
213       if (channels.get(i) == this) return i;
214     }
215     throw new AssertionError(
216         "Somehow when determining position we never found the TextChannel in the Guild's channels? wtf?");
217   }

219   @Override
220   public ChannelAction createCopy(Guild guild) {
221     Checks.notNull(guild, "Guild");
222     ChannelAction action =
223         guild.getController().createTextChannel(name).setNSFW(nsfw).setTopic(topic);
224     if (guild.equals(getGuild())) {
225       Category parent = getParent();
226       if (parent != null) action.setParent(parent);
227       for (PermissionOverride o : overrides.valueCollection()) {
228         if (o.isMemberOverride())
229           action.addPermissionOverride(o.getMember(), o.getAllowedRaw(), o.getDeniedRaw());
230         else action.addPermissionOverride(o.getRole(), o.getAllowedRaw(), o.getDeniedRaw());
231       }
232     }
233     return action;
234   }

361   @Override
362   public RestAction<Void> removeReactionById(String messageId, String unicode, User user) {
363     Checks.noWhitespace(messageId, "Message ID");
364     Checks.noWhitespace(unicode, "Unicode emoji");
365     Checks.notNull(user, "User");
366     if (!getJDA().getSelfUser().equals(user)) checkPermission(Permission.MESSAGE_MANAGE);
367     final String code = MiscUtil.encodeUTF8(unicode);
368     Route.CompiledRoute route;
369     if (user.equals(getJDA().getSelfUser()))
370       route = Route.Messages.REMOVE_OWN_REACTION.compile(getId(), messageId, code);
371     else route = Route.Messages.REMOVE_REACTION.compile(getId(), messageId, code, user.getId());
372     return new RestAction<Void>(getJDA(), route) {
373       @Override
374       protected void handleResponse(Response response, Request<Void> request) {
375         if (!response.isOk()) request.onFailure(response);
376         else request.onSuccess(null);
377       }
378     };
379   }

415   @Override
416   public int compareTo(TextChannel chan) {
417     Checks.notNull(chan, "Other TextChannel");
418     if (this == chan) return 0;
419     Checks.check(
420         getGuild().equals(chan.getGuild()),
421         "Cannot compare TextChannels that aren't from the same guild!");
422     if (this.getPositionRaw() == chan.getPositionRaw()) return Long.compare(id, chan.getIdLong());
423     return Integer.compare(rawPosition, chan.getPositionRaw());
424   }
