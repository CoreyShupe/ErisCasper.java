32  /**
33   * An object representing a single MessageReaction from Discord. This is an immutable object and is
34   * not updated by method calls or changes in Discord. A new snapshot instance built from Discord is
35   * needed to see changes.
36   *
37   * @since 3.0
38   * @author Florian Spieß
39   */
40  public class MessageReaction {
41  
42    private final MessageChannel channel;
43    private final ReactionEmote emote;
44    private final long messageId;
45    private final boolean self;
46    private final int count;
47  
48    /**
49     * Creates a new MessageReaction instance
50     *
51     * @param channel The {@link net.dv8tion.jda.core.entities.MessageChannel} this Reaction was used
52     *     in
53     * @param emote The {@link net.dv8tion.jda.core.entities.MessageReaction.ReactionEmote
54     *     ReactionEmote} that was used
55     * @param messageId The message id this reaction is attached to
56     * @param self Whether we already reacted with this Reaction
57     * @param count The amount of people that reacted with this Reaction
58     */
59    public MessageReaction(
60        MessageChannel channel, ReactionEmote emote, long messageId, boolean self, int count) {
61      this.channel = channel;
62      this.emote = emote;
63      this.messageId = messageId;
64      this.self = self;
65      this.count = count;
66    }

68    /**
69     * The JDA instance of this Reaction
70     *
71     * @return The JDA instance of this Reaction
72     */
73    public JDA getJDA() {
74      return channel.getJDA();
75    }

78     * Whether the currently logged in account has reacted with this reaction
79     *
80     * @return True, if we reacted with this reaction

87     * The amount of users that already reacted with this Reaction <br>
88     * <b>This is not updated, it is a {@code final int} per Reaction instance</b>
89     *
90     * <p>This value is not available in events such as {@link
91     * net.dv8tion.jda.core.events.message.react.MessageReactionAddEvent MessageReactionAddEvent} and
92     * {@link net.dv8tion.jda.core.events.message.react.MessageReactionRemoveEvent
93     * MessageReactionRemoveEvent} in which case an {@link java.lang.IllegalStateException
94     * IllegalStateException} is thrown!
95     *
96     * @throws java.lang.IllegalStateException If this MessageReaction is from an event which does not
97     *     provide a count
98     * @return The amount of users that reacted with this Reaction
99     */
100   public int getCount() {
101     if (count < 0)
102       throw new IllegalStateException("Cannot retrieve count for this MessageReaction!");
103     return count;
104   }

259   /**
260    * Removes this Reaction from the Message. <br>
261    * This will remove our own reaction as an overload of {@link #removeReaction(User)}.
262    *
263    * <p>Possible ErrorResponses include:
264    *
265    * <ul>
266    *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#UNKNOWN_MESSAGE UNKNOWN_MESSAGE} <br>
267    *       If the message this reaction was attached to got deleted.
268    *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#UNKNOWN_CHANNEL UNKNOWN_CHANNEL} <br>
269    *       If the channel this reaction was used in got deleted.
270    *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#MISSING_ACCESS MISSING_ACCESS} <br>
271    *       If we were removed from the channel/guild
272    * </ul>
273    *
274    * @return {@link net.dv8tion.jda.core.requests.RestAction RestAction} - Type: Void Nothing is
275    *     returned on success
276    */
277   public RestAction<Void> removeReaction() {
278     return removeReaction(getJDA().getSelfUser());
279   }

281   /**
282    * Removes this Reaction from the Message. <br>
283    * This will remove the reaction of the {@link net.dv8tion.jda.core.entities.User User} provided.
284    *
285    * <p>If the provided User did not react with this Reaction this does nothing.
286    *
287    * <p>Possible ErrorResponses include:
288    *
289    * <ul>
290    *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#UNKNOWN_MESSAGE UNKNOWN_MESSAGE} <br>
291    *       If the message this reaction was attached to got deleted.
292    *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#UNKNOWN_CHANNEL UNKNOWN_CHANNEL} <br>
293    *       If the channel this reaction was used in got deleted.
294    *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#MISSING_ACCESS MISSING_ACCESS} <br>
295    *       If we were removed from the channel/guild
296    * </ul>
297    *
298    * @param user The User of which to remove the reaction
299    * @throws java.lang.IllegalArgumentException If the provided {@code user} is null.
300    * @throws net.dv8tion.jda.core.exceptions.InsufficientPermissionException if the provided User is
301    *     not us and we do not have permission to {@link
302    *     net.dv8tion.jda.core.Permission#MESSAGE_MANAGE manage messages} in the channel this
303    *     reaction was used in
304    * @return {@link net.dv8tion.jda.core.requests.RestAction RestAction} - Type: Void Nothing is
305    *     returned on success
306    */
307   public RestAction<Void> removeReaction(User user) {
308     if (user == null) throw new IllegalArgumentException("Provided User was null!");
309     if (!user.equals(getJDA().getSelfUser())) {
310       if (channel.getType() == ChannelType.TEXT) {
311         Channel channel = (Channel) this.channel;
312         if (!channel.getGuild().getSelfMember().hasPermission(channel, Permission.MESSAGE_MANAGE))
313           throw new InsufficientPermissionException(Permission.MESSAGE_MANAGE);
314       } else {
315         throw new PermissionException(
316             "Unable to remove Reaction of other user in non-text channel!");
317       }
318     }
319 
320     String code =
321         emote.isEmote()
322             ? emote.getName() + ":" + emote.getId()
323             : MiscUtil.encodeUTF8(emote.getName());
324     Route.CompiledRoute route;
325     if (user.equals(getJDA().getSelfUser()))
326       route = Route.Messages.REMOVE_OWN_REACTION.compile(channel.getId(), getMessageId(), code);
327     else
328       route =
329           Route.Messages.REMOVE_REACTION.compile(
330               channel.getId(), getMessageId(), code, user.getId());
331     return new RestAction<Void>(getJDA(), route) {
332       @Override
333       protected void handleResponse(Response response, Request<Void> request) {
334         if (response.isOk()) request.onSuccess(null);
335         else request.onFailure(response);
336       }
337     };
338   }

340   @Override
341   public boolean equals(Object obj) {
342     if (obj == this) return true;
343     if (!(obj instanceof MessageReaction)) return false;
344     MessageReaction r = (MessageReaction) obj;
345     return r.emote.equals(emote) && r.self == self && r.messageId == messageId;
346   }

348   @Override
349   public String toString() {
350     return "MR:(M:(" + messageId + ") / " + emote + ")";
351   }

353   /**
354    * Represents an Emoji/Emote of a MessageReaction <br>
355    * This is used to wrap both emojis and emotes
356    */
357   public static class ReactionEmote implements ISnowflake {
358 
359     private final JDA api;
360     private final String name;
361     private final Long id;
362     private Emote emote = null;
363 
364     public ReactionEmote(String name, Long id, JDA api) {
365       this.name = name;
366       this.id = id;
367       this.api = api;
368     }
369 
370     public ReactionEmote(Emote emote) {
371       this(emote.getName(), emote.getIdLong(), emote.getJDA());
372       this.emote = emote;
373     }
374 
375     /**
376      * Whether this is an {@link net.dv8tion.jda.core.entities.Emote Emote} wrapper.
377      *
378      * @return True, if {@link #getId()} is not null
379      */
380     public boolean isEmote() {
381       return emote != null;
382     }
383 
384     @Override
385     public String getId() {
386       return id != null ? String.valueOf(id) : null;
387     }
388 
389     @Override
390     public long getIdLong() {
391       if (id == null) throw new IllegalStateException("No id available");
392       return id;
393     }
394 
395     /**
396      * The name for this emote/emoji <br>
397      * For unicode emojis this will be the unicode of said emoji.
398      *
399      * @return The name for this emote/emoji
400      */
401     public String getName() {
402       return name;
403     }
404 
405     /**
406      * The instance of {@link net.dv8tion.jda.core.entities.Emote Emote} for the Reaction instance.
407      * <br>
408      * Might be null if {@link #getId()} returns null.
409      *
410      * @return The possibly-null Emote for the Reaction instance
411      */
412     public Emote getEmote() {
413       return emote;
414     }
415 
416     /**
417      * The current JDA instance for the Reaction
418      *
419      * @return The JDA instance of the Reaction
420      */
421     public JDA getJDA() {
422       return api;
423     }
424 
425     @Override
426     public boolean equals(Object obj) {
427       return obj instanceof ReactionEmote
428           && Objects.equals(((ReactionEmote) obj).id, id)
429           && ((ReactionEmote) obj).getName().equals(name);
430     }
431 
432     @Override
433     public String toString() {
434       return "RE:" + (isEmote() ? getEmote() : getName() + "(" + id + ")");
435     }
436   }
