17  package net.dv8tion.jda.core.entities;
18  
19  import java.io.File;
20  import java.io.IOException;
21  import java.io.InputStream;
22  import java.io.UnsupportedEncodingException;
23  import java.util.Base64;
24  import net.dv8tion.jda.core.utils.Checks;
25  import net.dv8tion.jda.core.utils.IOUtil;
26  
27  /**
28   * Icon containing a base64 encoded jpeg/png/gif/gifv image. <br>
29   * Used to represent various base64 images in the Discord api. <br>
30   * Example: {@link net.dv8tion.jda.core.managers.AccountManager#setAvatar(Icon)}.
31   *
32   * @since 3.0
33   */
34  public class Icon {
35    protected final String encoding;
36  
37    protected Icon(String base64Encoding) {
38      // Note: the usage of `image/jpeg` does not mean png/gif are not supported!
39      this.encoding = "data:image/jpeg;base64," + base64Encoding;
40    }
41  
42    /**
43     * The base64 encoded data for this Icon
44     *
45     * @return String representation of the encoded data for this icon
46     */
47    public String getEncoding() {
48      return encoding;
49    }
50  
51    /**
52     * Creates an {@link Icon Icon} with the specified {@link java.io.File File}. <br>
53     * We here read the specified File and forward the retrieved byte data to {@link #from(byte[])}.
54     *
55     * @param file An existing, not-null file.
56     * @throws IllegalArgumentException if the provided file is either null or does not exist
57     * @throws IOException if there is a problem while reading the file.
58     * @return An Icon instance representing the specified File
59     * @see net.dv8tion.jda.core.utils.IOUtil#readFully(File)
60     */
61    public static Icon from(File file) throws IOException {
62      Checks.notNull(file, "Provided File");
63      Checks.check(file.exists(), "Provided file does not exist!");
64  
65      return from(IOUtil.readFully(file));
66    }
67  
68    /**
69     * Creates an {@link Icon Icon} with the specified {@link java.io.InputStream InputStream}. <br>
70     * We here read the specified InputStream and forward the retrieved byte data to {@link
71     * #from(byte[])}.
72     *
73     * @param stream A not-null InputStream.
74     * @throws IllegalArgumentException if the provided stream is null
75     * @throws IOException If the first byte cannot be read for any reason other than the end of the
76     *     file, if the input stream has been closed, or if some other I/O error occurs.
77     * @return An Icon instance representing the specified InputStream
78     * @see net.dv8tion.jda.core.utils.IOUtil#readFully(InputStream)
79     */
80    public static Icon from(InputStream stream) throws IOException {
81      Checks.notNull(stream, "InputStream");
82  
83      return from(IOUtil.readFully(stream));
84    }
85  
86    /**
87     * Creates an {@link Icon Icon} with the specified image data.
88     *
89     * @param data not-null image data bytes.
90     * @throws IllegalArgumentException if the provided data is null
91     * @return An Icon instance representing the specified image data
92     */
93    public static Icon from(byte[] data) {
94      Checks.notNull(data, "Provided byte[]");
95  
96      try {
97        return new Icon(new String(Base64.getEncoder().encode(data), "UTF-8"));
98      } catch (UnsupportedEncodingException e) {
99        throw new AssertionError(e); // thanks JDK 1.4
100     }
101   }
102 }
