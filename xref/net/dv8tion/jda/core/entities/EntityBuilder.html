97    public void createGuildFirstPass(JSONObject guild, Consumer<Guild> secondPassCallback) {
98      final long id = guild.getLong("id");
99      GuildImpl guildObj = ((GuildImpl) api.getGuildMap().get(id));
100     if (guildObj == null) {
101       guildObj = new GuildImpl(api, id);
102       api.getGuildMap().put(id, guildObj);
103     }
104     if (Helpers.optBoolean(guild, "unavailable")) {
105       guildObj.setAvailable(false);
106       // This is used for when GuildCreateHandler receives a guild that is currently unavailable.
107       // During normal READY
108       // loading for bots (which unavailable is always true) the secondPassCallback parameter will
109       // always
110       // be null.
111       if (secondPassCallback != null) secondPassCallback.accept(guildObj);
112       api.getGuildLock().lock(id);
113       return;
114     }
115 
116     // If we make it to here, the Guild is available. This means 1 of 2 things:
117     // Either:
118     // 1) This is Guild provided during READY for a Client account
119     // 2) This is a Guild received from GuildCreateHandler from a GUILD_CREATE event.
120     //      This could be triggered by joining a guild or due to discord finally
121     //      providing us with Guild information about a previously unavailable guild.
122     //      Whether it was unavailable due to Bot READY unavailability or due to an
123     //      outage within discord matters now.
124     //
125     // Either way, we now have enough information to fill in the general information about the
126     // Guild.
127     // This does NOT necessarily mean that we have all information to complete the guild.
128     // For Client accounts, we will also need to use op 12 (GUILD_SYNC) to get all presences of
129     // online users because
130     // discord only provides Online users that we have an open PM channel with or are friends with
131     // for Client accounts.
132     // On larger guilds we will still need to request all users using op 8 (GUILD_MEMBERS_CHUNK).
133     //
134     // The code below takes the information we -do- have and starts to fill in the Guild. It won't
135     // create anything
136     // that might rely on Users that we don't have due to needing the GUILD_MEMBERS_CHUNK
137     // This includes making VoiceStatus and PermissionOverrides
138 
139     guildObj
140         .setAvailable(true)
141         .setIconId(guild.optString("icon", null))
142         .setSplashId(guild.optString("splash", null))
143         .setRegion(guild.getString("region"))
144         .setName(guild.getString("name"))
145         .setAfkTimeout(Guild.Timeout.fromKey(guild.getInt("afk_timeout")))
146         .setVerificationLevel(Guild.VerificationLevel.fromKey(guild.getInt("verification_level")))
147         .setDefaultNotificationLevel(
148             Guild.NotificationLevel.fromKey(guild.getInt("default_message_notifications")))
149         .setRequiredMFALevel(Guild.MFALevel.fromKey(guild.getInt("mfa_level")))
150         .setExplicitContentLevel(
151             Guild.ExplicitContentLevel.fromKey(guild.getInt("explicit_content_filter")));
152 
153     if (guild.isNull("features")) guildObj.setFeatures(Collections.emptySet());
154     else {
155       guildObj.setFeatures(
156           StreamSupport.stream(guild.getJSONArray("features").spliterator(), false)
157               .map(String::valueOf)
158               .collect(Collectors.toSet()));
159     }
160 
161     JSONArray roles = guild.getJSONArray("roles");
162     for (int i = 0; i < roles.length(); i++) {
163       Role role = createRole(roles.getJSONObject(i), guildObj.getIdLong());
164       guildObj.getRolesMap().put(role.getIdLong(), role);
165       if (role.getIdLong() == guildObj.getIdLong()) guildObj.setPublicRole(role);
166     }
167 
168     if (!guild.isNull("emojis")) {
169       JSONArray array = guild.getJSONArray("emojis");
170       TLongObjectMap<Emote> emoteMap = guildObj.getEmoteMap();
171       for (int i = 0; i < array.length(); i++) {
172         JSONObject object = array.getJSONObject(i);
173         if (object.isNull("id")) {
174           LOG.error("Received GUILD_CREATE with an emoji with a null ID. JSON: {}", object);
175           continue;
176         }
177         JSONArray emoteRoles =
178             object.isNull("roles") ? new JSONArray() : object.getJSONArray("roles");
179         final long emoteId = object.getLong("id");
180 
181         EmoteImpl emoteObj = new EmoteImpl(emoteId, guildObj);
182         Set<Role> roleSet = emoteObj.getRoleSet();
183 
184         for (int j = 0; j < emoteRoles.length(); j++)
185           roleSet.add(guildObj.getRoleById(emoteRoles.getString(j)));
186         emoteMap.put(
187             emoteId,
188             emoteObj
189                 .setName(object.optString("name"))
190                 .setAnimated(object.optBoolean("animated"))
191                 .setManaged(Helpers.optBoolean(object, "managed")));
192       }
193     }
194 
195     if (guild.has("members")) {
196       JSONArray members = guild.getJSONArray("members");
197       createGuildMemberPass(guildObj, members);
198     }
199 
200     // This could be null for Client accounts. Will be fixed by GUILD_SYNC
201     Member owner = guildObj.getMemberById(guild.getLong("owner_id"));
202     if (owner != null) guildObj.setOwner(owner);
203 
204     if (guild.has("presences")) {
205       JSONArray presences = guild.getJSONArray("presences");
206       for (int i = 0; i < presences.length(); i++) {
207         JSONObject presence = presences.getJSONObject(i);
208         final long userId = presence.getJSONObject("user").getLong("id");
209         MemberImpl member = (MemberImpl) guildObj.getMembersMap().get(userId);
210 
211         if (member == null)
212           LOG.debug(
213               "Received a ghost presence in GuildFirstPass! UserId: {} Guild: {}",
214               userId,
215               guildObj);
216         else createPresence(member, presence);
217       }
218     }
219 
220     if (guild.has("channels")) {
221       JSONArray channels = guild.getJSONArray("channels");
222 
223       for (int i = 0; i < channels.length(); i++) {
224         JSONObject channel = channels.getJSONObject(i);
225         ChannelType type = ChannelType.fromId(channel.getInt("type"));
226         switch (type) {
227           case TEXT:
228             createTextChannel(channel, guildObj.getIdLong(), false);
229             break;
230           case VOICE:
231             createVoiceChannel(channel, guildObj.getIdLong(), false);
232             break;
233           case CATEGORY:
234             createCategory(channel, guildObj.getIdLong(), false);
235             break;
236           default:
237             LOG.error(
238                 "Received a channel for a guild that isn't a text, voice or category channel. JSON: {}",
239                 channel);
240         }
241       }
242     }
243 
244     if (!guild.isNull("system_channel_id"))
245       guildObj.setSystemChannel(
246           guildObj.getTextChannelsMap().get(guild.getLong("system_channel_id")));
247 
248     if (!guild.isNull("afk_channel_id"))
249       guildObj.setAfkChannel(guildObj.getVoiceChannelsMap().get(guild.getLong("afk_channel_id")));
250 
251     // If the members that we were provided with (and loaded above) were not all of the
252     //  the members in this guild, then we need to request more users from Discord using
253     //  op 9 (GUILD_MEMBERS_CHUNK). To do so, we will cache the guild's JSON so we can properly
254     //  load stuff that relies on Users like Channels, PermissionOverrides and VoiceStatuses
255     //  after we have the rest of the users. We will request the GUILD_MEMBERS_CHUNK information
256     //  which will be sent from discord over the main Websocket and will be handled by
257     //  GuildMemberChunkHandler. After the handler has received all users as determined by the
258     //  value set using `setExpectedGuildMembers`, it will do one of the following:
259     //    1) If this is a Bot account, immediately call EntityBuilder#createGuildSecondPass, thus
260     // finishing
261     //        the Guild object creation process.
262     //    2) If this is a Client account, it will request op 12 (GUILD_SYNC) to make sure we have
263     // all information
264     //        about online users as GUILD_MEMBERS_CHUNK does not include presence information, and
265     // when loading the
266     //        members from GUILD_MEMBERS_CHUNK, we assume they are offline. GUILD_SYNC makes sure
267     // that we mark them
268     //        properly. After GUILD_SYNC is received by GuildSyncHandler, it will call
269     // EntityBuilder#createGuildSecondPass
270     //
271     // If we actually -did- get all of the users needed, then we don't need to Chunk. Furthermore,
272     // we don't need to use GUILD_SYNC because we always get presences with users thus we have all
273     // information
274     // needed to guild the Guild. We will skip
275     if (guild.getJSONArray("members").length() != guild.getInt("member_count")) {
276       cachedGuildJsons.put(id, guild);
277       cachedGuildCallbacks.put(id, secondPassCallback);
278 
279       GuildMembersChunkHandler handler = api.getClient().getHandler("GUILD_MEMBERS_CHUNK");
280       handler.setExpectedGuildMembers(id, guild.getInt("member_count"));
281 
282       // If we are already past READY / RESUME, then chunk at runtime. Otherwise, pass back to the
283       // ReadyHandler
284       // and let it send a burst chunk request.
285       if (api.getClient().isReady()) {
286         JSONObject obj =
287             new JSONObject()
288                 .put("op", WebSocketCode.MEMBER_CHUNK_REQUEST)
289                 .put("d", new JSONObject().put("guild_id", id).put("query", "").put("limit", 0));
290         api.getClient().chunkOrSyncRequest(obj);
291       } else {
292         ReadyHandler readyHandler = api.getClient().getHandler("READY");
293         readyHandler.acknowledgeGuild(guildObj, true, true, false);
294       }
295 
296       api.getGuildLock().lock(id);
297       return;
298     }
299 
300     // As detailed in the comment above, if we've made it this far then we have all member
301     // information needed to
302     // create the Guild. Thus, we fill in the remaining information, unlock the guild, and provide
303     // the guild
304     // to the callback
305     // This should only occur on small user count guilds.
306 
307     JSONArray channels = guild.getJSONArray("channels");
308     createGuildChannelPass(guildObj, channels); // Actually creates PermissionOverrides
309 
310     JSONArray voiceStates = guild.getJSONArray("voice_states");
311     createGuildVoiceStatePass(guildObj, voiceStates);
312 
313     api.getGuildLock().unlock(guildObj.getIdLong());
314     if (secondPassCallback != null) secondPassCallback.accept(guildObj);
315   }

317   public void createGuildSecondPass(long guildId, List<JSONArray> memberChunks) {
318     JSONObject guildJson = cachedGuildJsons.remove(guildId);
319     Consumer<Guild> secondPassCallback = cachedGuildCallbacks.remove(guildId);
320     GuildImpl guildObj = (GuildImpl) api.getGuildMap().get(guildId);
321 
322     if (guildObj == null)
323       throw new IllegalStateException(
324           "Attempted to perform a second pass on an unknown Guild. Guild not in JDA "
325               + "mapping. GuildId: "
326               + guildId);
327     if (guildJson == null)
328       throw new IllegalStateException(
329           "Attempted to perform a second pass on an unknown Guild. No cached Guild "
330               + "for second pass. GuildId: "
331               + guildId);
332     if (secondPassCallback == null)
333       throw new IllegalArgumentException("No callback provided for the second pass on the Guild!");
334 
335     for (JSONArray chunk : memberChunks) createGuildMemberPass(guildObj, chunk);
336 
337     Member owner = guildObj.getMemberById(guildJson.getLong("owner_id"));
338     if (owner != null) guildObj.setOwner(owner);
339 
340     if (guildObj.getOwner() == null)
341       LOG.error(
342           "Never set the Owner of the Guild: {} because we don't have the owner User object! How?!",
343           guildObj.getId());
344 
345     JSONArray channels = guildJson.getJSONArray("channels");
346     createGuildChannelPass(guildObj, channels);
347 
348     JSONArray voiceStates = guildJson.getJSONArray("voice_states");
349     createGuildVoiceStatePass(guildObj, voiceStates);
350 
351     secondPassCallback.accept(guildObj);
352     api.getGuildLock().unlock(guildId);
353   }

356     for (int i = 0; i < members.length(); i++) {

365       MemberImpl member = (MemberImpl) guild.getMembersMap().get(userId);
366       if (member == null)
367         LOG.error("Received a Presence for a non-existent Member when dealing with GuildSync!");
368       else this.createPresence(member, presenceJson);
369     }
370   }

379   private void createGuildChannelPass(GuildImpl guildObj, JSONArray channels) {
380     for (int i = 0; i < channels.length(); i++) {
381       JSONObject channel = channels.getJSONObject(i);
382       ChannelType type = ChannelType.fromId(channel.getInt("type"));
383       Channel channelObj = null;
384       switch (type) {
385         case TEXT:
386           channelObj = api.getTextChannelById(channel.getLong("id"));
387           break;
388         case VOICE:
389           channelObj = api.getVoiceChannelById(channel.getLong("id"));
390           break;
391         case CATEGORY:
392           channelObj = api.getCategoryMap().get(channel.getLong("id"));
393           break;
394         default:
395           LOG.error(
396               "Received a channel for a guild that isn't a text, voice or category channel (ChannelPass). JSON: {}",
397               channel);
398       }
399 
400       if (channelObj != null) {
401         JSONArray permissionOverwrites = channel.getJSONArray("permission_overwrites");
402         createOverridesPass((AbstractChannelImpl<?>) channelObj, permissionOverwrites);
403       } else {
404         LOG.error(
405             "Got permission_override for unknown channel with id: {}", channel.getString("id"));
406       }
407     }
408   }

410   public void createGuildVoiceStatePass(GuildImpl guildObj, JSONArray voiceStates) {
411     for (int i = 0; i < voiceStates.length(); i++) {
412       JSONObject voiceStateJson = voiceStates.getJSONObject(i);
413       final long userId = voiceStateJson.getLong("user_id");
414       Member member = guildObj.getMembersMap().get(userId);
415       if (member == null) {
416         LOG.error(
417             "Received a VoiceState for a unknown Member! GuildId: "
418                 + guildObj.getId()
419                 + " MemberId: "
420                 + voiceStateJson.getString("user_id"));
421         continue;
422       }
423 
424       final long channelId = voiceStateJson.getLong("channel_id");
425       VoiceChannelImpl voiceChannel =
426           (VoiceChannelImpl) guildObj.getVoiceChannelsMap().get(channelId);
427       if (voiceChannel != null)
428         voiceChannel.getConnectedMembersMap().put(member.getUser().getIdLong(), member);
429       else
430         LOG.error(
431             "Received a GuildVoiceState with a channel ID for a non-existent channel! ChannelId: {} GuildId: {} UserId: {}",
432             channelId,
433             guildObj.getId(),
434             userId);
435 
436       // VoiceState is considered volatile so we don't expect anything to actually exist
437       GuildVoiceStateImpl voiceState = (GuildVoiceStateImpl) member.getVoiceState();
438       voiceState
439           .setSelfMuted(Helpers.optBoolean(voiceStateJson, "self_mute"))
440           .setSelfDeafened(Helpers.optBoolean(voiceStateJson, "self_deaf"))
441           .setGuildMuted(Helpers.optBoolean(voiceStateJson, "mute"))
442           .setGuildDeafened(Helpers.optBoolean(voiceStateJson, "deaf"))
443           .setSuppressed(Helpers.optBoolean(voiceStateJson, "suppress"))
444           .setSessionId(voiceStateJson.optString("session_id"))
445           .setConnectedChannel(voiceChannel);
446     }
447   }

457   private User createUser(JSONObject user, boolean fake, boolean modifyCache) {
458     final long id = user.getLong("id");
459     UserImpl userObj;
460 
461     userObj = (UserImpl) api.getUserMap().get(id);
462     if (userObj == null) {
463       userObj = (UserImpl) api.getFakeUserMap().get(id);
464       if (userObj != null) {
465         if (!fake && modifyCache) {
466           api.getFakeUserMap().remove(id);
467           userObj.setFake(false);
468           api.getUserMap().put(userObj.getIdLong(), userObj);
469           if (userObj.hasPrivateChannel()) {
470             PrivateChannelImpl priv = (PrivateChannelImpl) userObj.getPrivateChannel();
471             priv.setFake(false);
472             api.getFakePrivateChannelMap().remove(priv.getIdLong());
473             api.getPrivateChannelMap().put(priv.getIdLong(), priv);
474           }
475         }
476       } else {
477         userObj = new UserImpl(id, api).setFake(fake);
478         if (modifyCache) {
479           if (fake) api.getFakeUserMap().put(id, userObj);
480           else api.getUserMap().put(id, userObj);
481         }
482       }
483     }
484 
485     return userObj
486         .setName(user.getString("username"))
487         .setDiscriminator(user.get("discriminator").toString())
488         .setAvatarId(user.optString("avatar", null))
489         .setBot(Helpers.optBoolean(user, "bot"));
490   }

492   public Member createMember(GuildImpl guild, JSONObject memberJson) {
493     User user = createUser(memberJson.getJSONObject("user"));
494     MemberImpl member = (MemberImpl) guild.getMember(user);
495     if (member == null) {
496       member = new MemberImpl(guild, user);
497       guild.getMembersMap().put(user.getIdLong(), member);
498     }
499 
500     ((GuildVoiceStateImpl) member.getVoiceState())
501         .setGuildMuted(memberJson.getBoolean("mute"))
502         .setGuildDeafened(memberJson.getBoolean("deaf"));
503 
504     member
505         .setJoinDate(OffsetDateTime.parse(memberJson.getString("joined_at")))
506         .setNickname(memberJson.optString("nick", null));
507 
508     JSONArray rolesJson = memberJson.getJSONArray("roles");
509     for (int k = 0; k < rolesJson.length(); k++) {
510       final long roleId = rolesJson.getLong(k);
511       Role r = guild.getRolesMap().get(roleId);
512       if (r == null) {
513         LOG.debug(
514             "Received a Member with an unknown Role. MemberId: {} GuildId: {} roleId: {}",
515             member.getUser().getId(),
516             guild.getId(),
517             roleId);
518       } else {
519         member.getRoleSet().add(r);
520       }
521     }
522 
523     return member;
524   }

526   // Effectively the same as createFriendPresence
527   public void createPresence(Object memberOrFriend, JSONObject presenceJson) {
528     if (memberOrFriend == null) throw new NullPointerException("Provided memberOrFriend was null!");
529 
530     JSONObject gameJson = presenceJson.isNull("game") ? null : presenceJson.getJSONObject("game");
531     OnlineStatus onlineStatus = OnlineStatus.fromKey(presenceJson.getString("status"));
532     Game game = null;
533     boolean parsedGame = false;
534 
535     if (gameJson != null && !gameJson.isNull("name")) {
536       try {
537         game = createGame(gameJson);
538         parsedGame = true;
539       } catch (Exception ex) {
540         String userId;
541         if (memberOrFriend instanceof Member) userId = ((Member) memberOrFriend).getUser().getId();
542         else if (memberOrFriend instanceof Friend)
543           userId = ((Friend) memberOrFriend).getUser().getId();
544         else userId = "unknown";
545         if (LOG.isDebugEnabled())
546           LOG.warn(
547               "Encountered exception trying to parse a presence! UserId: {} JSON: {}",
548               userId,
549               gameJson,
550               ex);
551         else
552           LOG.warn(
553               "Encountered exception trying to parse a presence! UserId: {} Message: {} Enable debug for details",
554               userId,
555               ex.getMessage());
556       }
557     }
558     if (memberOrFriend instanceof Member) {
559       MemberImpl member = (MemberImpl) memberOrFriend;
560       member.setOnlineStatus(onlineStatus);
561       if (parsedGame) member.setGame(game);
562     } else if (memberOrFriend instanceof Friend) {
563       FriendImpl friend = (FriendImpl) memberOrFriend;
564       friend.setOnlineStatus(onlineStatus);
565       if (parsedGame) friend.setGame(game);
566 
567       OffsetDateTime lastModified =
568           OffsetDateTime.ofInstant(
569               Instant.ofEpochMilli(presenceJson.getLong("last_modified")),
570               TimeZone.getTimeZone("GMT").toZoneId());
571 
572       friend.setOnlineStatusModifiedTime(lastModified);
573     } else
574       throw new IllegalArgumentException(
575           "An object was provided to EntityBuilder#createPresence that wasn't a Member or Friend. JSON: "
576               + presenceJson);
577   }

579   public static Game createGame(JSONObject gameJson) {
580     String name = String.valueOf(gameJson.get("name"));
581     String url = gameJson.isNull("url") ? null : String.valueOf(gameJson.get("url"));
582     Game.GameType type;
583     try {
584       type =
585           gameJson.isNull("type")
586               ? Game.GameType.DEFAULT
587               : Game.GameType.fromKey(Integer.parseInt(gameJson.get("type").toString()));
588     } catch (NumberFormatException e) {
589       type = Game.GameType.DEFAULT;
590     }
591 
592     long id;
593     try {
594       id = gameJson.getLong("application_id");
595     } catch (JSONException ex) {
596       return new Game(name, url, type);
597     }
598     String details = gameJson.isNull("details") ? null : String.valueOf(gameJson.get("details"));
599     String state = gameJson.isNull("state") ? null : String.valueOf(gameJson.get("state"));
600     RichPresence.Timestamps timestamps = null;
601     if (!gameJson.isNull("timestamps")) {
602       JSONObject obj = gameJson.getJSONObject("timestamps");
603       long start, end;
604       start = obj.isNull("start") ? 0 : obj.getLong("start");
605       end = obj.isNull("end") ? 0 : obj.getLong("end");
606       timestamps = new RichPresence.Timestamps(start, end);
607     }
608 
609     RichPresence.Party party = null;
610     if (!gameJson.isNull("party")) {
611       JSONObject obj = gameJson.getJSONObject("party");
612       String partyId = obj.isNull("id") ? null : obj.getString("id");
613       JSONArray sizeArr = obj.isNull("size") ? null : obj.getJSONArray("size");
614       long size = 0, max = 0;
615       if (sizeArr != null && sizeArr.length() > 0) {
616         size = sizeArr.getLong(0);
617         max = sizeArr.isNull(1) ? 0 : sizeArr.getLong(1);
618       }
619       party = new RichPresence.Party(partyId, size, max);
620     }
621 
622     String smallImageKey = null, smallImageText = null;
623     String largeImageKey = null, largeImageText = null;
624     if (!gameJson.isNull("assets")) {
625       JSONObject assets = gameJson.getJSONObject("assets");
626       if (!assets.isNull("small_image")) {
627         smallImageKey = String.valueOf(assets.get("small_image"));
628         smallImageText =
629             assets.isNull("small_text") ? null : String.valueOf(assets.get("small_text"));
630       }
631       if (!assets.isNull("large_image")) {
632         largeImageKey = String.valueOf(assets.get("large_image"));
633         largeImageText =
634             assets.isNull("large_text") ? null : String.valueOf(assets.get("large_text"));
635       }
636     }
637     return new RichPresence(
638         type,
639         name,
640         url,
641         id,
642         party,
643         details,
644         state,
645         timestamps,
646         largeImageKey,
647         largeImageText,
648         smallImageKey,
649         smallImageText);
650   }

756   public void createOverridesPass(AbstractChannelImpl<?> channel, JSONArray overrides) {
757     for (int i = 0; i < overrides.length(); i++) {
758       try {
759         createPermissionOverride(overrides.getJSONObject(i), channel);
760       } catch (NoSuchElementException e) {
761         // Caused by Discord not properly clearing PermissionOverrides when a Member leaves a Guild.
762         LOG.debug("{}. Ignoring PermissionOverride.", e.getMessage());
763       } catch (IllegalArgumentException e) {
764         // Missing handling for a type
765         LOG.warn("{}. Ignoring PermissionOverride.", e.getMessage());
766       }
767     }

802   public Message createMessage(
803       JSONObject jsonObject, MessageChannel chan, boolean exceptionOnMissingUser) {
804     final long id = jsonObject.getLong("id");
805     String content = jsonObject.optString("content");
806 
807     JSONObject author = jsonObject.getJSONObject("author");
808     final long authorId = author.getLong("id");
809     final boolean fromWebhook = jsonObject.has("webhook_id");
810     final boolean pinned = Helpers.optBoolean(jsonObject, "pinned");
811     final boolean tts = Helpers.optBoolean(jsonObject, "tts");
812     final boolean mentionsEveryone = Helpers.optBoolean(jsonObject, "mention_everyone");
813     final OffsetDateTime editTime =
814         jsonObject.isNull("edited_timestamp")
815             ? null
816             : OffsetDateTime.parse(jsonObject.getString("edited_timestamp"));
817     final String nonce = jsonObject.isNull("nonce") ? null : jsonObject.get("nonce").toString();
818 
819     final List<Message.Attachment> attachments =
820         map(jsonObject, "attachments", this::createMessageAttachment);
821     final List<MessageEmbed> embeds = map(jsonObject, "embeds", this::createMessageEmbed);
822     final List<MessageReaction> reactions =
823         map(jsonObject, "reactions", (obj) -> createMessageReaction(chan, id, obj));
824 
825     User user;
826     switch (chan.getType()) {
827       case PRIVATE:
828         if (authorId == api.getSelfUser().getIdLong()) user = api.getSelfUser();
829         else user = ((PrivateChannel) chan).getUser();
830         break;
831       case GROUP:
832         user = api.getUserById(authorId);
833         if (user == null) user = api.getFakeUserMap().get(authorId);
834         if (user == null && fromWebhook) user = createFakeUser(author, false);
835         if (user == null) {
836           if (exceptionOnMissingUser)
837             throw new IllegalArgumentException(MISSING_USER); // Specifically for MESSAGE_CREATE
838           else user = createFakeUser(author, false); // Any other message creation
839         }
840 
841         if (user.isFake() && !fromWebhook) {
842           UserImpl impl = (UserImpl) user;
843           impl.setName(author.getString("username"))
844               .setDiscriminator(author.get("discriminator").toString())
845               .setAvatarId(author.optString("avatar", null))
846               .setBot(Helpers.optBoolean(author, "bot"));
847         }
848         break;
849       case TEXT:
850         Guild guild = ((TextChannel) chan).getGuild();
851         Member member = guild.getMemberById(authorId);
852         user = member != null ? member.getUser() : null;
853         if (user == null) {
854           if (fromWebhook || !exceptionOnMissingUser) user = createFakeUser(author, false);
855           else throw new IllegalArgumentException(MISSING_USER); // Specifically for MESSAGE_CREATE
856         }
857         break;
858       default:
859         throw new IllegalArgumentException(
860             "Invalid Channel for creating a Message [" + chan.getType() + ']');
861     }
862 
863     TLongSet mentionedRoles = new TLongHashSet();
864     TLongSet mentionedUsers = new TLongHashSet(map(jsonObject, "mentions", (o) -> o.getLong("id")));
865     JSONArray roleMentionArr = jsonObject.optJSONArray("mention_roles");
866     if (roleMentionArr != null) {
867       for (int i = 0; i < roleMentionArr.length(); i++)
868         mentionedRoles.add(roleMentionArr.getLong(i));
869     }
870 
871     MessageType type = MessageType.fromId(jsonObject.getInt("type"));
872     switch (type) {
873       case DEFAULT:
874         return new ReceivedMessage(
875             id,
876             chan,
877             type,
878             fromWebhook,
879             mentionsEveryone,
880             mentionedUsers,
881             mentionedRoles,
882             tts,
883             pinned,
884             content,
885             nonce,
886             user,
887             editTime,
888             reactions,
889             attachments,
890             embeds);
891       case UNKNOWN:
892         throw new IllegalArgumentException(UNKNOWN_MESSAGE_TYPE);
893       default:
894         return new SystemMessage(
895             id,
896             chan,
897             type,
898             fromWebhook,
899             mentionsEveryone,
900             mentionedUsers,
901             mentionedRoles,
902             tts,
903             pinned,
904             content,
905             nonce,
906             user,
907             editTime,
908             reactions,
909             attachments,
910             embeds);
911     }
912   }

1086   public PermissionOverride createPermissionOverride(JSONObject override, Channel chan) {
1087     PermissionOverrideImpl permOverride;
1088     final long id = override.getLong("id");
1089     long allow = override.getLong("allow");
1090     long deny = override.getLong("deny");
1091 
1092     // Throwing NoSuchElementException for common issues with overrides that are not cleared
1093     // properly by discord
1094     // when a member leaves or a role is deleted
1095     switch (override.getString("type")) {
1096       case "member":
1097         Member member = chan.getGuild().getMemberById(id);
1098         if (member == null)
1099           throw new NoSuchElementException(
1100               "Attempted to create a PermissionOverride for a non-existent user. Guild: "
1101                   + chan.getGuild()
1102                   + ", Channel: "
1103                   + chan
1104                   + ", JSON: "
1105                   + override);
1106 
1107         permOverride = (PermissionOverrideImpl) chan.getPermissionOverride(member);
1108         if (permOverride == null) {
1109           permOverride = new PermissionOverrideImpl(chan, member.getUser().getIdLong(), member);
1110           ((AbstractChannelImpl<?>) chan)
1111               .getOverrideMap()
1112               .put(member.getUser().getIdLong(), permOverride);
1113         }
1114         break;
1115       case "role":
1116         Role role = ((GuildImpl) chan.getGuild()).getRolesMap().get(id);
1117         if (role == null)
1118           throw new NoSuchElementException(
1119               "Attempted to create a PermissionOverride for a non-existent role! JSON: "
1120                   + override);
1121 
1122         permOverride = (PermissionOverrideImpl) chan.getPermissionOverride(role);
1123         if (permOverride == null) {
1124           permOverride = new PermissionOverrideImpl(chan, role.getIdLong(), role);
1125           ((AbstractChannelImpl<?>) chan).getOverrideMap().put(role.getIdLong(), permOverride);
1126         }
1127         break;
1128       default:
1129         throw new IllegalArgumentException(
1130             "Provided with an unknown PermissionOverride type! JSON: " + override);
1131     }
1132     return permOverride.setAllow(allow).setDeny(deny);
1133   }

1319   public AuditLogChange createAuditLogChange(JSONObject change) {
1320     final String key = change.getString("key");
1321     Object oldValue = change.isNull("old_value") ? null : change.get("old_value");
1322     Object newValue = change.isNull("new_value") ? null : change.get("new_value");
1323 
1324     // Don't confront users with JSON
1325     if (oldValue instanceof JSONArray || newValue instanceof JSONArray) {
1326       oldValue = oldValue instanceof JSONArray ? ((JSONArray) oldValue).toList() : oldValue;
1327       newValue = newValue instanceof JSONArray ? ((JSONArray) newValue).toList() : newValue;
1328     } else if (oldValue instanceof JSONObject || newValue instanceof JSONObject) {
1329       oldValue = oldValue instanceof JSONObject ? ((JSONObject) oldValue).toMap() : oldValue;
1330       newValue = newValue instanceof JSONObject ? ((JSONObject) newValue).toMap() : newValue;
1331     }
1332 
1333     return new AuditLogChange(oldValue, newValue, key);
1334   }

1342   private <T> List<T> map(JSONObject jsonObject, String key, Function<JSONObject, T> convert) {
1343     if (jsonObject.isNull(key)) return Collections.emptyList();
1344 
1345     final JSONArray arr = jsonObject.getJSONArray(key);
1346     final List<T> mappedObjects = new ArrayList<>(arr.length());
1347     for (int i = 0; i < arr.length(); i++) {
1348       JSONObject obj = arr.getJSONObject(i);
1349       mappedObjects.add(convert.apply(obj));
1350     }
1351 
1352     return mappedObjects;
1353   }
