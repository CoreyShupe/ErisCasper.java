17  package net.dv8tion.jda.core.entities;
18  
19  /**
20   * Represents a Discord Voice Channel. A specification of {@link
21   * net.dv8tion.jda.core.entities.AudioChannel}. <br>
22   * As a VoiceChannel has no extra functionality as its {@link net.dv8tion.jda.core.entities.Channel
23   * Channel} parent, this interface is empty. This interface only exists to distinct {@link
24   * net.dv8tion.jda.core.entities.Channel Channels} into VoiceChannels and {@link
25   * net.dv8tion.jda.core.entities.TextChannel TextChannels}.
26   */
27  public interface VoiceChannel extends Channel, AudioChannel, Comparable<VoiceChannel> {
28    /**
29     * The maximum amount of {@link net.dv8tion.jda.core.entities.Member Members} that can be in this
30     * {@link net.dv8tion.jda.core.entities.VoiceChannel VoiceChannel} at once. <br>
31     * 0 - No limit
32     *
33     * @return The maximum amount of members allowed in this channel at once.
34     */
35    int getUserLimit();
36  
37    /**
38     * The audio bitrate of the voice audio that is transmitted in this channel. While higher bitrates
39     * can be sent to this channel, it will be scaled down by the client. <br>
40     * Default and recommended value is 64000
41     *
42     * @return The audio bitrate of this voice channel.
43     */
44    int getBitrate();
45  }
