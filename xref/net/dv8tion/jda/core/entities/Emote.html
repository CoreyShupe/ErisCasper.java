17  package net.dv8tion.jda.core.entities;
18  
19  import java.util.List;
20  import net.dv8tion.jda.client.managers.EmoteManager;
21  import net.dv8tion.jda.client.managers.EmoteManagerUpdatable;
22  import net.dv8tion.jda.core.JDA;
23  import net.dv8tion.jda.core.requests.restaction.AuditableRestAction;
24  import net.dv8tion.jda.core.utils.PermissionUtil;
25  
26  /**
27   * Represents a Custom Emote. (Custom Emoji in official Discord API terminology)
28   *
29   * <p><b>This does not represent unicode emojis like they are used in the official client! (:smiley:
30   * is not a custom emoji)</b>
31   *
32   * @since 2.2
33   * @author Florian Spieß
34   */
35  public interface Emote extends ISnowflake, IMentionable, IFakeable {
36  
37    /**
38     * The {@link net.dv8tion.jda.core.entities.Guild Guild} this emote is attached to.
39     *
40     * <p><b>This is null if the emote is fake (retrieved from a Message)</b>
41     *
42     * @return Guild of this emote or null if it is a fake entity
43     */
44    Guild getGuild();
45  
46    /**
47     * Roles this emote is active for <br>
48     * <a href="https://discordapp.com/developers/docs/resources/guild#emoji-object"
49     * target="_blank">Learn More</a>
50     *
51     * @throws IllegalStateException If this Emote is fake ({@link #isFake()})
52     * @return An immutable list of the roles this emote is active for (all roles if empty)
53     */
54    List<Role> getRoles();
55  
56    /**
57     * The name of this emote
58     *
59     * @return String representation of this emote's name
60     */
61    String getName();
62  
63    /**
64     * Whether this emote is managed. A managed Emote is controlled by Discord, not the Guild
65     * administrator, typical via a service like BBTV in conjunction with Twitch. <br>
66     * <a href="https://discordapp.com/developers/docs/resources/guild#emoji-object"
67     * target="_blank">Learn More</a>
68     *
69     * @return True, if this emote is managed
70     */
71    boolean isManaged();
72  
73    /**
74     * The {@link net.dv8tion.jda.core.JDA JDA} instance of this Emote
75     *
76     * @return The JDA instance of this Emote
77     */
78    JDA getJDA();
79  
80    /**
81     * Deletes this Emote.
82     *
83     * <p>Possible ErrorResponses include:
84     *
85     * <ul>
86     *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#UNKNOWN_EMOJI UNKNOWN_EMOTE} <br>
87     *       If this Emote was already removed
88     *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#UNKNOWN_GUILD UNKNOWN_GUILD} <br>
89     *       If the Guild of this Emote was deleted
90     *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#MISSING_ACCESS MISSING_ACCESS} <br>
91     *       If we were removed from the Guild
92     * </ul>
93     *
94     * @throws IllegalStateException if this Emote is fake ({@link #isFake()})
95     * @throws java.lang.UnsupportedOperationException If this emote is managed by discord ({@link
96     *     #isManaged()})
97     * @throws net.dv8tion.jda.core.exceptions.InsufficientPermissionException if the Permission
98     *     {@link net.dv8tion.jda.core.Permission#MANAGE_EMOTES MANAGE_EMOTES} is not given
99     * @return {@link net.dv8tion.jda.core.requests.restaction.AuditableRestAction
100    *     AuditableRestAction} The RestAction to delete this Emote.
101    */
102   AuditableRestAction<Void> delete();
103 
104   /**
105    * The {@link net.dv8tion.jda.client.managers.EmoteManager Manager} for this emote, used to modify
106    * properties of the emote like name and role restrictions.
107    *
108    * <p>This will only work for {@link net.dv8tion.jda.core.AccountType#CLIENT AccountType.CLIENT}.
109    *
110    * @throws IllegalStateException if this emote is fake
111    * @return The EmoteManager for this Emote
112    */
113   EmoteManager getManager();
114 
115   /**
116    * An <b>updatable</b> manager for this Emote, used to modify properties of the emote like name
117    * and role restrictions.
118    *
119    * <p>This will only work for the client account type.
120    *
121    * <p>This specific Manager is used to modify multiple properties at once by setting the property
122    * and calling {@link EmoteManagerUpdatable#update()}
123    *
124    * @throws IllegalStateException if this emote is fake
125    * @return The EmoteManagerUpdatable for this Emote
126    */
127   EmoteManagerUpdatable getManagerUpdatable();
128 
129   /**
130    * Whether or not this Emote is animated.
131    *
132    * <p>Animated Emotes are available to Discord Nitro users as well as Bot accounts.
133    *
134    * @return Whether the Emote is animated or not.
135    */
136   boolean isAnimated();
137 
138   /**
139    * A String representation of the URL which leads to image displayed within the official
140    * Discord&trade; client when this Emote is used
141    *
142    * @return Discord CDN link to the Emote's image
143    */
144   default String getImageUrl() {
145     return "https://cdn.discordapp.com/emojis/" + getId() + (isAnimated() ? ".gif" : ".png");
146   }
147 
148   /**
149    * Usable representation of this Emote (used to display in the client just like mentions with a
150    * specific format) <br>
151    * Emotes are used with the format <code>
152    * &lt;:{@link #getName getName()}:{@link #getId getId()}&gt;</code>
153    *
154    * @return A usable String representation for this Emote
155    * @see <a
156    *     href="https://discordapp.com/developers/docs/resources/channel#message-formatting">Message
157    *     Formatting</a>
158    */
159   @Override
160   default String getAsMention() {
161     return (isAnimated() ? "<a:" : "<:") + getName() + ":" + getIdLong() + ">";
162   }
163 
164   /**
165    * Whether the specified Member can interact with this Emote
166    *
167    * @param issuer The User to test
168    * @return True, if the provided Member can use this Emote
169    * @see net.dv8tion.jda.core.utils.PermissionUtil#canInteract(Member, Emote)
170    * @see net.dv8tion.jda.core.utils.PermissionUtil#canInteract(User, Emote, MessageChannel)
171    */
172   default boolean canInteract(Member issuer) {
173     return PermissionUtil.canInteract(issuer, this);
174   }
175 
176   /**
177    * Whether the specified Member can interact with this Emote within the provided MessageChannel
178    * <br>
179    * Same logic as {@link #canInteract(User, MessageChannel, boolean) canInteract(issuer, channel,
180    * true)}!
181    *
182    * @param issuer The User to test
183    * @param channel The MessageChannel to test
184    * @return True, if the provided Member can use this Emote
185    * @see net.dv8tion.jda.core.utils.PermissionUtil#canInteract(Member, Emote)
186    * @see net.dv8tion.jda.core.utils.PermissionUtil#canInteract(User, Emote, MessageChannel)
187    */
188   default boolean canInteract(User issuer, MessageChannel channel) {
189     return PermissionUtil.canInteract(issuer, this, channel);
190   }
191 
192   /**
193    * Whether the specified Member can interact with this Emote within the provided MessageChannel
194    * <br>
195    * Special override to exclude elevated bot permissions in case of (for instance) reacting to
196    * messages.
197    *
198    * @param issuer The User to test
199    * @param channel The MessageChannel to test
200    * @param botOverride Whether bots can use non-managed emotes in other guilds
201    * @return True, if the provided Member can use this Emote
202    * @see net.dv8tion.jda.core.utils.PermissionUtil#canInteract(Member, Emote)
203    * @see net.dv8tion.jda.core.utils.PermissionUtil#canInteract(User, Emote, MessageChannel,
204    *     boolean)
205    */
206   default boolean canInteract(User issuer, MessageChannel channel, boolean botOverride) {
207     return PermissionUtil.canInteract(issuer, this, channel, botOverride);
208   }
209 }
