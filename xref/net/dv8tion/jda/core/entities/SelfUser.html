17  package net.dv8tion.jda.core.entities;
18  
19  import net.dv8tion.jda.core.exceptions.AccountTypeException;
20  import net.dv8tion.jda.core.managers.AccountManager;
21  import net.dv8tion.jda.core.managers.AccountManagerUpdatable;
22  
23  /** Represents the currently logged in account. */
24  public interface SelfUser extends User {
25  
26    /**
27     * The status of this account's verification. (Have you accepted the verification email)
28     *
29     * @return True, if this account is verified.
30     */
31    boolean isVerified();
32  
33    /**
34     * If true, this account is protected by Multi-Factor authorization. <br>
35     * If this is a Client account, then this describes the MFA status of the Client account. <br>
36     * If this is a Bot account, then this describes the MFA status of the Client account that owns
37     * this Bot.
38     *
39     * @return True, if this account has MFA protecting it.
40     */
41    boolean isMfaEnabled();
42  
43    /**
44     * Used to get the email of the currently logged in account. <br>
45     * <b>NOTE:</b> this is a {@link net.dv8tion.jda.core.AccountType#CLIENT AccountType.CLIENT}
46     * method only!
47     *
48     * @throws AccountTypeException If this method is called when {@link
49     *     net.dv8tion.jda.core.JDA#getAccountType() JDA#getAccountType()} does not return {@link
50     *     net.dv8tion.jda.core.AccountType#CLIENT AccountType.CLIENT}. E.g: If the logged in account
51     *     isn't a Client account!
52     * @return The email of the currently logged in account.
53     */
54    String getEmail() throws AccountTypeException;
55  
56    /**
57     * Shows whether there has ever been a mobile app connected to this account. <br>
58     * <b>NOTE:</b> this is a {@link net.dv8tion.jda.core.AccountType#CLIENT AccountType.CLIENT}
59     * method only!
60     *
61     * @throws AccountTypeException If this method is called when {@link
62     *     net.dv8tion.jda.core.JDA#getAccountType() JDA#getAccountType()} does not return {@link
63     *     net.dv8tion.jda.core.AccountType#CLIENT AccountType.CLIENT}. E.g: If the logged in account
64     *     isn't a Client account!
65     * @return {@code true} if the account is linked with a mobile app, otherwise {@code false}
66     */
67    boolean isMobile() throws AccountTypeException;
68  
69    /**
70     * The Discord Nitro status of this account. <br>
71     * <b>NOTE:</b> this is a {@link net.dv8tion.jda.core.AccountType#CLIENT AccountType.CLIENT}
72     * method only!
73     *
74     * @throws AccountTypeException If this method is called when {@link
75     *     net.dv8tion.jda.core.JDA#getAccountType() JDA#getAccountType()} does not return {@link
76     *     net.dv8tion.jda.core.AccountType#CLIENT AccountType.CLIENT}. E.g: If the logged in account
77     *     isn't a Client account!
78     * @return The Discord Nitro status of the currently logged in account.
79     */
80    boolean isNitro() throws AccountTypeException;
81  
82    /**
83     * Used to get the phone number of the currently logged in account if a phone number has been
84     * attached to it. <br>
85     * <b>NOTE:</b> this is a {@link net.dv8tion.jda.core.AccountType#CLIENT AccountType.CLIENT}
86     * method only!
87     *
88     * @throws AccountTypeException If this method is called when {@link
89     *     net.dv8tion.jda.core.JDA#getAccountType() JDA#getAccountType()} does not return {@link
90     *     net.dv8tion.jda.core.AccountType#CLIENT AccountType.CLIENT}. E.g: If the logged in account
91     *     isn't a Client account!
92     * @return The phone of the currently logged in account or null if there's no number associated
93     */
94    String getPhoneNumber() throws AccountTypeException;
95  
96    /**
97     * Returns the maximum size for files that can be uploaded with this account. <br>
98     * Returns {@value net.dv8tion.jda.core.entities.Message#MAX_FILE_SIZE} for bots and non-nitro
99     * client accounts and {@value net.dv8tion.jda.core.entities.Message#MAX_FILE_SIZE_NITRO} for
100    * client accounts with a active nitro subscription.
101    *
102    * @return The maximum size for files that can be uploaded with this account
103    * @see net.dv8tion.jda.core.entities.Message#MAX_FILE_SIZE
104    * @see net.dv8tion.jda.core.entities.Message#MAX_FILE_SIZE_NITRO
105    */
106   long getAllowedFileSize();
107 
108   /**
109    * The {@link net.dv8tion.jda.core.managers.AccountManager AccountManager} for the currently
110    * logged in account.
111    *
112    * <p>This can be used to atomically set account fields (like avatar/username)
113    *
114    * @return An AccountManager instance for the current account
115    */
116   AccountManager getManager();
117 
118   /**
119    * The {@link net.dv8tion.jda.core.managers.AccountManagerUpdatable AccountManagerUpdatable} for
120    * the currently logged in account.
121    *
122    * <p>This can be used to bulk update account fields (like avatar/username)
123    *
124    * @return An AccountManagerUpdatable instance for the current account
125    */
126   AccountManagerUpdatable getManagerUpdatable();
127 }
