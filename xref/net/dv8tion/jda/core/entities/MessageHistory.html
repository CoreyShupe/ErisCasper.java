34  /**
35   * Represents an access point to the {@link net.dv8tion.jda.core.entities.Message Message} history
36   * of a {@link net.dv8tion.jda.core.entities.MessageChannel MessageChannel}. <br>
37   * <b>Note:</b> Message order is always in recent to past order. I.e: A message at index 0 of a list
38   * is more recent than a message at index 1.
39   */
40  public class MessageHistory {
41    protected final MessageChannel channel;
42  
43    protected final ListOrderedMap<Long, Message> history = new ListOrderedMap<>();
44  
45    /**
46     * Creates a new MessageHistory object.
47     *
48     * @param channel The {@link net.dv8tion.jda.core.entities.MessageChannel MessageChannel} to
49     *     retrieval history from.
50     */
51    public MessageHistory(MessageChannel channel) {
52      this.channel = channel;
53      if (channel instanceof TextChannel) {
54        TextChannel tc = (TextChannel) channel;
55        if (!tc.getGuild().getSelfMember().hasPermission(tc, Permission.MESSAGE_HISTORY))
56          throw new InsufficientPermissionException(Permission.MESSAGE_HISTORY);
57      }
58    }

60    /**
61     * The corresponding JDA instance for this MessageHistory
62     *
63     * @return The corresponding JDA instance
64     */
65    public JDA getJDA() {
66      return channel.getJDA();
67    }

69    /**
70     * The amount of retrieved {@link net.dv8tion.jda.core.entities.Message Messages} by this
71     * MessageHistory. <br>
72     * This returns {@code 0} until any call to retrieve messages has completed. See {@link
73     * #retrievePast(int)} and {@link #retrieveFuture(int)}!
74     *
75     * @return Amount of retrieved messages
76     */
77    public int size() {
78      return history.size();
79    }

81    /**
82     * Whether this MessageHistory instance has retrieved any messages.
83     *
84     * @return True, If this MessageHistory instance has not retrieved any messages from discord.
85     */
86    public boolean isEmpty() {
87      return size() == 0;
88    }

100   /**
101    * Retrieves messages from Discord that were sent before the oldest sent message in
102    * MessageHistory's history cache ({@link #getRetrievedHistory()}). <br>
103    * Can only retrieve a <b>maximum</b> of {@code 100} messages at a time. <br>
104    * This method has 2 modes of operation: initial retrieval and additional retrieval.
105    *
106    * <ul>
107    *   <li><b>Initial Retrieval</b> <br>
108    *       This mode is what is used when no {@link net.dv8tion.jda.core.entities.Message Messages}
109    *       have been retrieved yet ({@link #getRetrievedHistory()}'s size is 0). Initial retrieval
110    *       starts from the most recent message sent to the channel and retrieves backwards from
111    *       there. So, if 50 messages are retrieved during this mode, the most recent 50 messages
112    *       will be retrieved.
113    *   <li><b>Additional Retrieval</b> <br>
114    *       This mode is used once some {@link net.dv8tion.jda.core.entities.Message Messages} have
115    *       already been retrieved from Discord and are stored in MessageHistory's history ({@link
116    *       #getRetrievedHistory()}). When retrieving messages in this mode, MessageHistory will
117    *       retrieve previous messages starting from the oldest message stored in MessageHistory.
118    *       <br>
119    *       E.g: If you initially retrieved 10 messages, the next call to this method to retrieve 10
120    *       messages would retrieve the <i>next</i> 10 messages, starting from the oldest message of
121    *       the 10 previously retrieved messages.
122    * </ul>
123    *
124    * <p>Possible {@link net.dv8tion.jda.core.requests.ErrorResponse ErrorResponses} include:
125    *
126    * <ul>
127    *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#UNKNOWN_MESSAGE UNKNOWN_MESSAGE} <br>
128    *       Can occur if retrieving in Additional Mode and the Message being used as the marker for
129    *       the last retrieved Message was deleted. Currently, to fix this, you need to create a new
130    *       {@link net.dv8tion.jda.core.entities.MessageHistory MessageHistory} instance.
131    *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#MISSING_ACCESS MISSING_ACCESS} <br>
132    *       Can occur if the request for history retrieval was executed <i>after</i> JDA lost access
133    *       to the Channel, typically due to the account being removed from the {@link
134    *       net.dv8tion.jda.core.entities.Guild Guild} or {@link
135    *       net.dv8tion.jda.client.entities.Group Group}.
136    *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#MISSING_PERMISSIONS
137    *       MISSING_PERMISSIONS} <br>
138    *       Can occur if the request for history retrieval was executed <i>after</i> JDA lost the
139    *       {@link net.dv8tion.jda.core.Permission#MESSAGE_HISTORY} permission.
140    *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#UNKNOWN_CHANNEL UNKNOWN_CHANNEL} <br>
141    *       The send request was attempted after the channel was deleted.
142    * </ul>
143    *
144    * @param amount The amount of {@link net.dv8tion.jda.core.entities.Message Messages} to retrieve.
145    * @throws java.lang.IllegalArgumentException The the {@code amount} is less than {@code 1} or
146    *     greater than {@code 100}.
147    * @return {@link net.dv8tion.jda.core.requests.RestAction RestAction} - Type: {@link
148    *     java.util.List List}{@literal <}{@link net.dv8tion.jda.core.entities.Message
149    *     Message}{@literal >} <br>
150    *     Retrieved Messages are placed in a List and provided in order of most recent to oldest with
151    *     most recent starting at index 0. If the list is empty, there were no more messages left to
152    *     retrieve.
153    */
154   public RestAction<List<Message>> retrievePast(int amount) {
155     if (amount > 100 || amount < 1)
156       throw new IllegalArgumentException(
157           "Message retrieval limit is between 1 and 100 messages. No more, no less. Limit provided: "
158               + amount);
159 
160     Route.CompiledRoute route =
161         Route.Messages.GET_MESSAGE_HISTORY
162             .compile(channel.getId())
163             .withQueryParams("limit", Integer.toString(amount));
164 
165     if (!history.isEmpty())
166       route = route.withQueryParams("before", String.valueOf(history.lastKey()));
167 
168     return new RestAction<List<Message>>(getJDA(), route) {
169       @Override
170       protected void handleResponse(Response response, Request<List<Message>> request) {
171         if (!response.isOk()) {
172           request.onFailure(response);
173           return;
174         }
175 
176         EntityBuilder builder = api.getEntityBuilder();
177         ;
178         LinkedList<Message> msgs = new LinkedList<>();
179         JSONArray historyJson = response.getArray();
180 
181         for (int i = 0; i < historyJson.length(); i++)
182           msgs.add(builder.createMessage(historyJson.getJSONObject(i)));
183 
184         msgs.forEach(msg -> history.put(msg.getIdLong(), msg));
185         request.onSuccess(msgs);
186       }
187     };
188   }

190   /**
191    * Retrieves messages from Discord that were sent more recently than the most recently sent
192    * message in MessageHistory's history cache ({@link #getRetrievedHistory()}). Use case for this
193    * method is for getting more recent messages after jumping to a specific point in history using
194    * something like {@link MessageChannel#getHistoryAround(String, int)}. <br>
195    * This method works in the same way as {@link #retrievePast(int)}'s Additional Retrieval mode.
196    *
197    * <p><b>Note:</b> This method can only be used after {@link net.dv8tion.jda.core.entities.Message
198    * Messages} have already been retrieved from Discord.
199    *
200    * <p>Possible {@link net.dv8tion.jda.core.requests.ErrorResponse ErrorResponses} include:
201    *
202    * <ul>
203    *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#UNKNOWN_MESSAGE UNKNOWN_MESSAGE} <br>
204    *       Can occur if retrieving in Additional Mode and the Message being used as the marker for
205    *       the last retrieved Message was deleted. Currently, to fix this, you need to create a new
206    *       {@link net.dv8tion.jda.core.entities.MessageHistory MessageHistory} instance.
207    *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#MISSING_ACCESS MISSING_ACCESS} <br>
208    *       Can occur if the request for history retrieval was executed <i>after</i> JDA lost access
209    *       to the Channel, typically due to the account being removed from the {@link
210    *       net.dv8tion.jda.core.entities.Guild Guild} or {@link
211    *       net.dv8tion.jda.client.entities.Group Group}.
212    *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#MISSING_PERMISSIONS
213    *       MISSING_PERMISSIONS} <br>
214    *       Can occur if the request for history retrieval was executed <i>after</i> JDA lost the
215    *       {@link net.dv8tion.jda.core.Permission#MESSAGE_HISTORY} permission.
216    *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#UNKNOWN_CHANNEL UNKNOWN_CHANNEL} <br>
217    *       The send request was attempted after the channel was deleted.
218    * </ul>
219    *
220    * @param amount The amount of {@link net.dv8tion.jda.core.entities.Message Messages} to retrieve.
221    * @throws java.lang.IllegalArgumentException The the {@code amount} is less than {@code 1} or
222    *     greater than {@code 100}.
223    * @throws java.lang.IllegalStateException If no messages have been retrieved by this
224    *     MessageHistory.
225    * @return {@link net.dv8tion.jda.core.requests.RestAction RestAction} - Type: {@link
226    *     java.util.List List}{@literal <}{@link net.dv8tion.jda.core.entities.Message
227    *     Message}{@literal >} <br>
228    *     Retrieved Messages are placed in a List and provided in order of most recent to oldest with
229    *     most recent starting at index 0. If the list is empty, there were no more messages left to
230    *     retrieve.
231    */
232   public RestAction<List<Message>> retrieveFuture(int amount) {
233     if (amount > 100 || amount < 1)
234       throw new IllegalArgumentException(
235           "Message retrieval limit is between 1 and 100 messages. No more, no less. Limit provided: "
236               + amount);
237 
238     if (history.isEmpty())
239       throw new IllegalStateException(
240           "No messages have been retrieved yet, so there is no message to act as a marker to retrieve more recent messages based on.");
241 
242     Route.CompiledRoute route =
243         Route.Messages.GET_MESSAGE_HISTORY
244             .compile(channel.getId())
245             .withQueryParams(
246                 "limit", Integer.toString(amount), "after", String.valueOf(history.firstKey()));
247     return new RestAction<List<Message>>(getJDA(), route) {
248       @Override
249       protected void handleResponse(Response response, Request<List<Message>> request) {
250         if (!response.isOk()) {
251           request.onFailure(response);
252           return;
253         }
254 
255         EntityBuilder builder = api.getEntityBuilder();
256         ;
257         LinkedList<Message> msgs = new LinkedList<>();
258         JSONArray historyJson = response.getArray();
259 
260         for (int i = 0; i < historyJson.length(); i++)
261           msgs.add(builder.createMessage(historyJson.getJSONObject(i)));
262 
263         for (Iterator<Message> it = msgs.descendingIterator(); it.hasNext(); ) {
264           Message m = it.next();
265           history.put(0, m.getIdLong(), m);
266         }
267 
268         request.onSuccess(msgs);
269       }
270     };
271   }

273   /**
274    * Returns a List of Messages, sorted starting from newest to oldest, of all message that have
275    * already been retrieved from Discord with this MessageHistory object using the {@link
276    * #retrievePast(int)}, {@link #retrieveFuture(int)}, and {@link
277    * net.dv8tion.jda.core.entities.MessageChannel#getHistoryAround(String, int)} methods.
278    *
279    * @return A List of Messages, sorted newest to oldest.
280    */
281   public List<Message> getRetrievedHistory() {
282     int size = size();
283     if (size == 0) return Collections.emptyList();
284     else if (size == 1) return Collections.singletonList(history.getValue(0));
285     return Collections.unmodifiableList(new ArrayList<>(history.values()));
286   }

473   private static void checkArguments(MessageChannel channel, String messageId) {
474     Checks.noWhitespace(messageId, "Message ID");
475     Checks.notNull(channel, "Channel");
476     if (channel.getType() == ChannelType.TEXT) {
477       TextChannel t = (TextChannel) channel;

479         throw new InsufficientPermissionException(Permission.MESSAGE_HISTORY);

483   /**
484    * Constructs a MessageHistory object with initially retrieved Messages before or after a certain
485    * pivot message id. <br>
486    * Allows to {@link #limit(Integer) limit} the amount to retrieve for better performance!
487    */
488   public static class MessageRetrieveAction extends RestAction<MessageHistory> {
489     private final MessageChannel channel;
490     private Integer limit;
491 
492     protected MessageRetrieveAction(Route.CompiledRoute route, MessageChannel channel) {
493       super(channel.getJDA(), route);
494       this.channel = channel;
495     }
496 
497     /**
498      * Limit between 1-100 messages that should be retrieved.
499      *
500      * @param limit The limit to use, or {@code null} to use default 50
501      * @throws java.lang.IllegalArgumentException If the provided limit is not between 1-100
502      * @return The current MessageRetrieveAction for chaining convenience
503      */
504     public MessageRetrieveAction limit(Integer limit) {
505       if (limit != null) {
506         Checks.positive(limit, "Limit");
507         Checks.check(limit <= 100, "Limit may not exceed 100!");
508       }
509       this.limit = limit;
510       return this;
511     }
512 
513     @Override
514     protected Route.CompiledRoute finalizeRoute() {
515       final Route.CompiledRoute route = super.finalizeRoute();
516       return limit == null ? route : route.withQueryParams("limit", String.valueOf(limit));
517     }
518 
519     @Override
520     protected void handleResponse(Response response, Request<MessageHistory> request) {
521       if (!response.isOk()) {
522         request.onFailure(response);
523         return;
524       }
525       final MessageHistory result = new MessageHistory(channel);
526       final JSONArray array = response.getArray();
527       final EntityBuilder builder = api.getEntityBuilder();
528       for (int i = 0; i < array.length(); i++) {
529         try {
530           JSONObject obj = array.getJSONObject(i);
531           result.history.put(obj.getLong("id"), builder.createMessage(obj, channel, false));
532         } catch (JSONException | NullPointerException e) {
533           LOG.warn("Encountered exception in MessagePagination", e);
534         }
535       }
536       request.onSuccess(result);
537     }
538   }
