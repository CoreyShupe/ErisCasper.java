22  /**
23   * Represents a Discord {@link net.dv8tion.jda.core.entities.Game Game}. <br>
24   * This should contain all information provided from Discord about a Game.
25   *
26   * @since 2.1
27   * @author John A. Grosh
28   */
29  public class Game {
30    protected final String name;
31    protected final String url;
32    protected final Game.GameType type;
33  
34    protected Game(String name) {
35      this(name, null, GameType.DEFAULT);
36    }
37  
38    protected Game(String name, String url) {
39      this(name, url, GameType.STREAMING);
40    }
41  
42    protected Game(String name, String url, GameType type) {
43      this.name = name;
44      this.url = url;
45      this.type = type;
46    }

48    /**
49     * Whether this is a <a href="https://discordapp.com/developers/docs/rich-presence/best-practices"
50     * target="_blank">Rich Presence</a> <br>
51     * If {@code false} the result of {@link #asRichPresence()} is {@code null}
52     *
53     * @return {@code true} if this is a {@link net.dv8tion.jda.core.entities.RichPresence
54     *     RichPresence}
55     */
56    public boolean isRich() {
57      return false;
58    }
59  
60    /**
61     * {@link net.dv8tion.jda.core.entities.RichPresence RichPresence} representation of this Game.
62     *
63     * @return RichPresence or {@code null} if {@link #isRich()} returns {@code false}
64     */
65    public RichPresence asRichPresence() {
66      return null;
67    }

69    /**
70     * The displayed name of the {@link net.dv8tion.jda.core.entities.Game Game}. If no name has been
71     * set, this returns null.

73     * @return Possibly-null String containing the Game's name.
74     */
75    public String getName() {
76      return name;
77    }

79    /**
80     * The URL of the {@link net.dv8tion.jda.core.entities.Game Game} if the game is actually a

83     *
84     * @return Possibly-null String containing the Game's URL.

86    public String getUrl() {

88    }

90    /**
91     * The type of {@link net.dv8tion.jda.core.entities.Game Game}.

93     * @return Never-null {@link net.dv8tion.jda.core.entities.Game.GameType GameType} representing

96    public GameType getType() {

100   @Override
101   public boolean equals(Object o) {
102     if (!(o instanceof Game)) return false;
103     if (o == this) return true;
104 
105     Game oGame = (Game) o;
106     return oGame.getType() == type
107         && Objects.equals(name, oGame.getName())
108         && Objects.equals(url, oGame.getUrl());
109   }
110 
111   @Override
112   public int hashCode() {
113     return Objects.hash(name, type, url);
114   }

116   @Override
117   public String toString() {
118     if (url != null) return String.format("Game(%s | %s)", name, url);
119     else return String.format("Game(%s)", name);
120   }

122   /**
123    * Creates a new Game instance with the specified name. <br>
124    * In order to appear as "streaming" in the official client you must provide a valid (see
125    * documentation of method) streaming URL in {@link #streaming(String, String)
126    * Game.streaming(String, String)}.
127    *
128    * @param name The not-null name of the newly created game
129    * @throws IllegalArgumentException if the specified name is null, empty or blank
130    * @return A valid Game instance with the provided name with {@link GameType#DEFAULT}
131    */
132   public static Game playing(String name) {
133     Checks.notBlank(name, "Name");
134     return new Game(name, null, GameType.DEFAULT);
135   }
136 
137   /**
138    * Creates a new Game instance with the specified name and url. <br>
139    * The specified URL must be valid according to discord standards in order to display as
140    * "streaming" in the official client. A valid streaming URL must be derived from {@code
141    * https://twitch.tv/} and can be verified using {@link #isValidStreamingUrl(String)}. (see
142    * documentation)
143    *
144    * @param name The not-null name of the newly created game
145    * @param url The streaming url to use, required to display as "streaming"
146    * @throws IllegalArgumentException If the specified name is null or empty
147    * @return A valid Game instance with the provided name and url
148    * @see #isValidStreamingUrl(String)
149    */
150   public static Game streaming(String name, String url) {
151     Checks.notEmpty(name, "Provided game name");
152     GameType type;
153     if (isValidStreamingUrl(url)) type = GameType.STREAMING;
154     else type = GameType.DEFAULT;
155     return new Game(name, url, type);
156   }

158   /**
159    * Creates a new Game instance with the specified name. <br>
160    * This will display as {@code Listening name} in the official client
161    *
162    * @param name The not-null name of the newly created game
163    * @throws IllegalArgumentException if the specified name is null, empty or blank
164    * @return A valid Game instance with the provided name with {@link GameType#LISTENING}
165    */
166   public static Game listening(String name) {
167     Checks.notBlank(name, "Name");
168     return new Game(name, null, GameType.LISTENING);
169   }
170 
171   /**
172    * Creates a new Game instance with the specified name. <br>
173    * This will display as {@code Watching name} in the official client
174    *
175    * @param name The not-null name of the newly created game
176    * @throws IllegalArgumentException if the specified name is null, empty or blank
177    * @return A valid Game instance with the provided name with {@link GameType#WATCHING}
178    */
179   public static Game watching(String name) {
180     Checks.notBlank(name, "Name");
181     return new Game(name, null, GameType.WATCHING);
182   }
183 
184   /**
185    * Creates a new Game instance with the specified name and url.
186    *
187    * @param type The {@link net.dv8tion.jda.core.entities.Game.GameType GameType} to use
188    * @param name The not-null name of the newly created game
189    * @throws IllegalArgumentException If the specified name is null or empty
190    * @return A valid Game instance with the provided name and url
191    */
192   public static Game of(GameType type, String name) {
193     return of(type, name, null);
194   }
195 
196   /**
197    * Creates a new Game instance with the specified name and url. <br>
198    * The provided url would only be used for {@link
199    * net.dv8tion.jda.core.entities.Game.GameType#STREAMING GameType.STREAMING} and should be a
200    * twitch url.
201    *
202    * @param type The {@link net.dv8tion.jda.core.entities.Game.GameType GameType} to use
203    * @param name The not-null name of the newly created game
204    * @param url The streaming url to use, required to display as "streaming".
205    * @throws IllegalArgumentException If the specified name is null or empty
206    * @return A valid Game instance with the provided name and url
207    * @see #isValidStreamingUrl(String)
208    */
209   public static Game of(GameType type, String name, String url) {
210     Checks.notNull(type, "Type");
211     switch (type) {
212       case DEFAULT:
213         return playing(name);
214       case STREAMING:
215         return streaming(name, url);
216       case LISTENING:
217         return listening(name);
218       case WATCHING:
219         return watching(name);
220       default:
221         throw new IllegalArgumentException("GameType " + type + " is not supported!");
222     }
223   }
224 
225   /**
226    * Checks if a given String is a valid Twitch url (ie, one that will display "Streaming" on the
227    * Discord client).
228    *
229    * @param url The url to check.
230    * @return True if the provided url is valid for triggering Discord's streaming status
231    */
232   public static boolean isValidStreamingUrl(String url) {
233     return url != null && url.matches("https?://(www\\.)?twitch\\.tv/.+");
234   }

236   /** The type game being played, differentiating between a game and stream types. */
237   public enum GameType {
238     /**
239      * The GameType used to represent a normal {@link net.dv8tion.jda.core.entities.Game Game}
240      * status.
241      */
242     DEFAULT(0),
243     /**
244      * Used to indicate that the {@link net.dv8tion.jda.core.entities.Game Game} is a stream <br>
245      * This type is displayed as "Streaming" in the discord client.
246      */
247     STREAMING(1),
248     /**
249      * Used to indicate that the {@link net.dv8tion.jda.core.entities.Game Game} should display as
250      * {@code Listening...} in the official client.
251      */
252     LISTENING(2),
253     /**
254      * Used to indicate that the {@link net.dv8tion.jda.core.entities.Game Game} should display as
255      * {@code Watching...} in the official client.
256      */
257     WATCHING(3);
258 
259     private final int key;
260 
261     GameType(int key) {
262       this.key = key;
263     }
264 
265     /**
266      * The Discord defined id key for this GameType.
267      *
268      * @return the id key.
269      */
270     public int getKey() {
271       return key;
272     }
273 
274     /**
275      * Gets the GameType related to the provided key. <br>
276      * If an unknown key is provided, this returns {@link #DEFAULT}
277      *
278      * @param key The Discord key referencing a GameType.
279      * @return The GameType that has the key provided, or {@link #DEFAULT} for unknown key.
280      */
281     public static GameType fromKey(int key) {
282       switch (key) {
283         case 0:
284         default:
285           return DEFAULT;
286         case 1:
287           return STREAMING;
288         case 2:
289           return LISTENING;
290         case 3:
291           return WATCHING;
292       }
293     }
294   }
