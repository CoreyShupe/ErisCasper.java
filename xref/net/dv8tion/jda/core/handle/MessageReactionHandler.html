17  package net.dv8tion.jda.core.handle;

20  import net.dv8tion.jda.client.events.message.group.react.GroupMessageReactionRemoveEvent;

28  import net.dv8tion.jda.core.events.message.guild.react.GuildMessageReactionAddEvent;
29  import net.dv8tion.jda.core.events.message.guild.react.GuildMessageReactionRemoveEvent;
30  import net.dv8tion.jda.core.events.message.priv.react.PrivateMessageReactionAddEvent;
31  import net.dv8tion.jda.core.events.message.priv.react.PrivateMessageReactionRemoveEvent;

33  import net.dv8tion.jda.core.events.message.react.MessageReactionRemoveEvent;

35  import org.json.JSONObject;
36  import org.slf4j.Logger;
37  import org.slf4j.LoggerFactory;
38  
39  public class MessageReactionHandler extends SocketHandler {
40  
41    private static final Logger LOG = LoggerFactory.getLogger(MessageReactionHandler.class);
42  
43    private final boolean add;
44  
45    public MessageReactionHandler(JDAImpl api, boolean add) {
46      super(api);
47      this.add = add;
48    }
49  
50    @Override
51    protected Long handleInternally(JSONObject content) {
52      JSONObject emoji = content.getJSONObject("emoji");
53  
54      final long userId = content.getLong("user_id");
55      final long messageId = content.getLong("message_id");
56      final long channelId = content.getLong("channel_id");
57  
58      final Long emojiId = emoji.isNull("id") ? null : emoji.getLong("id");
59      String emojiName = emoji.optString("name", null);
60      final boolean emojiAnimated = emoji.optBoolean("animated");
61  
62      if (emojiId == null && emojiName == null) {
63        LOG.debug(
64            "Received a reaction {} with no name nor id. json: {}", add ? "add" : "remove", content);
65        return null;
66      }
67  
68      User user = api.getUserById(userId);
69      if (user == null) user = api.getFakeUserMap().get(userId);
70      if (user == null) {
71        api.getEventCache()
72            .cache(EventCache.Type.USER, userId, () -> handle(responseNumber, allContent));
73        EventCache.LOG.debug(
74            "Received a reaction for a user that JDA does not currently have cached");
75        return null;
76      }
77  
78      MessageChannel channel = api.getTextChannelById(channelId);
79      if (channel == null) {
80        channel = api.getPrivateChannelById(channelId);
81      }
82      if (channel == null) {
83        channel = api.getFakePrivateChannelMap().get(channelId);
84      }
85      if (channel == null && api.getAccountType() == AccountType.CLIENT) {
86        channel = api.asClient().getGroupById(channelId);
87      }
88      if (channel == null) {
89        api.getEventCache()
90            .cache(EventCache.Type.CHANNEL, channelId, () -> handle(responseNumber, allContent));
91        EventCache.LOG.debug(
92            "Received a reaction for a channel that JDA does not currently have cached");
93        return null;
94      }
95  
96      MessageReaction.ReactionEmote rEmote;
97      if (emojiId != null) {
98        Emote emote = api.getEmoteById(emojiId);
99        if (emote == null) {
100         if (emojiName != null) {
101           emote = new EmoteImpl(emojiId, api).setAnimated(emojiAnimated).setName(emojiName);
102         } else {
103           LOG.debug(
104               "Received a reaction {} with a null name. json: {}", add ? "add" : "remove", content);
105           return null;
106         }
107       }
108       rEmote = new MessageReaction.ReactionEmote(emote);
109     } else {
110       rEmote = new MessageReaction.ReactionEmote(emojiName, null, api);
111     }
112     MessageReaction reaction =
113         new MessageReaction(channel, rEmote, messageId, user.equals(api.getSelfUser()), -1);
114 
115     if (add) onAdd(reaction, user);
116     else onRemove(reaction, user);
117     return null;
118   }
119 
120   private void onAdd(MessageReaction reaction, User user) {
121     IEventManager manager = api.getEventManager();
122     switch (reaction.getChannelType()) {
123       case TEXT:
124         manager.handle(
125             new GuildMessageReactionAddEvent(
126                 api, responseNumber,
127                 user, reaction));
128         break;
129       case GROUP:
130         manager.handle(
131             new GroupMessageReactionAddEvent(
132                 api, responseNumber,
133                 user, reaction));
134         break;
135       case PRIVATE:
136         manager.handle(
137             new PrivateMessageReactionAddEvent(
138                 api, responseNumber,
139                 user, reaction));
140     }
141 
142     manager.handle(
143         new MessageReactionAddEvent(
144             api, responseNumber,
145             user, reaction));
146   }
147 
148   private void onRemove(MessageReaction reaction, User user) {
149     IEventManager manager = api.getEventManager();
150     switch (reaction.getChannelType()) {
151       case TEXT:
152         manager.handle(
153             new GuildMessageReactionRemoveEvent(
154                 api, responseNumber,
155                 user, reaction));
156         break;
157       case GROUP:
158         manager.handle(
159             new GroupMessageReactionRemoveEvent(
160                 api, responseNumber,
161                 user, reaction));
162         break;
163       case PRIVATE:
164         manager.handle(
165             new PrivateMessageReactionRemoveEvent(
166                 api, responseNumber,
167                 user, reaction));
168     }
169 
170     manager.handle(
171         new MessageReactionRemoveEvent(
172             api, responseNumber,
173             user, reaction));
174   }
175 }
