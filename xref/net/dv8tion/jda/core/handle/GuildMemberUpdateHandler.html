17  package net.dv8tion.jda.core.handle;
18  
19  import java.util.*;
20  import net.dv8tion.jda.core.entities.Role;
21  import net.dv8tion.jda.core.entities.impl.GuildImpl;
22  import net.dv8tion.jda.core.entities.impl.JDAImpl;
23  import net.dv8tion.jda.core.entities.impl.MemberImpl;
24  import net.dv8tion.jda.core.events.guild.member.GuildMemberNickChangeEvent;
25  import net.dv8tion.jda.core.events.guild.member.GuildMemberRoleAddEvent;
26  import net.dv8tion.jda.core.events.guild.member.GuildMemberRoleRemoveEvent;
27  import org.json.JSONArray;
28  import org.json.JSONObject;
29  
30  public class GuildMemberUpdateHandler extends SocketHandler {
31  
32    public GuildMemberUpdateHandler(JDAImpl api) {
33      super(api);
34    }
35  
36    @Override
37    protected Long handleInternally(JSONObject content) {
38      final long id = content.getLong("guild_id");
39      if (api.getGuildLock().isLocked(id)) return id;
40  
41      JSONObject userJson = content.getJSONObject("user");
42      final long userId = userJson.getLong("id");
43      GuildImpl guild = (GuildImpl) api.getGuildMap().get(id);
44      if (guild == null) {
45        api.getEventCache()
46            .cache(
47                EventCache.Type.GUILD,
48                userId,
49                () -> {
50                  handle(responseNumber, allContent);
51                });
52        EventCache.LOG.debug(
53            "Got GuildMember update but JDA currently does not have the Guild cached. {}", content);
54        return null;
55      }
56  
57      MemberImpl member = (MemberImpl) guild.getMembersMap().get(userId);
58      if (member == null) {
59        api.getEventCache()
60            .cache(
61                EventCache.Type.USER,
62                userId,
63                () -> {
64                  handle(responseNumber, allContent);
65                });
66        EventCache.LOG.debug(
67            "Got GuildMember update but Member is not currently present in Guild. {}", content);
68        return null;
69      }
70  
71      Set<Role> currentRoles = member.getRoleSet();
72      List<Role> newRoles = toRolesList(guild, content.getJSONArray("roles"));
73  
74      // If newRoles is null that means that we didn't find a role that was in the array and was
75      // cached this event
76      if (newRoles == null) return null;
77  
78      // Find the roles removed.
79      List<Role> removedRoles = new LinkedList<>();
80      each:
81      for (Role role : currentRoles) {
82        for (Iterator<Role> it = newRoles.iterator(); it.hasNext(); ) {
83          Role r = it.next();
84          if (role.equals(r)) {
85            it.remove();
86            continue each;
87          }
88        }
89        removedRoles.add(role);
90      }
91  
92      if (removedRoles.size() > 0) currentRoles.removeAll(removedRoles);
93      if (newRoles.size() > 0) currentRoles.addAll(newRoles);
94  
95      if (removedRoles.size() > 0) {
96        api.getEventManager()
97            .handle(new GuildMemberRoleRemoveEvent(api, responseNumber, guild, member, removedRoles));
98      }
99      if (newRoles.size() > 0) {
100       api.getEventManager()
101           .handle(new GuildMemberRoleAddEvent(api, responseNumber, guild, member, newRoles));
102     }
103     if (content.has("nick")) {
104       String prevNick = member.getNickname();
105       String newNick = content.optString("nick", null);
106       if (!Objects.equals(prevNick, newNick)) {
107         member.setNickname(newNick);
108         api.getEventManager()
109             .handle(
110                 new GuildMemberNickChangeEvent(
111                     api, responseNumber, guild, member, prevNick, newNick));
112       }
113     }
114     return null;
115   }
116 
117   private List<Role> toRolesList(GuildImpl guild, JSONArray array) {
118     LinkedList<Role> roles = new LinkedList<>();
119     for (int i = 0; i < array.length(); i++) {
120       final long id = array.getLong(i);
121       Role r = guild.getRolesMap().get(id);
122       if (r != null) {
123         roles.add(r);
124       } else {
125         api.getEventCache()
126             .cache(
127                 EventCache.Type.ROLE,
128                 id,
129                 () -> {
130                   handle(responseNumber, allContent);
131                 });
132         EventCache.LOG.debug(
133             "Got GuildMember update but one of the Roles for the Member is not yet cached.");
134         return null;
135       }
136     }
137     return roles;
138   }
139 }
