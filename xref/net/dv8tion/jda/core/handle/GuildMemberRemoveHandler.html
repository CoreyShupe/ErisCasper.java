17  package net.dv8tion.jda.core.handle;
18  
19  import net.dv8tion.jda.client.entities.Group;
20  import net.dv8tion.jda.core.AccountType;
21  import net.dv8tion.jda.core.entities.VoiceChannel;
22  import net.dv8tion.jda.core.entities.impl.*;
23  import net.dv8tion.jda.core.events.guild.member.GuildMemberLeaveEvent;
24  import net.dv8tion.jda.core.events.guild.voice.GuildVoiceLeaveEvent;
25  import net.dv8tion.jda.core.requests.WebSocketClient;
26  import org.json.JSONObject;
27  
28  public class GuildMemberRemoveHandler extends SocketHandler {
29  
30    public GuildMemberRemoveHandler(JDAImpl api) {
31      super(api);
32    }
33  
34    @Override
35    protected Long handleInternally(JSONObject content) {
36      final long id = content.getLong("guild_id");
37      if (api.getGuildLock().isLocked(id)) return id;
38  
39      GuildImpl guild = (GuildImpl) api.getGuildMap().get(id);
40      if (guild == null) {
41        // We probably just left the guild and this event is trying to remove us from the guild,
42        // therefore ignore
43        return null;
44      }
45  
46      final long userId = content.getJSONObject("user").getLong("id");
47      if (userId == api.getSelfUser().getIdLong()) {
48        // We probably just left the guild and this event is trying to remove us from the guild,
49        // therefore ignore
50        return null;
51      }
52      MemberImpl member = (MemberImpl) guild.getMembersMap().remove(userId);
53  
54      if (member == null) {
55        WebSocketClient.LOG.debug(
56            "Received GUILD_MEMBER_REMOVE for a Member that does not exist in the specified Guild.");
57        return null;
58      }
59  
60      if (member
61          .getVoiceState()
62          .inVoiceChannel()) // If this user was in a VoiceChannel, fire VoiceLeaveEvent.
63      {
64        GuildVoiceStateImpl vState = (GuildVoiceStateImpl) member.getVoiceState();
65        VoiceChannel channel = vState.getChannel();
66        vState.setConnectedChannel(null);
67        ((VoiceChannelImpl) channel).getConnectedMembersMap().remove(member.getUser().getIdLong());
68        api.getEventManager()
69            .handle(
70                new GuildVoiceLeaveEvent(
71                    api, responseNumber,
72                    member, channel));
73      }
74  
75      // The user is not in a different guild that we share
76      // The user also is not a friend of this account in the case that the logged in account is a
77      // client account.
78      if (userId != api.getSelfUser().getIdLong() // don't remove selfUser from cache
79          && api.getGuildMap()
80              .valueCollection()
81              .stream()
82              .noneMatch(g -> ((GuildImpl) g).getMembersMap().containsKey(userId))
83          && !(api.getAccountType() == AccountType.CLIENT
84              && api.asClient().getFriendById(userId) != null)) {
85        UserImpl user = (UserImpl) api.getUserMap().remove(userId);
86        if (user.hasPrivateChannel()) {
87          PrivateChannelImpl priv = (PrivateChannelImpl) user.getPrivateChannel();
88          user.setFake(true);
89          priv.setFake(true);
90          api.getFakeUserMap().put(user.getIdLong(), user);
91          api.getFakePrivateChannelMap().put(priv.getIdLong(), priv);
92        } else if (api.getAccountType() == AccountType.CLIENT) {
93          // While the user might not have a private channel, if this is a client account then the
94          // user
95          // could be in a Group, and if so we need to change the User object to be fake and
96          // place it in the FakeUserMap
97          for (Group grp : api.asClient().getGroups()) {
98            if (grp.getNonFriendUsers().contains(user)) {
99              user.setFake(true);
100             api.getFakeUserMap().put(user.getIdLong(), user);
101             break; // Breaks from groups loop
102           }
103         }
104       }
105       api.getEventCache().clear(EventCache.Type.USER, userId);
106     }
107     api.getEventManager()
108         .handle(
109             new GuildMemberLeaveEvent(
110                 api, responseNumber,
111                 guild, member));
112     return null;
113   }
114 }
