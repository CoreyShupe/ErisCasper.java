17  package net.dv8tion.jda.core.handle;
18  
19  import gnu.trove.iterator.TLongIterator;
20  import gnu.trove.set.TLongSet;
21  import gnu.trove.set.hash.TLongHashSet;
22  import net.dv8tion.jda.core.WebSocketCode;
23  import net.dv8tion.jda.core.entities.ChannelType;
24  import net.dv8tion.jda.core.entities.EntityBuilder;
25  import net.dv8tion.jda.core.entities.Guild;
26  import net.dv8tion.jda.core.entities.impl.JDAImpl;
27  import net.dv8tion.jda.core.requests.WebSocketClient;
28  import org.json.JSONArray;
29  import org.json.JSONObject;
30  
31  public class ReadyHandler extends SocketHandler {
32    private final TLongSet incompleteGuilds = new TLongHashSet();
33    private final TLongSet acknowledgedGuilds = new TLongHashSet();
34    private final TLongSet unavailableGuilds = new TLongHashSet();
35    private final TLongSet guildsRequiringChunking = new TLongHashSet();
36    private final TLongSet guildsRequiringSyncing = new TLongHashSet();
37  
38    public ReadyHandler(JDAImpl api) {
39      super(api);
40    }
41  
42    @Override
43    protected Long handleInternally(JSONObject content) {
44      EntityBuilder builder = api.getEntityBuilder();
45  
46      // Core
47      JSONArray guilds = content.getJSONArray("guilds");
48      JSONObject selfJson = content.getJSONObject("user");
49  
50      builder.createSelfUser(selfJson);
51  
52      // Keep a list of all guilds in incompleteGuilds that need to be setup (GuildMemberChunk /
53      // GuildSync)
54      // Send all guilds to the EntityBuilder's first pass to setup caching for when GUILD_CREATE
55      // comes
56      // or, for Client accounts, to start the setup process (since we already have guild info)
57      // Callback points to guildSetupComplete so that when MemberChunking and GuildSync processes are
58      // done, we can
59      // "check off" the completed guild from the set of guilds in incompleteGuilds.
60  
61      for (int i = 0; i < guilds.length(); i++) {
62        JSONObject guild = guilds.getJSONObject(i);
63        incompleteGuilds.add(guild.getLong("id"));
64      }
65  
66      // We use two different for-loops here so that we cache all of the ids before sending them off
67      // to the EntityBuilder
68      //  due to the checks in checkIfReadyToSendRequests and guildSetupComplete triggering too soon
69      // otherwise.
70      // Specifically: incompleteGuilds.size() == acknowledgedGuilds.size() and
71      //  incompleteGuilds.size() == unavailableGuilds.size() respectively.
72  
73      for (int i = 0; i < guilds.length(); i++) {
74        JSONObject guild = guilds.getJSONObject(i);
75  
76        // If a Guild isn't unavailable, then it is possible that we were given all information
77        // needed to fully load the guild. In this case, we provide the method `guildSetupComplete`
78        // as the secondPassCallback so it can immediately be called to signify that the provided
79        // guild
80        // is loaded and ready to go.
81        // If a Guild is unavailable it won't have the information needed, so we pass null as the
82        // secondPassCallback
83        // for now and wait for the GUILD_CREATE event to give us the required information.
84        if (guild.has("unavailable") && guild.getBoolean("unavailable"))
85          builder.createGuildFirstPass(guild, null);
86        else builder.createGuildFirstPass(guild, this::guildSetupComplete);
87      }
88  
89      if (guilds.length() == 0) guildLoadComplete(content);
90  
91      return null;
92    }
93  
94    public void guildLoadComplete(JSONObject content) {
95      api.getClient().setChunkingAndSyncing(false);
96      EntityBuilder builder = api.getEntityBuilder();
97      JSONArray privateChannels = content.getJSONArray("private_channels");
98  
99      for (int i = 0; i < privateChannels.length(); i++) {
100       JSONObject chan = privateChannels.getJSONObject(i);
101       ChannelType type = ChannelType.fromId(chan.getInt("type"));
102 
103       switch (type) {
104         case PRIVATE:
105           builder.createPrivateChannel(chan);
106           break;
107         case GROUP:
108           builder.createGroup(chan);
109           break;
110         default:
111           WebSocketClient.LOG.warn(
112               "Received a Channel in the priv_channels array in READY of an unknown type! JSON: {}",
113               type);
114       }
115     }
116 
117     api.getClient().ready();
118   }
119 
120   public void acknowledgeGuild(
121       Guild guild, boolean available, boolean requiresChunking, boolean requiresSync) {
122     acknowledgedGuilds.add(guild.getIdLong());
123     if (available) {
124       // We remove from unavailable guilds because it is possible that we were told it was
125       // unavailable, but
126       // during a long READY load it could have become available and was sent to us.
127       unavailableGuilds.remove(guild.getIdLong());
128       if (requiresChunking) guildsRequiringChunking.add(guild.getIdLong());
129       if (requiresSync) guildsRequiringSyncing.add(guild.getIdLong());
130     } else unavailableGuilds.add(guild.getIdLong());
131 
132     checkIfReadyToSendRequests();
133   }
134 
135   public void guildSetupComplete(Guild guild) {
136     if (!incompleteGuilds.remove(guild.getIdLong()))
137       WebSocketClient.LOG.error(
138           "Completed the setup for Guild: {} without matching id in ReadyHandler cache", guild);
139     if (incompleteGuilds.size() == unavailableGuilds.size())
140       guildLoadComplete(allContent.getJSONObject("d"));
141     else checkIfReadyToSendRequests();
142   }
143 
144   public void clearCache() {
145     incompleteGuilds.clear();
146     acknowledgedGuilds.clear();
147     unavailableGuilds.clear();
148     guildsRequiringChunking.clear();
149     guildsRequiringSyncing.clear();
150   }
151 
152   private void checkIfReadyToSendRequests() {
153     if (acknowledgedGuilds.size() == incompleteGuilds.size()) {
154       api.getClient().setChunkingAndSyncing(true);
155       sendMemberChunkRequests();
156     }
157   }
158 
159   private void sendGuildSyncRequests() {
160     if (guildsRequiringSyncing.isEmpty()) return;
161 
162     JSONArray guildIds = new JSONArray();
163 
164     for (TLongIterator it = guildsRequiringSyncing.iterator(); it.hasNext(); ) {
165       guildIds.put(it.next());
166 
167       // We can only request 50 guilds in a single request, so after we've reached 50, send them
168       // and reset the
169       if (guildIds.length() == 50) {
170         api.getClient()
171             .chunkOrSyncRequest(
172                 new JSONObject().put("op", WebSocketCode.GUILD_SYNC).put("d", guildIds));
173         guildIds = new JSONArray();
174       }
175     }
176 
177     // Send the remaining guilds that need to be sent
178     if (guildIds.length() > 0) {
179       api.getClient()
180           .chunkOrSyncRequest(
181               new JSONObject().put("op", WebSocketCode.GUILD_SYNC).put("d", guildIds));
182     }
183     guildsRequiringSyncing.clear();
184   }
185 
186   private void sendMemberChunkRequests() {
187     if (guildsRequiringChunking.isEmpty()) return;
188 
189     JSONArray guildIds = new JSONArray();
190     for (TLongIterator it = guildsRequiringChunking.iterator(); it.hasNext(); ) {
191       guildIds.put(it.next());
192 
193       // We can only request 50 guilds in a single request, so after we've reached 50, send them
194       // and reset the
195       if (guildIds.length() == 50) {
196         api.getClient()
197             .chunkOrSyncRequest(
198                 new JSONObject()
199                     .put("op", WebSocketCode.MEMBER_CHUNK_REQUEST)
200                     .put(
201                         "d",
202                         new JSONObject()
203                             .put("guild_id", guildIds)
204                             .put("query", "")
205                             .put("limit", 0)));
206         guildIds = new JSONArray();
207       }
208     }
209 
210     // Send the remaining guilds that need to be sent
211     if (guildIds.length() > 0) {
212       api.getClient()
213           .chunkOrSyncRequest(
214               new JSONObject()
215                   .put("op", WebSocketCode.MEMBER_CHUNK_REQUEST)
216                   .put(
217                       "d",
218                       new JSONObject().put("guild_id", guildIds).put("query", "").put("limit", 0)));
219     }
220     guildsRequiringChunking.clear();
221   }
