17  package net.dv8tion.jda.core.handle;
18  
19  import net.dv8tion.jda.client.events.group.GroupJoinEvent;
20  import net.dv8tion.jda.core.entities.ChannelType;
21  import net.dv8tion.jda.core.entities.impl.JDAImpl;
22  import net.dv8tion.jda.core.events.channel.category.CategoryCreateEvent;
23  import net.dv8tion.jda.core.events.channel.priv.PrivateChannelCreateEvent;
24  import net.dv8tion.jda.core.events.channel.text.TextChannelCreateEvent;
25  import net.dv8tion.jda.core.events.channel.voice.VoiceChannelCreateEvent;
26  import org.json.JSONObject;
27  
28  public class ChannelCreateHandler extends SocketHandler {
29    public ChannelCreateHandler(JDAImpl api) {
30      super(api);
31    }
32  
33    @Override
34    protected Long handleInternally(JSONObject content) {
35      ChannelType type = ChannelType.fromId(content.getInt("type"));
36  
37      long guildId = 0;
38      if (type.isGuild()) {
39        guildId = content.getLong("guild_id");
40        if (api.getGuildLock().isLocked(guildId)) return guildId;
41      }
42  
43      switch (type) {
44        case TEXT:
45          {
46            api.getEventManager()
47                .handle(
48                    new TextChannelCreateEvent(
49                        api,
50                        responseNumber,
51                        api.getEntityBuilder().createTextChannel(content, guildId)));
52            break;
53          }
54        case VOICE:
55          {
56            api.getEventManager()
57                .handle(
58                    new VoiceChannelCreateEvent(
59                        api,
60                        responseNumber,
61                        api.getEntityBuilder().createVoiceChannel(content, guildId)));
62            break;
63          }
64        case CATEGORY:
65          {
66            api.getEventManager()
67                .handle(
68                    new CategoryCreateEvent(
69                        api,
70                        responseNumber,
71                        api.getEntityBuilder().createCategory(content, guildId)));
72            break;
73          }
74        case PRIVATE:
75          {
76            api.getEventManager()
77                .handle(
78                    new PrivateChannelCreateEvent(
79                        api, responseNumber, api.getEntityBuilder().createPrivateChannel(content)));
80            break;
81          }
82        case GROUP:
83          {
84            api.getEventManager()
85                .handle(
86                    new GroupJoinEvent(
87                        api, responseNumber, api.getEntityBuilder().createGroup(content)));
88            break;
89          }
90        default:
91          throw new IllegalArgumentException(
92              "Discord provided an CREATE_CHANNEL event with an unknown channel type! JSON: "
93                  + content);
94      }
95      api.getEventCache().playbackCache(EventCache.Type.CHANNEL, content.getLong("id"));
96      return null;
97    }
98  }
