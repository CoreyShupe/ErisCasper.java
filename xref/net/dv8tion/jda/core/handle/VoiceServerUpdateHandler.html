17  package net.dv8tion.jda.core.handle;
18  
19  import net.dv8tion.jda.core.audio.AudioConnection;
20  import net.dv8tion.jda.core.audio.AudioWebSocket;
21  import net.dv8tion.jda.core.entities.Guild;
22  import net.dv8tion.jda.core.entities.impl.JDAImpl;
23  import net.dv8tion.jda.core.managers.impl.AudioManagerImpl;
24  import net.dv8tion.jda.core.requests.WebSocketClient;
25  import org.json.JSONObject;
26  
27  public class VoiceServerUpdateHandler extends SocketHandler {
28    public VoiceServerUpdateHandler(JDAImpl api) {
29      super(api);
30    }
31  
32    @Override
33    protected Long handleInternally(JSONObject content) {
34      final long guildId = content.getLong("guild_id");
35      Guild guild = api.getGuildMap().get(guildId);
36      if (guild == null)
37        throw new IllegalArgumentException(
38            "Attempted to start audio connection with Guild that doesn't exist! JSON: " + content);
39  
40      api.getClient()
41          .updateAudioConnection(guildId, guild.getSelfMember().getVoiceState().getChannel());
42  
43      if (api.getGuildLock().isLocked(guildId)) return guildId;
44  
45      if (content.isNull("endpoint")) {
46        // Discord did not provide an endpoint yet, we are to wait until discord has resources to
47        // provide
48        // an endpoint, which will result in them sending another VOICE_SERVER_UPDATE which we will
49        // handle
50        // to actually connect to the audio server.
51        return null;
52      }
53  
54      String endpoint = content.getString("endpoint");
55      String token = content.getString("token");
56      String sessionId = guild.getSelfMember().getVoiceState().getSessionId();
57      if (sessionId == null)
58        throw new IllegalArgumentException(
59            "Attempted to create audio connection without having a session ID. Did VOICE_STATE_UPDATED fail?");
60  
61      // Strip the port from the endpoint.
62      endpoint = endpoint.replace(":80", "");
63  
64      AudioManagerImpl audioManager = (AudioManagerImpl) guild.getAudioManager();
65      synchronized (
66          audioManager
67              .CONNECTION_LOCK) // Synchronized to prevent attempts to close while setting up initial
68      // objects.
69      {
70        if (audioManager.isConnected()) audioManager.prepareForRegionChange();
71        if (!audioManager.isAttemptingToConnect()) {
72          WebSocketClient.LOG.debug(
73              "Received a VOICE_SERVER_UPDATE but JDA is not currently connected nor attempted to connect "
74                  + "to a VoiceChannel. Assuming that this is caused by another client running on this account. Ignoring the event.");
75          return null;
76        }
77  
78        AudioWebSocket socket =
79            new AudioWebSocket(
80                audioManager.getListenerProxy(),
81                endpoint,
82                api,
83                guild,
84                sessionId,
85                token,
86                audioManager.isAutoReconnect());
87        AudioConnection connection =
88            new AudioConnection(socket, audioManager.getQueuedAudioConnection());
89        audioManager.setAudioConnection(connection);
90        socket.startConnection();
91  
92        return null;
93      }
94    }
95  }
