39  public class VoiceStateUpdateHandler extends SocketHandler {
40    public VoiceStateUpdateHandler(JDAImpl api) {
41      super(api);
42    }
43  
44    @Override
45    protected Long handleInternally(JSONObject content) {
46      final Long guildId = content.has("guild_id") ? content.getLong("guild_id") : null;
47      if (guildId != null && api.getGuildLock().isLocked(guildId)) return guildId;
48  
49      if (guildId != null) handleGuildVoiceState(content);
50      else handleCallVoiceState(content);
51      return null;
52    }
53  
54    private void handleGuildVoiceState(JSONObject content) {
55      final long userId = content.getLong("user_id");
56      final long guildId = content.getLong("guild_id");
57      final Long channelId = !content.isNull("channel_id") ? content.getLong("channel_id") : null;
58      final String sessionId = !content.isNull("session_id") ? content.getString("session_id") : null;
59      boolean selfMuted = content.getBoolean("self_mute");
60      boolean selfDeafened = content.getBoolean("self_deaf");
61      boolean guildMuted = content.getBoolean("mute");
62      boolean guildDeafened = content.getBoolean("deaf");
63      boolean suppressed = content.getBoolean("suppress");
64  
65      Guild guild = api.getGuildById(guildId);
66      if (guild == null) {
67        api.getEventCache()
68            .cache(EventCache.Type.GUILD, guildId, () -> handle(responseNumber, allContent));
69        EventCache.LOG.debug(
70            "Received a VOICE_STATE_UPDATE for a Guild that has yet to be cached. JSON: {}", content);
71        return;
72      }
73  
74      VoiceChannelImpl channel =
75          channelId != null ? (VoiceChannelImpl) guild.getVoiceChannelById(channelId) : null;
76      if (channel == null && channelId != null) {
77        api.getEventCache()
78            .cache(EventCache.Type.CHANNEL, channelId, () -> handle(responseNumber, allContent));
79        EventCache.LOG.debug(
80            "Received VOICE_STATE_UPDATE for a VoiceChannel that has yet to be cached. JSON: {}",
81            content);
82        return;
83      }
84  
85      MemberImpl member = (MemberImpl) guild.getMemberById(userId);
86      if (member == null) {
87        // Caching of this might not be valid. It is possible that we received this
88        // update due to this Member leaving the guild while still connected to a voice channel.
89        // In that case, we should not cache this because it could cause problems if they rejoined.
90        // However, we can't just ignore it completely because it could be a user that joined off of
91        // an invite to a VoiceChannel, so the GUILD_MEMBER_ADD and the VOICE_STATE_UPDATE may have
92        // come out of order. Not quite sure what to do. Going to cache for now however.
93        // At the worst, this will just cause a few events to fire with bad data if the member rejoins
94        // the guild if
95        // in fact the issue was that the VOICE_STATE_UPDATE was sent after they had left, however, by
96        // caching
97        // it we will preserve the integrity of the cache in the event that it was actually a
98        // mis-ordering of
99        // GUILD_MEMBER_ADD and VOICE_STATE_UPDATE. I'll take some bad-data events over an invalid
100       // cache.
101       api.getEventCache()
102           .cache(EventCache.Type.USER, userId, () -> handle(responseNumber, allContent));
103       EventCache.LOG.debug(
104           "Received VOICE_STATE_UPDATE for a Member that has yet to be cached. JSON: {}", content);
105       return;
106     }
107 
108     GuildVoiceStateImpl vState = (GuildVoiceStateImpl) member.getVoiceState();
109     vState.setSessionId(sessionId); // Cant really see a reason for an event for this
110 
111     if (!Objects.equals(channel, vState.getChannel())) {
112       VoiceChannelImpl oldChannel = (VoiceChannelImpl) vState.getChannel();
113       vState.setConnectedChannel(channel);
114 
115       if (oldChannel == null) {
116         channel.getConnectedMembersMap().put(userId, member);
117         api.getEventManager().handle(new GuildVoiceJoinEvent(api, responseNumber, member));
118       } else if (channel == null) {
119         oldChannel.getConnectedMembersMap().remove(userId);
120         if (guild.getSelfMember().equals(member))
121           api.getClient().updateAudioConnection(guildId, null);
122         api.getEventManager()
123             .handle(
124                 new GuildVoiceLeaveEvent(
125                     api, responseNumber,
126                     member, oldChannel));
127       } else {
128         AudioManagerImpl mng = (AudioManagerImpl) api.getAudioManagerMap().get(guildId);
129 
130         // If the currently connected account is the one that is being moved
131         if (guild.getSelfMember().equals(member) && mng != null) {
132           // And this instance of JDA is connected or attempting to connect,
133           // then change the channel we expect to be connected to.
134           if (mng.isConnected() || mng.isAttemptingToConnect()) mng.setConnectedChannel(channel);
135 
136           // If we have connected (VOICE_SERVER_UPDATE received and AudioConnection created (actual
137           // connection might still be setting up)),
138           // then we need to stop sending audioOpen/Move requests through the MainWS if the channel
139           // we have just joined / moved to is the same as the currently queued audioRequest
140           // (handled by updateAudioConnection)
141           if (mng.isConnected()) api.getClient().updateAudioConnection(guildId, channel);
142           // If we are not already connected this will be removed by VOICE_SERVER_UPDATE
143         }
144 
145         channel.getConnectedMembersMap().put(userId, member);
146         oldChannel.getConnectedMembersMap().remove(userId);
147         api.getEventManager()
148             .handle(
149                 new GuildVoiceMoveEvent(
150                     api, responseNumber,
151                     member, oldChannel));
152       }
153     }
154 
155     boolean wasMute = vState.isMuted();
156     boolean wasDeaf = vState.isDeafened();
157 
158     if (selfMuted != vState.isSelfMuted()) {
159       vState.setSelfMuted(selfMuted);
160       api.getEventManager().handle(new GuildVoiceSelfMuteEvent(api, responseNumber, member));
161     }
162     if (selfDeafened != vState.isSelfDeafened()) {
163       vState.setSelfDeafened(selfDeafened);
164       api.getEventManager().handle(new GuildVoiceSelfDeafenEvent(api, responseNumber, member));
165     }
166     if (guildMuted != vState.isGuildMuted()) {
167       vState.setGuildMuted(guildMuted);
168       api.getEventManager().handle(new GuildVoiceGuildMuteEvent(api, responseNumber, member));
169     }
170     if (guildDeafened != vState.isGuildDeafened()) {
171       vState.setGuildDeafened(guildDeafened);
172       api.getEventManager().handle(new GuildVoiceGuildDeafenEvent(api, responseNumber, member));
173     }
174     if (suppressed != vState.isSuppressed()) {
175       vState.setSuppressed(suppressed);
176       api.getEventManager().handle(new GuildVoiceSuppressEvent(api, responseNumber, member));
177     }
178     if (wasMute != vState.isMuted())
179       api.getEventManager().handle(new GuildVoiceMuteEvent(api, responseNumber, member));
180     if (wasDeaf != vState.isDeafened())
181       api.getEventManager().handle(new GuildVoiceDeafenEvent(api, responseNumber, member));
182   }

184   private void handleCallVoiceState(JSONObject content) {
185     final long userId = content.getLong("user_id");
186     final Long channelId = content.isNull("channel_id") ? null : content.getLong("channel_id");
187     String sessionId = content.optString("session_id", null);
188     boolean selfMuted = content.getBoolean("self_mute");
189     boolean selfDeafened = content.getBoolean("self_deaf");
190 
191     // Joining a call
192     CallableChannel channel;
193     CallVoiceStateImpl vState;
194     if (channelId != null) {
195       channel = api.asClient().getGroupById(channelId);
196       if (channel == null) channel = api.getPrivateChannelMap().get(channelId);
197 
198       if (channel == null) {
199         api.getEventCache()
200             .cache(EventCache.Type.CHANNEL, channelId, () -> handle(responseNumber, allContent));
201         EventCache.LOG.debug(
202             "Received a VOICE_STATE_UPDATE for a Group/PrivateChannel that was not yet cached! JSON: {}",
203             content);
204         return;
205       }
206 
207       CallImpl call = (CallImpl) channel.getCurrentCall();
208       if (call == null) {
209         api.getEventCache()
210             .cache(EventCache.Type.CALL, channelId, () -> handle(responseNumber, allContent));
211         EventCache.LOG.debug(
212             "Received a VOICE_STATE_UPDATE for a Call that is not yet cached. JSON: {}", content);
213         return;
214       }
215 
216       CallUser cUser = api.asClient().getCallUserMap().get(userId);
217       if (cUser != null && channelId != cUser.getCall().getCallableChannel().getIdLong()) {
218         WebSocketClient.LOG.error(
219             "Received a VOICE_STATE_UPDATE for a user joining a call, but the user was already in a different call! Big error! JSON: {}",
220             content);
221         ((CallVoiceStateImpl) cUser.getVoiceState()).setInCall(false);
222       }
223 
224       cUser = call.getCallUserMap().get(userId);
225       if (cUser == null) {
226         api.getEventCache()
227             .cache(EventCache.Type.USER, userId, () -> handle(responseNumber, allContent));
228         EventCache.LOG.debug(
229             "Received a VOICE_STATE_UPDATE for a user that is not yet a a cached CallUser for the call. (groups only). JSON: {}",
230             content);
231         return;
232       }
233 
234       api.asClient().getCallUserMap().put(userId, cUser);
235       vState = (CallVoiceStateImpl) cUser.getVoiceState();
236       vState.setSessionId(sessionId);
237       vState.setInCall(true);
238 
239       api.getEventManager().handle(new CallVoiceJoinEvent(api, responseNumber, cUser));
240     } else // Leaving a call
241     {
242       CallUser cUser = api.asClient().getCallUserMap().remove(userId);
243       if (cUser == null) {
244         api.getEventCache()
245             .cache(EventCache.Type.USER, userId, () -> handle(responseNumber, allContent));
246         EventCache.LOG.debug(
247             "Received a VOICE_STATE_UPDATE for a User leaving a Call, but the Call was not yet cached! JSON: {}",
248             content);
249         return;
250       }
251 
252       Call call = cUser.getCall();
253       channel = call.getCallableChannel();
254       vState = (CallVoiceStateImpl) cUser.getVoiceState();
255       vState.setSessionId(sessionId);
256       vState.setInCall(false);
257 
258       api.getEventManager().handle(new CallVoiceLeaveEvent(api, responseNumber, cUser));
259     }
260 
261     // Now that we're done dealing with the joins and leaves, we can deal with the mute/deaf
262     // changes.
263     if (selfMuted != vState.isSelfMuted()) {
264       vState.setSelfMuted(selfMuted);
265       api.getEventManager()
266           .handle(new CallVoiceSelfMuteEvent(api, responseNumber, vState.getCallUser()));
267     }
268     if (selfDeafened != vState.isSelfDeafened()) {
269       vState.setSelfDeafened(selfDeafened);
270       api.getEventManager()
271           .handle(new CallVoiceSelfDeafenEvent(api, responseNumber, vState.getCallUser()));
272     }
273   }
