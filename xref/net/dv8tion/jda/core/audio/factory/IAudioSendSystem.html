17  package net.dv8tion.jda.core.audio.factory;
18  
19  import java.util.concurrent.ConcurrentMap;
20  
21  /**
22   * Interface that acts as a UDP audio packet sending loop. <br>
23   * This interface is provided so that developers can provide their own implementation for different
24   * management of thread pools, process usage, and even for forwarding to native binaries implemented
25   * in other languages like C to avoid problems with JVM GC StopTheWorld events. <br>
26   * JDA, by default, uses {@link net.dv8tion.jda.core.audio.factory.DefaultSendSystem
27   * DefaultSendSystem} for its UDP audio packet sending loop.
28   */
29  public interface IAudioSendSystem {
30    /**
31     * This represents the start of the loop, similar to {@link Thread#start()}, and after a call to
32     * this method JDA assumes that the instance will be sending UDP audio packets in a loop.
33     *
34     * <p><b>Note:</b> The packet sending loop should NOT be started on the current thread. I.E: This
35     * method should not block forever, in the same way that {@link Thread#start()} does not. Just
36     * like in Thread, the running action of this system should be implemented asynchronously.
37     */
38    void start();
39  
40    /**
41     * This represents the destruction of this instance and should be used to perform all necessary
42     * cleanup and shutdown operations needed to free resources.
43     *
44     * <p><b>Note:</b> This method can be called at any time after instance creation ({@link #start()}
45     * may not yet have been called), and it is possible that this method could be called more than
46     * once due to edge-case shutdown conditions.
47     */
48    void shutdown();
49  
50    /**
51     * Called with the internal JDA {@link org.slf4j.MDC MDC} context map. <br>
52     * This is guaranteed to be called before {@link #start()}.
53     *
54     * @param contextMap The JDA internal MDC context map, or {@code null} if disabled
55     */
56    default void setContextMap(ConcurrentMap<String, String> contextMap) {}
57  }
