17  package net.dv8tion.jda.core.audio;
18  
19  import javax.sound.sampled.AudioFormat;
20  
21  /** Interface used to send audio to Discord through JDA. */
22  public interface AudioSendHandler {
23    /**
24     * Audio Input Format expected by JDA if {@link #isOpus()} returns false. 48KHz 16bit stereo
25     * signed BigEndian PCM.
26     */
27    AudioFormat INPUT_FORMAT = new AudioFormat(48000f, 16, 2, true, true);
28  
29    /**
30     * If this method returns true JDA will attempt to retrieve audio data from this handler by
31     * calling {@link #provide20MsAudio()}. The return value is checked each time JDA attempts send
32     * audio, so if the developer wanted to start and stop sending audio it could be done by changing
33     * the value returned by this method at runtime.
34     *
35     * @return If true, JDA will attempt to retrieve audio data from {@link #provide20MsAudio()}
36     */
37    boolean canProvide();
38  
39    /**
40     * If {@link #canProvide()} returns true JDA will call this method in an attempt to retrieve audio
41     * data from the handler. This method need to provide 20 Milliseconds of audio data as a byte
42     * array.
43     *
44     * <p>Considering this system needs to be low-latency / high-speed, it is recommended that the
45     * loading of audio data be done before hand or in parallel and not loaded from disk when this
46     * method is called by JDA. Attempting to load all audio data from disk when this method is called
47     * will most likely cause issues due to IO blocking this thread.
48     *
49     * <p>The provided audio data needs to be in the format: 48KHz 16bit stereo signed BigEndian PCM.
50     * <br>
51     * Defined by: {@link net.dv8tion.jda.core.audio.AudioSendHandler#INPUT_FORMAT
52     * AudioSendHandler.INPUT_FORMAT}. <br>
53     * If {@link #isOpus()} is set to return true, then it should be in pre-encoded Opus format
54     * instead.
55     *
56     * @return Should return a byte[] containing 20 Milliseconds of audio.
57     */
58    byte[] provide20MsAudio();
59  
60    /**
61     * If this method returns true JDA will treat the audio data provided by {@link
62     * #provide20MsAudio()} as a pre-encoded 20 Millisecond packet of Opus audio. This means that JDA
63     * <b>WILL NOT</b> attempt to encode the audio as Opus, but will provide it to Discord <b>exactly
64     * as it is given</b>.
65     *
66     * @return If true, JDA will not attempt to encode the provided audio data as Opus. <br>
67     *     Default - False.
68     */
69    default boolean isOpus() {
70      return false;
71    }
72  }
