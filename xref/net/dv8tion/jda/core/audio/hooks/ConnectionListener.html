17  package net.dv8tion.jda.core.audio.hooks;
18  
19  import net.dv8tion.jda.core.entities.User;
20  
21  /**
22   * Used to monitor an audio connection, ping, and speaking users. <br>
23   * This provides functionality similar to the functionalities present in the Discord client related
24   * to an audio connection.
25   */
26  public interface ConnectionListener {
27    /**
28     * Called when JDA send a heartbeat packet to Discord and Discord sends an acknowledgement. The
29     * time difference between sending and receiving the acknowledgement is calculated as the ping.
30     *
31     * @param ping The time, in milliseconds, for round-trip packet travel to discord.
32     */
33    void onPing(long ping);
34  
35    /**
36     * Called when the status of the audio channel changes. Used to track the connection state of the
37     * audio connection for easy debug and status display for clients.
38     *
39     * @param status The new {@link net.dv8tion.jda.core.audio.hooks.ConnectionStatus
40     *     ConnectionStatus} of the audio connection.
41     */
42    void onStatusChange(ConnectionStatus status);
43  
44    /**
45     * This method is an easy way to detect if a user is talking. Discord sends us an event when a
46     * user starts or stops talking and it is parallel to the audio socket, so this event could come
47     * milliseconds before or after audio begins or stops. This method is brilliant for clients
48     * wanting to display that a user is currently talking.
49     *
50     * <p>Unlike the {@link
51     * net.dv8tion.jda.core.audio.AudioReceiveHandler#handleCombinedAudio(net.dv8tion.jda.core.audio.CombinedAudio)
52     * AudioReceiveHandler.handleCombinedAudio(CombinedAudio)} and {@link
53     * net.dv8tion.jda.core.audio.AudioReceiveHandler#handleUserAudio(net.dv8tion.jda.core.audio.UserAudio)
54     * AudioReceiveHandler.handleUserAudio(UserAudio)} methods which are fired extremely often, this
55     * method is fired as a flag for the beginning and ending of audio transmission, and as such is
56     * only fired when that changes. So while the {@link
57     * net.dv8tion.jda.core.audio.AudioReceiveHandler#handleUserAudio(net.dv8tion.jda.core.audio.UserAudio)
58     * AudioReceiveHandler.handleUserAudio(UserAudio)} method is fired every time JDA receives audio
59     * data from Discord, this is only fired when that stream starts and when it stops. <br>
60     * If the user speaks for 3 minutes straight without ever stopping, then this would fire 2 times,
61     * once at the beginning and once after 3 minutes when they stop talking even though the {@link
62     * net.dv8tion.jda.core.audio.AudioReceiveHandler#handleUserAudio(net.dv8tion.jda.core.audio.UserAudio)
63     * AudioReceiveHandler.handleUserAudio(UserAudio)} method was fired thousands of times over the
64     * course of the 3 minutes.
65     *
66     * @param user Never-null {@link net.dv8tion.jda.core.entities.User User} who's talking status has
67     *     changed.
68     * @param speaking If true, the user has begun transmitting audio.
69     */
70    void onUserSpeaking(User user, boolean speaking);
71  }
