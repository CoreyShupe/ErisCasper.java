17  package net.dv8tion.jda.core.audio.hooks;
18  
19  /** Represents the connection status of an audio connection. */
20  public enum ConnectionStatus {
21    /**
22     * Indicates that there is no open connection or that the connection was closed by choice, not by
23     * error.
24     */
25    NOT_CONNECTED,
26    /** JDA is waiting on Discord to send a valid endpoint which to connect the audio websocket to. */
27    CONNECTING_AWAITING_ENDPOINT,
28    /**
29     * JDA has received a valid endpoint and is attempting to setup and connect the audio websocket
30     */
31    CONNECTING_AWAITING_WEBSOCKET_CONNECT,
32    /**
33     * JDA has connected the audio websocket to Discord and has sent the authentication information,
34     * awaiting reply.
35     */
36    CONNECTING_AWAITING_AUTHENTICATING,
37    /**
38     * JDA successfully authenticated the audio websocket and it now attempting UDP discovery. UDP
39     * discovery involves opening a UDP socket and sending a packet to a provided Discord remote
40     * resource which responds with the external ip and port which the packet was sent from.
41     */
42    CONNECTING_ATTEMPTING_UDP_DISCOVERY,
43    /**
44     * After determining our external ip and port, JDA forwards this information to Discord so that it
45     * can send audio packets for us to properly receive. At this point, JDA is waiting for final
46     * websocket READY.
47     */
48    CONNECTING_AWAITING_READY,
49    /** The audio connection has been successfully setup and is ready for use. */
50    CONNECTED,
51    /**
52     * Indicates that the logged in account lost the {@link
53     * net.dv8tion.jda.core.Permission#VOICE_CONNECT Permission.VOICE_CONNECT} and cannot connect to
54     * the channel. <br>
55     * <b>This is a non-reconnectable error.</b>
56     */
57    DISCONNECTED_LOST_PERMISSION,
58    /**
59     * Indicates that the channel which the audio connection was connected to was deleted, thus the
60     * connection was severed. <br>
61     * <b>This is a non-reconnectable error.</b>
62     */
63    DISCONNECTED_CHANNEL_DELETED,
64    /**
65     * Indicates that the logged in account was removed from the {@link
66     * net.dv8tion.jda.core.entities.Guild Guild} that this audio connection was connected to, thus
67     * the connection was severed. <br>
68     * <b>This is a non-reconnectable error.</b>
69     */
70    DISCONNECTED_REMOVED_FROM_GUILD,
71    /** Indicates that our token was not valid. */
72    DISCONNECTED_AUTHENTICATION_FAILURE,
73    /**
74     * Indicates that the audio connection was closed due to the {@link net.dv8tion.jda.core.Region
75     * Region} of the audio connection being changed. JDA will automatically attempt to reconnect the
76     * audio connection regardless of the value of the {@link
77     * net.dv8tion.jda.core.managers.AudioManager#isAutoReconnect() AudioManager.isAutoReconnect()}.
78     */
79    AUDIO_REGION_CHANGE,
80  
81    // All will attempt to reconnect unless autoReconnect is disabled
82    /**
83     * Indicates that the connection was lost, either via UDP socket problems or the audio Websocket
84     * disconnecting. <br>
85     * This is typically caused by a brief loss of internet which results in connection loss. <br>
86     * JDA automatically attempts to resume the session when this error occurs.
87     */
88    ERROR_LOST_CONNECTION,
89    /**
90     * Indicates that the audio WebSocket was unable to resume an active session. <br>
91     * JDA automatically attempts to reconnect when this error occurs.
92     */
93    ERROR_CANNOT_RESUME,
94    /**
95     * Indicates that the audio Websocket was unable to connect to discord. This could be due to an
96     * internet problem causing a connection problem or an error on Discord's side (possibly due to
97     * load) <br>
98     * JDA automatically attempts to reconnect when this error occurs.
99     */
100   ERROR_WEBSOCKET_UNABLE_TO_CONNECT,
101   /**
102    * Indicates that the UDP setup failed. This is caused when JDA cannot properly communicate with
103    * Discord to discover the system's external IP and port which audio data will be sent from.
104    * Typically caused by an internet problem or an overly aggressive NAT port table. <br>
105    * JDA automatically attempts to reconnect when this error occurs.
106    */
107   ERROR_UDP_UNABLE_TO_CONNECT,
108   /**
109    * Occurs when it takes longer than {@link
110    * net.dv8tion.jda.core.managers.AudioManager#getConnectTimeout()
111    * AudioManager.getConnectTimeout()} to establish the Websocket connection and setup the UDP
112    * connection. <br>
113    * JDA automatically attempts to reconnect when this error occurs.
114    */
115   ERROR_CONNECTION_TIMEOUT
116 }
