52    public AudioConnection(AudioWebSocket webSocket, VoiceChannel channel) {
53      this.channel = channel;
54      this.webSocket = webSocket;
55      this.webSocket.audioConnection = this;
56  
57      final JDAImpl api = (JDAImpl) channel.getJDA();
58      this.threadIdentifier =
59          api.getIdentifierString() + " AudioConnection Guild: " + channel.getGuild().getId();
60      this.contextMap = api.getContextMap();
61    }

63    public void ready() {
64      Thread readyThread =
65          new Thread(
66              AudioManagerImpl.AUDIO_THREADS,
67              () -> {
68                if (contextMap != null) MDC.setContextMap(contextMap);
69                final long timeout = getGuild().getAudioManager().getConnectTimeout();
70  
71                final long started = System.currentTimeMillis();
72                boolean connectionTimeout = false;
73                while (!webSocket.isReady()) {
74                  if (timeout > 0 && System.currentTimeMillis() - started > timeout) {
75                    connectionTimeout = true;
76                    break;
77                  }
78  
79                  try {
80                    Thread.sleep(10);
81                  } catch (InterruptedException e) {
82                    LOG.error("AudioConnection ready thread got interrupted while sleeping", e);
83                    Thread.currentThread().interrupt();
84                  }
85                }
86                if (!connectionTimeout) {
87                  this.udpSocket = webSocket.getUdpSocket();
88  
89                  setupReceiveSystem();
90                } else {
91                  webSocket.close(ConnectionStatus.ERROR_CONNECTION_TIMEOUT);
92                }
93              });
94      readyThread.setUncaughtExceptionHandler(
95          (thread, throwable) -> {
96            LOG.error("Uncaught exception in Audio ready-thread", throwable);
97            JDAImpl api = (JDAImpl) getJDA();
98            api.getEventManager().handle(new ExceptionEvent(api, throwable, true));
99          });
100     readyThread.setDaemon(true);
101     readyThread.setName(threadIdentifier + " Ready Thread");
102     readyThread.start();
103   }

126   public void removeUserSSRC(long userId) {
127     final AtomicInteger ssrcRef = new AtomicInteger(0);
128     final boolean modified =
129         ssrcMap.retainEntries(
130             (ssrc, id) -> {
131               final boolean isEntry = id == userId;
132               if (isEntry) ssrcRef.set(ssrc);
133               // if isEntry == true we don't want to retain it
134               return !isEntry;
135             });
136     if (!modified) return;
137   }
138 
139   protected void updateUserSSRC(int ssrc, long userId) {
140     if (ssrcMap.containsKey(ssrc)) {
141       long previousId = ssrcMap.get(ssrc);
142       if (previousId != userId) {
143         // Different User already existed with this ssrc. What should we do? Just replace? Probably
144         // should nuke the old opusDecoder.
145         // Log for now and see if any user report the error.
146         LOG.error(
147             "Yeah.. So.. JDA received a UserSSRC update for an ssrc that already had a User set. Inform DV8FromTheWorld.\nChannelId: {} SSRC: {} oldId: {} newId: {}",
148             channel.getId(),
149             ssrc,
150             previousId,
151             userId);
152       }
153     } else {
154       ssrcMap.put(ssrc, userId);
155     }
156   }

158   public void close(ConnectionStatus closeStatus) {
159     shutdown();
160     webSocket.close(closeStatus);
161   }

163   public synchronized void shutdown() {
164     //        setSpeaking(false);
165     if (sendSystem != null) {
166       sendSystem.shutdown();
167       sendSystem = null;
168     }
169     if (receiveThread != null) {
170       receiveThread.interrupt();
171       receiveThread = null;
172     }
173     if (combinedAudioExecutor != null) {
174       combinedAudioExecutor.shutdownNow();
175       combinedAudioExecutor = null;
176     }
177   }
178 
179   private synchronized void setupReceiveSystem() {
180     if (udpSocket != null
181         && !udpSocket.isClosed()
182         && receiveHandler != null
183         && receiveThread == null) {
184 
185     } else if (receiveHandler == null && receiveThread != null) {
186       receiveThread.interrupt();
187       receiveThread = null;
188 
189       if (combinedAudioExecutor != null) {
190         combinedAudioExecutor.shutdownNow();
191         combinedAudioExecutor = null;
192       }
193     } else if (receiveHandler != null
194         && !receiveHandler.canReceiveCombined()
195         && combinedAudioExecutor != null) {
196       combinedAudioExecutor.shutdownNow();
197       combinedAudioExecutor = null;
198     }
199   }
