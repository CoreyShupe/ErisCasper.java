30  /**
31   * Builder system used to build {@link net.dv8tion.jda.core.entities.Message Messages}. <br>
32   * Internally the builder uses a {@link java.lang.StringBuilder} to take advantage of the
33   * efficiencies offered by the StringBuilder, and the methods provided by this class are a
34   * combination of those offered by the StringBuilder and {@link String#format(String, Object...)}.
35   *
36   * @since 1.0
37   * @author Michael Ritter
38   * @author Aljoscha Grebe
39   */
40  public class MessageBuilder implements Appendable {
41    protected final StringBuilder builder = new StringBuilder();
42  
43    protected boolean isTTS = false;
44    protected String nonce;
45    protected MessageEmbed embed;
46  
47    public MessageBuilder() {}
48  
49    public MessageBuilder(CharSequence content) {
50      if (content != null) builder.append(content);
51    }
52  
53    public MessageBuilder(Message message) {
54      if (message != null) {
55        isTTS = message.isTTS();
56        builder.append(message.getContentRaw());
57        List<MessageEmbed> embeds = message.getEmbeds();
58        if (embeds != null && !embeds.isEmpty()) embed = embeds.get(0);
59      }
60    }
61  
62    public MessageBuilder(MessageBuilder builder) {
63      if (builder != null) {
64        this.isTTS = builder.isTTS;
65        this.builder.append(builder.builder);
66        this.nonce = builder.nonce;
67        this.embed = builder.embed;
68      }
69    }
70  
71    public MessageBuilder(EmbedBuilder builder) {
72      if (builder != null) this.embed = builder.build();
73    }

75    public MessageBuilder(MessageEmbed embed) {
76      this.embed = embed;
77    }

124   /**
125    * Sets the content of the resulting Message <br>
126    * This will replace already added content.
127    *
128    * @param content The content to use, or {@code null} to reset the content
129    * @throws java.lang.IllegalArgumentException If the provided content exceeds {@link
130    *     net.dv8tion.jda.core.entities.Message#MAX_CONTENT_LENGTH Message.MAX_CONTENT_LENGTH}
131    * @return The MessageBuilder instance. Useful for chaining.
132    * @see net.dv8tion.jda.core.entities.Message#getContentRaw()
133    */
134   public MessageBuilder setContent(String content) {
135     if (content == null) {
136       builder.setLength(0);
137     } else {
138       Checks.check(
139           content.length() <= Message.MAX_CONTENT_LENGTH,
140           "Content length may not exceed %d!",
141           Message.MAX_CONTENT_LENGTH);
142       final int newLength = Math.max(builder.length(), content.length());
143       builder.replace(0, newLength, content);
144     }
145     return this;
146   }

191   /**
192    * Appends a String using the specified chat {@link net.dv8tion.jda.core.MessageBuilder.Formatting
193    * Formatting(s)}.
194    *
195    * @param text the text to append.
196    * @param format the format(s) to apply to the text.
197    * @return The MessageBuilder instance. Useful for chaining.
198    */
199   public MessageBuilder append(CharSequence text, Formatting... format) {
200     boolean blockPresent = false;
201     for (Formatting formatting : format) {
202       if (formatting == Formatting.BLOCK) {
203         blockPresent = true;
204         continue;
205       }
206       builder.append(formatting.getTag());
207     }
208     if (blockPresent) builder.append(Formatting.BLOCK.getTag());
209 
210     builder.append(text);
211 
212     if (blockPresent) builder.append(Formatting.BLOCK.getTag());
213     for (int i = format.length - 1; i >= 0; i--) {
214       if (format[i] == Formatting.BLOCK) continue;
215       builder.append(format[i].getTag());
216     }
217     return this;
218   }

220   /**
221    * This method is an extended form of {@link String#format(String, Object...)}. It allows for all
222    * of the token replacement functionality that String.format(String, Object...) supports. <br>
223    * A lot of JDA entities implement {@link java.util.Formattable Formattable} and will provide
224    * specific format outputs for their specific type.
225    *
226    * <ul>
227    *   <li>{@link net.dv8tion.jda.core.entities.IMentionable IMentionable} <br>
228    *       These will output their {@link net.dv8tion.jda.core.entities.IMentionable#getAsMention()
229    *       getAsMention} by default, some implementations have alternatives such as {@link
230    *       net.dv8tion.jda.core.entities.User User} and {@link
231    *       net.dv8tion.jda.core.entities.TextChannel TextChannel}.
232    *   <li>{@link net.dv8tion.jda.core.entities.MessageChannel MessageChannel} <br>
233    *       All message channels format to {@code "#" + getName()} by default, TextChannel has
234    *       special handling and uses the getAsMention output by default and the MessageChannel
235    *       output as alternative ({@code #} flag).
236    *   <li>{@link net.dv8tion.jda.core.entities.Message Message} <br>
237    *       Messages by default output their {@link
238    *       net.dv8tion.jda.core.entities.Message#getContentDisplay() getContentDisplay()} value and
239    *       as alternative use the {@link net.dv8tion.jda.core.entities.Message#getContentRaw()
240    *       getContentRaw()} value
241    * </ul>
242    *
243    * <p>Example: <br>
244    * If you placed the following code in an method handling a {@link
245    * net.dv8tion.jda.core.events.message.MessageReceivedEvent MessageReceivedEvent} <br>
246    *
247    * <pre>{@code
248    * User user = event.getAuthor();
249    * MessageBuilder builder = new MessageBuilder();
250    * builder.appendFormat("%#s is really cool!", user);
251    * builder.build();
252    * }</pre>
253    *
254    * It would build a message that mentions the author and says that he is really cool!. If the
255    * user's name was "Minn" and his discriminator "6688", it would say: <br>
256    *
257    * <pre>  "Minn#6688 is really cool!"</pre>
258    *
259    * <br>
260    * Note that this uses the {@code #} flag to utilize the alternative format for {@link
261    * net.dv8tion.jda.core.entities.User User}. <br>
262    * By default it would fallback to {@link
263    * net.dv8tion.jda.core.entities.IMentionable#getAsMention()}
264    *
265    * @param format a format string.
266    * @param args an array objects that will be used to replace the tokens, they must be provided in
267    *     the order that the tokens appear in the provided format string.
268    * @throws java.lang.IllegalArgumentException If the provided format string is {@code null} or
269    *     empty
270    * @throws java.util.IllegalFormatException If a format string contains an illegal syntax, a
271    *     format specifier that is incompatible with the given arguments, insufficient arguments
272    *     given the format string, or other illegal conditions. For specification of all possible
273    *     formatting errors, see the <a href="../util/Formatter.html#detail">Details</a> section of
274    *     the formatter class specification.
275    * @return The MessageBuilder instance. Useful for chaining.
276    */
277   public MessageBuilder appendFormat(String format, Object... args) {
278     Checks.notEmpty(format, "Format String");
279     this.append(String.format(format, args));
280     return this;
281   }

324   /**
325    * Replaces each substring that matches the target string with the specified replacement string.
326    * The replacement proceeds from the beginning of the string to the end, for example, replacing
327    * "aa" with "b" in the message "aaa" will result in "ba" rather than "ab".
328    *
329    * @param target the sequence of char values to be replaced
330    * @param replacement the replacement sequence of char values
331    * @return The MessageBuilder instance. Useful for chaining.
332    */
333   public MessageBuilder replace(String target, String replacement) {
334     int index = builder.indexOf(target);
335     while (index != -1) {
336       builder.replace(index, index + target.length(), replacement);
337       index = builder.indexOf(target, index + replacement.length());
338     }
339     return this;
340   }

441   private MessageBuilder stripMentions(JDA jda, Guild guild, Message.MentionType... types) {
442     if (types == null) return this;
443 
444     String string = null;
445 
446     for (Message.MentionType mention : types) {
447       if (mention != null) {
448         switch (mention) {
449           case EVERYONE:
450             replace("@everyone", "@\u0435veryone");
451             break;
452           case HERE:
453             replace("@here", "@h\u0435re");
454             break;
455           case CHANNEL:
456             {
457               if (string == null) {
458                 string = builder.toString();
459               }
460 
461               Matcher matcher = Message.MentionType.CHANNEL.getPattern().matcher(string);
462               while (matcher.find()) {
463                 TextChannel channel = jda.getTextChannelById(matcher.group(1));
464                 if (channel != null) {
465                   replace(matcher.group(), "#" + channel.getName());
466                 }
467               }
468               break;
469             }
470           case ROLE:
471             {
472               if (string == null) {
473                 string = builder.toString();
474               }
475 
476               Matcher matcher = Message.MentionType.ROLE.getPattern().matcher(string);
477               while (matcher.find()) {
478                 for (Guild g : jda.getGuilds()) {
479                   Role role = g.getRoleById(matcher.group(1));
480                   if (role != null) {
481                     replace(matcher.group(), "@" + role.getName());
482                     break;
483                   }
484                 }
485               }
486               break;
487             }
488           case USER:
489             {
490               if (string == null) {
491                 string = builder.toString();
492               }
493 
494               Matcher matcher = Message.MentionType.USER.getPattern().matcher(string);
495               while (matcher.find()) {
496                 User user = jda.getUserById(matcher.group(1));
497                 String replacement;
498 
499                 if (user == null) continue;
500 
501                 Member member;
502 
503                 if (guild != null && (member = guild.getMember(user)) != null)
504                   replacement = member.getEffectiveName();
505                 else replacement = user.getName();
506 
507                 replace(matcher.group(), "@" + replacement);
508               }
509               break;
510             }
511         }
512       }
513     }
514 
515     return this;
516   }

539   /**
540    * Returns the index within this string of the first occurrence of the specified substring between
541    * the specified indices.
542    *
543    * <p>If no such value of {@code target} exists, then {@code -1} is returned.
544    *
545    * @param target the substring to search for.
546    * @param fromIndex the index from which to start the search.
547    * @param endIndex the index at which to end the search.
548    * @throws java.lang.IndexOutOfBoundsException
549    *     <ul>
550    *       <li>If the {@code fromIndex} is outside of the range of {@code 0} to {@link #length()}
551    *       <li>If the {@code endIndex} is outside of the range of {@code 0} to {@link #length()}
552    *       <li>If the {@code fromIndex} is greater than {@code endIndex}
553    *     </ul>
554    *
555    * @return the index of the first occurrence of the specified substring between the specified
556    *     indices or {@code -1} if there is no such occurrence.
557    */
558   public int indexOf(CharSequence target, int fromIndex, int endIndex) {
559     if (fromIndex < 0) throw new IndexOutOfBoundsException("index out of range: " + fromIndex);
560     if (endIndex < 0) throw new IndexOutOfBoundsException("index out of range: " + endIndex);
561     if (fromIndex > length()) throw new IndexOutOfBoundsException("fromIndex > length()");
562     if (fromIndex > endIndex) throw new IndexOutOfBoundsException("fromIndex > endIndex");
563 
564     if (endIndex >= builder.length()) {
565       endIndex = builder.length() - 1;
566     }
567 
568     int targetCount = target.length();
569     if (targetCount == 0) {
570       return fromIndex;
571     }
572 
573     char strFirstChar = target.charAt(0);
574     int max = endIndex + targetCount - 1;
575 
576     lastCharSearch:
577     for (int i = fromIndex; i <= max; i++) {
578       if (builder.charAt(i) == strFirstChar) {
579         for (int j = 1; j < targetCount; j++) {
580           if (builder.charAt(i + j) != target.charAt(j)) {
581             continue lastCharSearch;
582           }
583         }
584         return i;
585       }
586     }
587     return -1;
588   }

590   /**
591    * Returns the index within this string of the last occurrence of the specified substring between
592    * the specified indices.
593    *
594    * <p>If no such value of {@code target} exists, then {@code -1} is returned.
595    *
596    * @param target the substring to search for.
597    * @param fromIndex the index from which to start the search.
598    * @param endIndex the index at which to end the search.
599    * @throws java.lang.IndexOutOfBoundsException
600    *     <ul>
601    *       <li>If the {@code fromIndex} is outside of the range of {@code 0} to {@link #length()}
602    *       <li>If the {@code endIndex} is outside of the range of {@code 0} to {@link #length()}
603    *       <li>If the {@code fromIndex} is greater than {@code endIndex}
604    *     </ul>
605    *
606    * @return the index of the last occurrence of the specified substring between the specified
607    *     indices or {@code -1} if there is no such occurrence.
608    */
609   public int lastIndexOf(CharSequence target, int fromIndex, int endIndex) {
610     if (fromIndex < 0) throw new IndexOutOfBoundsException("index out of range: " + fromIndex);
611     if (endIndex < 0) throw new IndexOutOfBoundsException("index out of range: " + endIndex);
612     if (fromIndex > length()) throw new IndexOutOfBoundsException("fromIndex > length()");
613     if (fromIndex > endIndex) throw new IndexOutOfBoundsException("fromIndex > endIndex");
614 
615     if (endIndex >= builder.length()) {
616       endIndex = builder.length() - 1;
617     }
618 
619     int targetCount = target.length();
620     if (targetCount == 0) {
621       return endIndex;
622     }
623 
624     int rightIndex = endIndex - targetCount;
625 
626     if (fromIndex > rightIndex) {
627       fromIndex = rightIndex;
628     }
629 
630     int strLastIndex = targetCount - 1;
631     char strLastChar = target.charAt(strLastIndex);
632 
633     int min = fromIndex + targetCount - 1;
634 
635     lastCharSearch:
636     for (int i = endIndex; i >= min; i--) {
637       if (builder.charAt(i) == strLastChar) {
638         for (int j = strLastIndex - 1, k = 1; j >= 0; j--, k++) {
639           if (builder.charAt(i - k) != target.charAt(j)) {
640             continue lastCharSearch;
641           }
642         }
643         return i - target.length() + 1;
644       }
645     }
646     return -1;
647   }

649   /**
650    * Creates a {@link net.dv8tion.jda.core.requests.restaction.MessageAction MessageAction} with the
651    * current settings without building a {@link net.dv8tion.jda.core.entities.Message Message}
652    * instance first.
653    *
654    * @param channel The not-null target {@link net.dv8tion.jda.core.entities.MessageChannel
655    *     MessageChannel}
656    * @throws java.lang.IllegalArgumentException If the provided channel is {@code null}
657    * @throws net.dv8tion.jda.core.exceptions.PermissionException If the currently logged in account
658    *     does not have permission to send or read messages in this channel, or if this is a
659    *     PrivateChannel and both users (sender and receiver) are bots.
660    * @return {@link net.dv8tion.jda.core.requests.restaction.MessageAction MessageAction}
661    */
662   public MessageAction sendTo(MessageChannel channel) {
663     Checks.notNull(channel, "Target Channel");
664     switch (channel.getType()) {
665       case TEXT:
666         final TextChannel text = (TextChannel) channel;
667         final Member self = text.getGuild().getSelfMember();
668         if (!self.hasPermission(text, Permission.MESSAGE_READ))
669           throw new InsufficientPermissionException(Permission.MESSAGE_READ);
670         if (!self.hasPermission(text, Permission.MESSAGE_WRITE))
671           throw new InsufficientPermissionException(Permission.MESSAGE_WRITE);
672         break;
673       case PRIVATE:
674         final PrivateChannel priv = (PrivateChannel) channel;
675         if (priv.getUser().isBot() && channel.getJDA().getAccountType() == AccountType.BOT)
676           throw new UnsupportedOperationException("Cannot send a private message between bots.");
677     }
678     final Route.CompiledRoute route = Route.Messages.SEND_MESSAGE.compile(channel.getId());
679     final MessageAction action = new MessageAction(channel.getJDA(), route, channel, builder);
680     return action.tts(isTTS).embed(embed).nonce(nonce);
681   }

683   /**
684    * Creates a {@link net.dv8tion.jda.core.entities.Message Message} object from this MessageBuilder
685    *
686    * <p><b>Hint:</b> You can use {@link #build(int, int)} or {@link
687    * #buildAll(net.dv8tion.jda.core.MessageBuilder.SplitPolicy...) buildAll(SplitPolicy...)} as
688    * possible ways to deal with the 2000 character cap.
689    *
690    * @throws java.lang.IllegalStateException
691    *     <ul>
692    *       <li>If you attempt to build() an empty Message ({@link #length()} is {@code 0} and no
693    *           {@link net.dv8tion.jda.core.entities.MessageEmbed} was provided to {@link
694    *           #setEmbed(net.dv8tion.jda.core.entities.MessageEmbed)})
695    *       <li>If you attempt to build() a Message with more than 2000 characters of content.
696    *     </ul>
697    *
698    * @return the created {@link net.dv8tion.jda.core.entities.Message Message}
699    */
700   public Message build() {
701     String message = builder.toString();
702     if (this.isEmpty())
703       throw new IllegalStateException(
704           "Cannot build a Message with no content. (You never added any content to the message)");
705     if (message.length() > Message.MAX_CONTENT_LENGTH)
706       throw new IllegalStateException(
707           "Cannot build a Message with more than 2000 characters. Please limit your input.");
708 
709     return new DataMessage(isTTS, message, nonce, embed);
710   }

712   /**
713    * Creates a {@link java.util.Queue Queue} of {@link net.dv8tion.jda.core.entities.Message
714    * Message} objects from this MessageBuilder.
715    *
716    * <p>This method splits the content if it exceeds 2000 chars. The splitting behaviour can be
717    * customized using {@link SplitPolicy SplitPolicies}. The method will try the policies in the
718    * order they are passed to it. <br>
719    * If no SplitPolicy is provided each message will be split after exactly 2000 chars.
720    *
721    * <p><b>This is not Markdown safe.</b> An easy workaround is to include <a
722    * href="https://en.wikipedia.org/wiki/Zero-width_space">Zero Width Spaces</a> as predetermined
723    * breaking points to the message and only split on them.
724    *
725    * @param policy The {@link net.dv8tion.jda.core.MessageBuilder.SplitPolicy} defining how to split
726    *     the text in the MessageBuilder into different, individual messages.
727    * @return the created {@link net.dv8tion.jda.core.entities.Message Messages}
728    */
729   public Queue<Message> buildAll(SplitPolicy... policy) {
730     if (this.isEmpty())
731       throw new UnsupportedOperationException(
732           "Cannot build a Message with no content. (You never added any content to the message)");
733 
734     LinkedList<Message> messages = new LinkedList<>();
735 
736     if (builder.length() <= 2000) {
737       messages.add(this.build());
738       return messages;
739     }
740 
741     if (policy == null || policy.length == 0) {
742       policy = new SplitPolicy[] {SplitPolicy.ANYWHERE};
743     }
744 
745     int currentBeginIndex = 0;
746 
747     messageLoop:
748     while (currentBeginIndex < builder.length() - 2001) {
749       for (int i = 0; i < policy.length; i++) {
750         int currentEndIndex = policy[i].nextMessage(currentBeginIndex, this);
751         if (currentEndIndex != -1) {
752           messages.add(build(currentBeginIndex, currentEndIndex));
753           currentBeginIndex = currentEndIndex;
754           continue messageLoop;
755         }
756       }
757       throw new IllegalStateException("Failed to split the messages");
758     }
759 
760     if (currentBeginIndex < builder.length() - 1) {
761       messages.add(build(currentBeginIndex, builder.length() - 1));
762     }
763 
764     if (this.embed != null) {
765       ((DataMessage) messages.get(messages.size() - 1)).setEmbed(embed);
766     }
767 
768     return messages;
769   }
770 
771   protected DataMessage build(int beginIndex, int endIndex) {
772     return new DataMessage(isTTS, builder.substring(beginIndex, endIndex), null, null);
773   }

775   /**
776    * Interface to allow custom implementation of Splitting rules for {@link
777    * #buildAll(net.dv8tion.jda.core.MessageBuilder.SplitPolicy...)
778    * MessageBuilder.buildAll(SplitPolicy...)}.
779    */
780   public interface SplitPolicy {
781     /** Splits on newline chars {@code `\n`}. */
782     SplitPolicy NEWLINE = new CharSequenceSplitPolicy("\n", true);
783 
784     /** Splits on space chars {@code `\u0020`}. */
785     SplitPolicy SPACE = new CharSequenceSplitPolicy(" ", true);
786 
787     /** Splits exactly after 2000 chars. */
788     SplitPolicy ANYWHERE = (i, b) -> Math.min(i + 2000, b.length());
789 
790     /**
791      * Creates a new {@link SplitPolicy} splitting on the specified chars.
792      *
793      * @param chars the chars to split on
794      * @param remove weather to remove the chars when splitting on them
795      * @return a new {@link SplitPolicy}
796      */
797     static SplitPolicy onChars(CharSequence chars, boolean remove) {
798       return new CharSequenceSplitPolicy(chars, remove);
799     }
800 
801     /** Default {@link SplitPolicy} implementation. Splits on a specified {@link CharSequence}. */
802     class CharSequenceSplitPolicy implements SplitPolicy {
803       private final boolean remove;
804       private final CharSequence chars;
805 
806       private CharSequenceSplitPolicy(final CharSequence chars, final boolean remove) {
807         this.chars = chars;
808         this.remove = remove;
809       }
810 
811       @Override
812       public int nextMessage(final int currentBeginIndex, final MessageBuilder builder) {
813         int currentEndIndex =
814             builder.lastIndexOf(
815                 this.chars,
816                 currentBeginIndex,
817                 currentBeginIndex + 2000 - (this.remove ? this.chars.length() : 0));
818         if (currentEndIndex < 0) {
819           return -1;
820         } else {
821           return currentEndIndex + this.chars.length();
822         }
823       }
824     }
825 
826     /**
827      * Calculates the endIndex for the next {@link net.dv8tion.jda.core.entities.Message Message}.
828      *
829      * @param currentBeginIndex the index the next {@link net.dv8tion.jda.core.entities.Message
830      *     Message} should start from
831      * @param builder the {@link net.dv8tion.jda.core.MessageBuilder MessageBuilder}
832      * @return the end Index of the next {@link net.dv8tion.jda.core.entities.Message Message}
833      * @throws java.lang.IllegalStateException when splitting fails
834      */
835     int nextMessage(int currentBeginIndex, MessageBuilder builder);
836   }
