17  package net.dv8tion.jda.core.managers;
18  
19  import net.dv8tion.jda.core.JDA;
20  import net.dv8tion.jda.core.audio.AudioReceiveHandler;
21  import net.dv8tion.jda.core.audio.AudioSendHandler;
22  import net.dv8tion.jda.core.audio.hooks.ConnectionListener;
23  import net.dv8tion.jda.core.audio.hooks.ConnectionStatus;
24  import net.dv8tion.jda.core.entities.Guild;
25  import net.dv8tion.jda.core.entities.VoiceChannel;
26  
27  /**
28   * AudioManager deals with creating, managing and severing audio connections to {@link
29   * net.dv8tion.jda.core.entities.VoiceChannel VoiceChannels}. Also controls audio handlers.
30   */
31  public interface AudioManager {
32  
33    long DEFAULT_CONNECTION_TIMEOUT = 10000;
34  
35    /**
36     * Starts the process to create an audio connection with a {@link
37     * net.dv8tion.jda.core.entities.VoiceChannel VoiceChannel} or, if an audio connection is already
38     * open, JDA will move the connection to the provided VoiceChannel. <br>
39     * <b>Note</b>: Currently you can only be connected to a single {@link
40     * net.dv8tion.jda.core.entities.VoiceChannel VoiceChannel} per {@link
41     * net.dv8tion.jda.core.entities.Guild Guild}.
42     *
43     * <p>This method will automatically move the current connection if one connection is already open
44     * in this underlying {@link Guild}. <br>
45     * Current connections can be closed with {@link #closeAudioConnection()}.
46     *
47     * <p>Client accounts can only connect to a single {@link
48     * net.dv8tion.jda.core.entities.VoiceChannel VoiceChannel} at once. It is an Account-Wide
49     * limitation!
50     *
51     * @param channel The {@link net.dv8tion.jda.core.entities.VoiceChannel VoiceChannel} to open an
52     *     audio connection with.
53     * @throws IllegalArgumentException
54     *     <ul>
55     *       <li>If the provided channel was {@code null}.
56     *       <li>If the provided channel is not part of the Guild that the current audio connection is
57     *           connected to.
58     *     </ul>
59     *
60     * @throws UnsupportedOperationException If audio is disabled due to an internal JDA error
61     * @throws net.dv8tion.jda.core.exceptions.GuildUnavailableException If the Guild is temporarily
62     *     unavailable
63     * @throws net.dv8tion.jda.core.exceptions.InsufficientPermissionException
64     *     <ul>
65     *       <li>If the currently logged in account does not have the Permission {@link
66     *           net.dv8tion.jda.core.Permission#VOICE_CONNECT VOICE_CONNECT}
67     *       <li>If the currently logged in account does not have the Permission {@link
68     *           net.dv8tion.jda.core.Permission#VOICE_MOVE_OTHERS VOICE_MOVE_OTHERS} and the {@link
69     *           net.dv8tion.jda.core.entities.VoiceChannel#getUserLimit() user limit} has been
70     *           exceeded!
71     *     </ul>
72     */
73    void openAudioConnection(VoiceChannel channel);
74  
75    /**
76     * Close down the current audio connection of this {@link net.dv8tion.jda.core.entities.Guild
77     * Guild} and disconnects from the {@link net.dv8tion.jda.core.entities.VoiceChannel
78     * VoiceChannel}. <br>
79     * If this is called when JDA doesn't have an audio connection, nothing happens.
80     */
81    void closeAudioConnection();
82  
83    /**
84     * Gets the {@link net.dv8tion.jda.core.JDA JDA} instance that this AudioManager is a part of.
85     *
86     * @return The corresponding JDA instance
87     */
88    JDA getJDA();
89  
90    /**
91     * Gets the {@link net.dv8tion.jda.core.entities.Guild Guild} instance that this AudioManager is
92     * used for.
93     *
94     * @return The Guild that this AudioManager manages.
95     */
96    Guild getGuild();
97  
98    /**
99     * This can be used to find out if JDA is currently attempting to setup an audio connection. <br>
100    * If this returns true then {@link #getQueuedAudioConnection()} will return the {@link
101    * net.dv8tion.jda.core.entities.VoiceChannel VoiceChannel} that JDA is attempting to setup an
102    * audio connection to.
103    *
104    * @return True, if JDA is currently attempting to create an audio connection.
105    */
106   boolean isAttemptingToConnect();
107 
108   /**
109    * This can be used to find out what {@link net.dv8tion.jda.core.entities.VoiceChannel
110    * VoiceChannel} JDA is currently attempting to setup an audio connection for. If JDA isn't trying
111    * to create an audio connection this will return null. <br>
112    * In addition, if JDA is already connected, this will return null. To determine if JDA is already
113    * has an audio connection with a {@link net.dv8tion.jda.core.entities.VoiceChannel VoiceChannel}
114    * use {@link #isConnected()}
115    *
116    * @return The {@link net.dv8tion.jda.core.entities.VoiceChannel VoiceChannel} that JDA is
117    *     attempting to create an audio connection with, or {@code null} if JDA isn't attempting to
118    *     create a connection.
119    */
120   VoiceChannel getQueuedAudioConnection();
121 
122   /**
123    * The {@link net.dv8tion.jda.core.entities.VoiceChannel VoiceChannel} that JDA currently has an
124    * audio connection to. If JDA currently doesn't have an audio connection to a {@link
125    * net.dv8tion.jda.core.entities.VoiceChannel VoiceChannel} this will return {@code null}.
126    *
127    * @return The {@link net.dv8tion.jda.core.entities.VoiceChannel VoiceChannel} the audio
128    *     connection is connected to or {@code null} if not connected.
129    */
130   VoiceChannel getConnectedChannel();
131 
132   /**
133    * This can be used to find out if JDA currently has an active audio connection with a {@link
134    * net.dv8tion.jda.core.entities.VoiceChannel VoiceChannel}. If this returns true, then {@link
135    * #getConnectedChannel()} will return the {@link net.dv8tion.jda.core.entities.VoiceChannel
136    * VoiceChannel} which JDA is connected to.
137    *
138    * @return True, if JDA currently has an active audio connection.
139    */
140   boolean isConnected();
141 
142   /**
143    * Sets the amount of time, in milliseconds, that will be used as the timeout when waiting for the
144    * audio connection to successfully connect. The default value is 10 second (10,000 milliseconds).
145    * <br>
146    * <b>Note</b>: If you set this value to 0, you can remove timeout functionality and JDA will wait
147    * FOREVER for the connection to be established. This is no advised as it is possible that the
148    * connection may never be established.
149    *
150    * @param timeout The amount of time, in milliseconds, that should be waited when waiting for the
151    *     audio connection to be established.
152    */
153   void setConnectTimeout(long timeout);
154 
155   /**
156    * The currently set timeout value, in <b>milliseconds</b>, used when waiting for an audio
157    * connection to be established.
158    *
159    * @return The currently set timeout.
160    */
161   long getConnectTimeout();
162 
163   /**
164    * Sets the {@link net.dv8tion.jda.core.audio.AudioSendHandler} that the manager will use to
165    * provide audio data to an audio connection. <br>
166    * The handler provided here will persist between audio connection connect and disconnects.
167    * Furthermore, you don't need to have an audio connection to set a handler. When JDA sets up a
168    * new audio connection it will use the handler provided here. <br>
169    * Setting this to null will remove the audio handler.
170    *
171    * <p>JDA recommends <a href="https://github.com/sedmelluq/lavaplayer"
172    * target="_blank">LavaPlayer</a> as an {@link net.dv8tion.jda.core.audio.AudioSendHandler
173    * AudioSendHandler}. It provides a <a
174    * href="https://github.com/sedmelluq/lavaplayer/tree/master/demo-jda" target="_blank">demo</a>
175    * targeted at JDA users.
176    *
177    * @param handler The {@link net.dv8tion.jda.core.audio.AudioSendHandler AudioSendHandler} used to
178    *     provide audio data.
179    */
180   void setSendingHandler(AudioSendHandler handler);
181 
182   /**
183    * The currently set {@link net.dv8tion.jda.core.audio.AudioSendHandler AudioSendHandler}. If
184    * there is no sender currently set, this method will return {@code null}.
185    *
186    * @return The currently active {@link net.dv8tion.jda.core.audio.AudioSendHandler
187    *     AudioSendHandler} or {@code null}.
188    */
189   AudioSendHandler getSendingHandler();
190 
191   /**
192    * Sets the {@link net.dv8tion.jda.core.audio.AudioReceiveHandler AudioReceiveHandler} that the
193    * manager will use to process audio data received from an audio connection.
194    *
195    * <p>The handler provided here will persist between audio connection connect and disconnects.
196    * Furthermore, you don't need to have an audio connection to set a handler. When JDA sets up a
197    * new audio connection it will use the handler provided here. <br>
198    * Setting this to null will remove the audio handler.
199    *
200    * @param handler The {@link net.dv8tion.jda.core.audio.AudioReceiveHandler AudioReceiveHandler}
201    *     used to process received audio data.
202    */
203   void setReceivingHandler(AudioReceiveHandler handler);
204 
205   /**
206    * The currently set {@link net.dv8tion.jda.core.audio.AudioReceiveHandler AudioReceiveHandler}.
207    * If there is no receiver currently set, this method will return {@code null}.
208    *
209    * @return The currently active {@link net.dv8tion.jda.core.audio.AudioReceiveHandler
210    *     AudioReceiveHandler} or {@code null}.
211    */
212   AudioReceiveHandler getReceiveHandler();
213 
214   /**
215    * Sets the {@link net.dv8tion.jda.core.audio.hooks.ConnectionListener ConnectionListener} for
216    * this AudioManager. It will be informed about meta data of any audio connection established
217    * through this AudioManager. Further information can be found in the {@link
218    * net.dv8tion.jda.core.audio.hooks.ConnectionListener ConnectionListener} documentation!
219    *
220    * @param listener A {@link net.dv8tion.jda.core.audio.hooks.ConnectionListener
221    *     ConnectionListener} instance
222    */
223   void setConnectionListener(ConnectionListener listener);
224 
225   /**
226    * The currently set {@link net.dv8tion.jda.core.audio.hooks.ConnectionListener
227    * ConnectionListener} or {@code null} if no {@link
228    * net.dv8tion.jda.core.audio.hooks.ConnectionListener ConnectionListener} has been {@link
229    * #setConnectionListener(ConnectionListener) set}.
230    *
231    * @return The current {@link net.dv8tion.jda.core.audio.hooks.ConnectionListener
232    *     ConnectionListener} instance for this AudioManager.
233    */
234   ConnectionListener getConnectionListener();
235 
236   /**
237    * The current {@link net.dv8tion.jda.core.audio.hooks.ConnectionStatus ConnectionStatus}. <br>
238    * This status indicates represents the connection status of an audio connection.
239    *
240    * @return The current {@link net.dv8tion.jda.core.audio.hooks.ConnectionStatus ConnectionStatus}.
241    */
242   ConnectionStatus getConnectionStatus();
243 
244   /**
245    * Sets whether audio connections from this AudioManager should automatically reconnect or not.
246    * Default {@code true}
247    *
248    * @param shouldReconnect Whether audio connections from this AudioManager should automatically
249    *     reconnect
250    */
251   void setAutoReconnect(boolean shouldReconnect);
252 
253   /**
254    * Whether audio connections from this AudioManager automatically reconnect
255    *
256    * @return Whether audio connections from this AudioManager automatically reconnect
257    */
258   boolean isAutoReconnect();
259 
260   /**
261    * Set this to {@code true} if the current connection should be displayed as muted, this will
262    * cause the {@link net.dv8tion.jda.core.audio.AudioSendHandler AudioSendHandler} packages to not
263    * be ignored by Discord!
264    *
265    * @param muted Whether the connection should stop sending audio and display as muted.
266    */
267   void setSelfMuted(boolean muted);
268 
269   /**
270    * Whether connections from this AudioManager are muted, if this is {@code true} packages by the
271    * registered {@link net.dv8tion.jda.core.audio.AudioSendHandler AudioSendHandler} will be ignored
272    * by Discord.
273    *
274    * @return Whether connections from this AudioManager are muted
275    */
276   boolean isSelfMuted();
277 
278   /**
279    * Sets whether connections from this AudioManager should be deafened. <br>
280    * This does not include being muted, that value can be set individually from {@link
281    * #setSelfMuted(boolean)} and checked via {@link #isSelfMuted()}
282    *
283    * @param deafened Whether connections from this AudioManager should be deafened.
284    */
285   void setSelfDeafened(boolean deafened);
286 
287   /**
288    * Whether connections from this AudioManager are deafened. <br>
289    * This does not include being muted, that value can be set individually from {@link
290    * #setSelfMuted(boolean)} and checked via {@link #isSelfMuted()}
291    *
292    * @return True, if connections from this AudioManager are deafened
293    */
294   boolean isSelfDeafened();
295 }
