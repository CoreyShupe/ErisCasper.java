31  /**
32   * An {@link #update() updatable} manager that allows to modify webhook settings like the {@link
33   * #getNameField() default name} or the {@link #getAvatarField() default avatar}.
34   *
35   * <p>This manager allows to modify multiple fields at once by getting the {@link
36   * net.dv8tion.jda.core.managers.fields.WebhookField WebhookField} for specific properties and
37   * setting or resetting their values; followed by a call of {@link #update()}!
38   *
39   * <p>The {@link net.dv8tion.jda.core.managers.WebhookManager WebhookManager} implementation
40   * simplifies this process by giving simple setters that return the {@link #update() update} {@link
41   * net.dv8tion.jda.core.requests.RestAction RestAction}
42   *
43   * <p><b>Note</b>: To {@link #update() update} this manager the currently logged in account requires
44   * the Permission {@link net.dv8tion.jda.core.Permission#MANAGE_WEBHOOKS MANAGE_WEBHOOKS} in the
45   * parent TextChannel
46   */
47  public class WebhookManagerUpdatable {
48    protected final Webhook webhook;
49  
50    protected WebhookField<String> name;
51    protected WebhookField<Icon> avatar;
52    protected WebhookField<TextChannel> channel;
53  
54    /**
55     * Creates a new WebhookManagerUpdatable instance
56     *
57     * @param webhook The target {@link net.dv8tion.jda.core.entities.Webhook Webhook} to modify
58     */
59    public WebhookManagerUpdatable(Webhook webhook) {
60      this.webhook = webhook;
61      setupFields();
62    }

65     * The {@link net.dv8tion.jda.core.JDA JDA} instance of this Manager

67     * @return the corresponding JDA instance

70      return webhook.getJDA();
71    }

159    * @return {@link net.dv8tion.jda.core.managers.fields.WebhookField WebhookField} - Type: {@link

161    */

163     return channel;
164   }

166   /**
167    * Resets all {@link net.dv8tion.jda.core.managers.fields.Field Fields} for this manager instance
168    * by calling {@link net.dv8tion.jda.core.managers.fields.Field#reset() Field.reset()}
169    * sequentially <br>
170    * This is automatically called by {@link #update()}
171    */
172   public void reset() {
173     name.reset();
174     avatar.reset();
175     channel.reset();
176   }

178   /**
179    * Creates a new {@link net.dv8tion.jda.core.requests.RestAction RestAction} instance that will
180    * apply <b>all</b> changes that have been made to this manager instance. <br>
181    * If no changes have been made this will simply return {@link
182    * net.dv8tion.jda.core.requests.RestAction.EmptyRestAction EmptyRestAction}.
183    *
184    * <p>Before applying new changes it is recommended to call {@link #reset()} to reset previous
185    * changes. <br>
186    * This is automatically called if this method returns successfully.
187    *
188    * <p>Possible {@link net.dv8tion.jda.core.requests.ErrorResponse ErrorResponses} for this update
189    * include the following:
190    *
191    * <ul>
192    *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#UNKNOWN_CHANNEL UNKNOWN_CHANNEL} <br>
193    *       If the TextChannel was deleted before finishing the task
194    *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#MISSING_ACCESS MISSING_ACCESS} <br>
195    *       If the currently logged in account was removed from the Guild before finishing the task
196    *   <li>{@link net.dv8tion.jda.core.requests.ErrorResponse#MISSING_PERMISSIONS
197    *       MISSING_PERMISSIONS} <br>
198    *       If the currently logged in account loses the {@link
199    *       net.dv8tion.jda.core.Permission#MANAGE_WEBHOOKS MANAGE_WEBHOOKS Permission}
200    * </ul>
201    *
202    * @throws net.dv8tion.jda.core.exceptions.InsufficientPermissionException If the currently logged
203    *     in account does not have the Permission {@link
204    *     net.dv8tion.jda.core.Permission#MANAGE_WEBHOOKS MANAGE_WEBHOOKS} in either the current or
205    *     selected new TextChannel.
206    * @return {@link net.dv8tion.jda.core.requests.restaction.AuditableRestAction
207    *     AuditableRestAction} <br>
208    *     Applies all changes that have been made in a single api-call.
209    */
210   public AuditableRestAction<Void> update() {
211     Member self = getGuild().getSelfMember();
212     if (!self.hasPermission(webhook.getChannel(), Permission.MANAGE_WEBHOOKS))
213       throw new InsufficientPermissionException(Permission.MANAGE_WEBHOOKS);
214     if (channel.isSet() && !self.hasPermission(channel.getValue(), Permission.MANAGE_WEBHOOKS))
215       throw new InsufficientPermissionException(
216           Permission.MANAGE_WEBHOOKS, "Permission not available in selected new channel");
217     if (!shouldUpdate()) return new AuditableRestAction.EmptyRestAction<>(getJDA(), null);
218 
219     JSONObject data = new JSONObject();
220     data.put("name", name.getOriginalValue());
221 
222     if (channel.shouldUpdate()) data.put("channel_id", channel.getValue().getId());
223     if (name.shouldUpdate()) data.put("name", name.getValue());
224     if (avatar.shouldUpdate()) {
225       Icon value = avatar.getValue();
226       data.put("avatar", value != null ? value.getEncoding() : JSONObject.NULL);
227     }
228 
229     Route.CompiledRoute route = Route.Webhooks.MODIFY_WEBHOOK.compile(webhook.getId());
230     return new AuditableRestAction<Void>(getJDA(), route, data) {
231       @Override
232       protected void handleResponse(Response response, Request<Void> request) {
233         if (response.isOk()) request.onSuccess(null);
234         else request.onFailure(response);
235       }
236     };
237   }
238 
239   protected boolean shouldUpdate() {
240     return name.shouldUpdate() || avatar.shouldUpdate() || channel.shouldUpdate();
241   }
242 
243   protected void setupFields() {
244     name =
245         new WebhookField<String>(this, webhook::getName) {
246           @Override
247           public void checkValue(String value) {
248             Checks.notNull(value, "default name");
249           }
250         };
251 
252     avatar =
253         new WebhookField<Icon>(this, null) {
254           @Override
255           public void checkValue(Icon value) {}
256 
257           @Override
258           public Icon getOriginalValue() {
259             throw new UnsupportedOperationException(
260                 "Cannot easily provide the original Avatar. Use User#getIconUrl() and download it yourself.");
261           }
262 
263           @Override
264           public boolean shouldUpdate() {
265             return isSet();
266           }
267         };
268 
269     channel =
270         new WebhookField<TextChannel>(this, webhook::getChannel) {
271           @Override
272           public void checkValue(TextChannel value) {
273             Checks.notNull(value, "channel");
274             Checks.check(
275                 value.getGuild().equals(getGuild()), "Channel is not from the same Guild!");
276           }
277         };
278   }
