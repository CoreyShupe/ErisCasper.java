17  package net.dv8tion.jda.webhook;
18  
19  import java.util.concurrent.Executors;
20  import java.util.concurrent.ScheduledExecutorService;
21  import java.util.concurrent.ThreadFactory;
22  import java.util.regex.Matcher;
23  import java.util.regex.Pattern;
24  import net.dv8tion.jda.core.entities.Webhook;
25  import net.dv8tion.jda.core.utils.Checks;
26  import net.dv8tion.jda.core.utils.MiscUtil;
27  import okhttp3.OkHttpClient;
28  
29  /**
30   * Builder that creates a new {@link net.dv8tion.jda.webhook.WebhookClient WebhookClient} instance
31   */
32  public class WebhookClientBuilder {
33    public static final OkHttpClient.Builder DEFAULT_HTTP_BUILDER = new OkHttpClient.Builder();
34    private static final Pattern WEBHOOK_PATTERN =
35        Pattern.compile(
36            "(?:https?://)?(?:\\w+\\.)?discordapp\\.com/api(?:/v\\d+)?/webhooks/(\\d+)/([\\w-]+)(?:/(?:\\w+)?)?");
37  
38    protected final long id;
39    protected final String token;
40    protected ScheduledExecutorService pool;
41    protected OkHttpClient.Builder builder;
42    protected OkHttpClient client;
43    protected ThreadFactory threadFactory;
44    protected boolean isDaemon;
45  
46    /**
47     * Creates a new WebhookClientBuilder with the provided id and token
48     *
49     * @param id The snowflake id of the target webhook
50     * @param token The authorization token of the target webhook <br>
51     *     <b>This is not a bot/client token!</b>
52     * @throws java.lang.IllegalArgumentException If the provided token is {@code null} or contains
53     *     any whitespace!
54     */
55    public WebhookClientBuilder(final long id, final String token) {
56      Checks.noWhitespace(token, "Token");
57      this.id = id;
58      this.token = token;
59    }
60  
61    /**
62     * Creates a new WebhookClientBuilder with the provided webhook URL
63     *
64     * @param url The URL of the webhook. May be directly copied from Discord's UI <br>
65     *     Example: {@code https://discordapp.com/api/webhooks/123456789012345678/my-webhook-token}
66     *     <br>
67     *     This constructor also parses URLs pointing to subdomains of {@code discordapp.com}
68     * @throws java.lang.IllegalArgumentException If the provided URL is {@code null} or is
69     *     incorrectly formatted
70     */
71    public WebhookClientBuilder(String url) {
72      Matcher matcher = WEBHOOK_PATTERN.matcher(url);
73      if (!matcher.matches()) {
74        throw new IllegalArgumentException("Failed to parse webhook URL");
75      }
76  
77      this.id = MiscUtil.parseSnowflake(matcher.group(1));
78      this.token = matcher.group(2);
79    }
80  
81    /**
82     * Creates a new WebhookClientBuilder with the provided id and token
83     *
84     * @param webhook The target {@link net.dv8tion.jda.core.entities.Webhook Webhook}
85     * @throws java.lang.NullPointerException If the provided {@link
86     *     net.dv8tion.jda.core.entities.Webhook Webhook} is {@code null}
87     */
88    public WebhookClientBuilder(Webhook webhook) {
89      this(webhook.getIdLong(), webhook.getToken());
90    }
91  
92    /**
93     * The {@link java.util.concurrent.ScheduledExecutorService ScheduledExecutorService} that should
94     * be used to handle rate limits! <br>
95     * By default this creates a new executor with 1 core thread!
96     *
97     * <p><b><u>Closing the {@link net.dv8tion.jda.webhook.WebhookClient WebhookClient} will close
98     * this executor service!</u></b>
99     *
100    * @param executorService The executor service that should be used
101    * @return The current WebhookClientBuilder for chaining convenience
102    */
103   public WebhookClientBuilder setExecutorService(ScheduledExecutorService executorService) {
104     this.pool = executorService;
105     return this;
106   }
107 
108   /**
109    * The {@link okhttp3.OkHttpClient OkHttpClient} that should be used to make HTTP requests!
110    *
111    * <p>Setting this will skip the {@link #setHttpClientBuilder(okhttp3.OkHttpClient.Builder)
112    * setHttpClientBuilder(OkHttpClient.Builder)} setting and directly use the provided client!
113    *
114    * @param client The client that should be used
115    * @return The current WebhookClientBuilder for chaining convenience
116    */
117   public WebhookClientBuilder setHttpClient(OkHttpClient client) {
118     this.client = client;
119     return this;
120   }
121 
122   /**
123    * The {@link okhttp3.OkHttpClient.Builder OkHttpClient.Builder} that should be used to create the
124    * {@link okhttp3.OkHttpClient OkHttpClient} of the resulting {@link
125    * net.dv8tion.jda.webhook.WebhookClient WebhookClient}! <br>
126    * If not set or {@code null} this builder will use the {@link #DEFAULT_HTTP_BUILDER} instance.
127    *
128    * <p>This setting is ignored if {@link #setHttpClient(okhttp3.OkHttpClient)} is set!
129    *
130    * @param builder The builder that should be used
131    * @return The current WebhookClientBuilder for chaining convenience
132    */
133   public WebhookClientBuilder setHttpClientBuilder(OkHttpClient.Builder builder) {
134     Checks.notNull(builder, "Builder");
135     this.builder = builder;
136     return this;
137   }
138 
139   /**
140    * Factory that should be used by the default {@link java.util.concurrent.ScheduledExecutorService
141    * ScheduledExecutorService} to create Threads for rate limitation handling of the created {@link
142    * net.dv8tion.jda.webhook.WebhookClient WebhookClient}! <br>
143    * This allows changing thread information such as name without having to create your own
144    * executor.
145    *
146    * @param factory The {@link java.util.concurrent.ThreadFactory ThreadFactory} that will be used
147    *     when no {@link java.util.concurrent.ScheduledExecutorService ScheduledExecutorService} has
148    *     been set via {@link #setExecutorService(ScheduledExecutorService)}
149    * @return The current WebhookClientBuilder for chaining convenience
150    */
151   public WebhookClientBuilder setThreadFactory(ThreadFactory factory) {
152     this.threadFactory = factory;
153     return this;
154   }
155 
156   /**
157    * Whether rate limit threads of the created {@link net.dv8tion.jda.webhook.WebhookClient
158    * WebhookClient} should be treated as {@link Thread#isDaemon()} or not. <br>
159    * <b>Default: false</b>
160    *
161    * <p>This will not be used when the default thread pool has been set via {@link
162    * #setExecutorService(ScheduledExecutorService)}!
163    *
164    * @param isDaemon True, if the threads should be daemon
165    * @return The current WebhookClientBuilder for chaining convenience
166    */
167   public WebhookClientBuilder setDaemon(boolean isDaemon) {
168     this.isDaemon = isDaemon;
169     return this;
170   }
171 
172   /**
173    * Builds a new {@link net.dv8tion.jda.webhook.WebhookClient WebhookClient} instance with the
174    * current state of this builder.
175    *
176    * <p><b><u>Remember to close the WebhookClient once you don't need it anymore to free
177    * resources!</u></b>
178    *
179    * @return The new WebhookClient instance
180    */
181   public WebhookClient build() {
182     OkHttpClient client = this.client;
183     if (client == null) {
184       if (builder == null) builder = DEFAULT_HTTP_BUILDER;
185       client = builder.build();
186     }
187     if (pool == null) {
188       if (threadFactory == null) threadFactory = new DefaultWebhookThreadFactory();
189       pool = Executors.newSingleThreadScheduledExecutor(threadFactory);
190     }
191     return new WebhookClient(id, token, client, pool);
192   }
193 
194   public final class DefaultWebhookThreadFactory implements ThreadFactory {
195     @Override
196     public Thread newThread(Runnable r) {
197       final Thread thread = new Thread(r, "Webhook-RateLimit Thread WebhookID: " + id);
198       thread.setDaemon(isDaemon);
199       return thread;
200     }
201   }
