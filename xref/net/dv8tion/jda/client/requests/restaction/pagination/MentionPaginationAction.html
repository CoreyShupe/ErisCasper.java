17  package net.dv8tion.jda.client.requests.restaction.pagination;
18  
19  import java.util.LinkedList;
20  import java.util.List;
21  import net.dv8tion.jda.core.JDA;
22  import net.dv8tion.jda.core.entities.EntityBuilder;
23  import net.dv8tion.jda.core.entities.Guild;
24  import net.dv8tion.jda.core.entities.Message;
25  import net.dv8tion.jda.core.requests.Request;
26  import net.dv8tion.jda.core.requests.Response;
27  import net.dv8tion.jda.core.requests.Route;
28  import net.dv8tion.jda.core.requests.restaction.pagination.PaginationAction;
29  import org.json.JSONArray;
30  import org.json.JSONException;
31  
32  /**
33   * {@link net.dv8tion.jda.core.requests.restaction.pagination.PaginationAction PaginationAction}
34   * that paginates the endpoint {@link net.dv8tion.jda.core.requests.Route.Self#GET_RECENT_MENTIONS
35   * Route.Self.GET_RECENT_MENTIONS}.
36   *
37   * <p><b>Must provide not-null {@link net.dv8tion.jda.core.entities.Guild Guild} to compile a valid
38   * guild mentions pagination route.</b>, else it uses the global pagination route.
39   *
40   * <h2>Limits:</h2>
41   *
42   * Minimum - 1 <br>
43   * Maximum - 100
44   *
45   * <h1>Example</h1>
46   *
47   * <pre><code>
48   * MentionPaginationAction mentions = guild.getRecentMentions();
49   * mentions.setEveryone(false);
50   * for (Message message : mentions)
51   * {
52   *     System.out.printf("%#s: %s\n", message.getAuthor(), message.getContent());
53   * }
54   * </code></pre>
55   *
56   * @since 3.0
57   * @author Florian Spieß
58   */
59  public class MentionPaginationAction extends PaginationAction<Message, MentionPaginationAction> {
60  
61    protected final Guild guild;
62  
63    protected boolean isEveryone = true;
64    protected boolean isRole = true;
65  
66    /**
67     * Creates a new MentionPaginationAction <br>
68     * This constructor effectively makes this target all recent mentions to get the recent mentions
69     * for a specific {@link net.dv8tion.jda.core.entities.Guild Guild} use {@link
70     * #MentionPaginationAction(net.dv8tion.jda.core.entities.Guild)} instead!
71     *
72     * @param api The current JDA entity
73     */
74    public MentionPaginationAction(JDA api) {
75      this(api, null);
76    }
77  
78    /**
79     * Creates a new MentionPaginationAction <br>
80     * This constructor effectively makes this target specifically only the recent mentions for the
81     * specified {@link net.dv8tion.jda.core.entities.Guild Guild}! <br>
82     * To get the global scope use {@link #MentionPaginationAction(net.dv8tion.jda.core.JDA)} instead.
83     *
84     * @param guild The Non-Null target {@link net.dv8tion.jda.core.entities.Guild Guild}
85     * @throws java.lang.NullPointerException If the provided {@code guild} is {@code null}
86     */
87    public MentionPaginationAction(Guild guild) {
88      this(guild.getJDA(), guild);
89    }
90  
91    private MentionPaginationAction(JDA api, Guild guild) {
92      super(api, Route.Self.GET_RECENT_MENTIONS.compile(), 1, 100, 100);
93  
94      this.guild = guild;
95    }
96  
97    /**
98     * The current target {@link net.dv8tion.jda.core.entities.Guild Guild} for this
99     * MentionPaginationAction. <br>
100    * This can be {@code null} if this MentionPaginationAction does not target mentions from a
101    * specific Guild!
102    *
103    * @return Possibly-null target Guild
104    */
105   public Guild getGuild() {
106     return guild;
107   }
108 
109   /**
110    * Sets whether this MentionPaginationAction should include mentions that mention the public role
111    * of a Guild. <br>
112    * Default: {@code true}
113    *
114    * @param isEveryoneMention Whether to include everyone mentions
115    * @return The current MentionPaginationAction for chaining convenience
116    */
117   public MentionPaginationAction setEveryone(boolean isEveryoneMention) {
118     this.isEveryone = isEveryoneMention;
119     return this;
120   }
121 
122   /**
123    * Sets whether this MentionPaginationAction should include mentions that mention a role in a
124    * Guild. <br>
125    * Default: {@code true}
126    *
127    * @param isRoleMention Whether to include role mentions
128    * @return The current MentionPaginationAction for chaining convenience
129    */
130   public MentionPaginationAction setRole(boolean isRoleMention) {
131     this.isRole = isRoleMention;
132     return this;
133   }
134 
135   @Override
136   protected Route.CompiledRoute finalizeRoute() {
137     Route.CompiledRoute route = super.finalizeRoute();
138 
139     String limit, before, everyone, role;
140     limit = String.valueOf(super.getLimit());
141     before = last != null ? last.getId() : null;
142     everyone = String.valueOf(isEveryone);
143     role = String.valueOf(isRole);
144 
145     route = route.withQueryParams("limit", limit, "roles", role, "everyone", everyone);
146 
147     if (guild != null) route = route.withQueryParams("guild_id", guild.getId());
148 
149     if (before != null) route = route.withQueryParams("before", before);
150 
151     return route;
152   }
153 
154   @Override
155   protected void handleResponse(Response response, Request<List<Message>> request) {
156     if (!response.isOk()) {
157       request.onFailure(response);
158       return;
159     }
160 
161     EntityBuilder builder = api.getEntityBuilder();
162     ;
163     List<Message> mentions = new LinkedList<>();
164     JSONArray arr = response.getArray();
165     for (int i = 0; i < arr.length(); i++) {
166       try {
167         final Message msg = builder.createMessage(arr.getJSONObject(i), false);
168         mentions.add(msg);
169         if (useCache) cached.add(msg);
170         last = msg;
171       } catch (JSONException | NullPointerException e) {
172         LOG.warn("Encountered exception in MentionPagination", e);
173       }
174     }
175 
176     request.onSuccess(mentions);
177   }
178 }
