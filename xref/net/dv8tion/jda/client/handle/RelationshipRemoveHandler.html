17  package net.dv8tion.jda.client.handle;
18  
19  import net.dv8tion.jda.client.entities.Group;
20  import net.dv8tion.jda.client.entities.Relationship;
21  import net.dv8tion.jda.client.entities.RelationshipType;
22  import net.dv8tion.jda.client.events.relationship.FriendRemovedEvent;
23  import net.dv8tion.jda.client.events.relationship.FriendRequestCanceledEvent;
24  import net.dv8tion.jda.client.events.relationship.FriendRequestIgnoredEvent;
25  import net.dv8tion.jda.client.events.relationship.UserUnblockedEvent;
26  import net.dv8tion.jda.core.entities.User;
27  import net.dv8tion.jda.core.entities.impl.GuildImpl;
28  import net.dv8tion.jda.core.entities.impl.JDAImpl;
29  import net.dv8tion.jda.core.entities.impl.PrivateChannelImpl;
30  import net.dv8tion.jda.core.entities.impl.UserImpl;
31  import net.dv8tion.jda.core.handle.EventCache;
32  import net.dv8tion.jda.core.handle.SocketHandler;
33  import net.dv8tion.jda.core.requests.WebSocketClient;
34  import org.json.JSONObject;
35  
36  public class RelationshipRemoveHandler extends SocketHandler {
37    public RelationshipRemoveHandler(JDAImpl api) {
38      super(api);
39    }
40  
41    @Override
42    protected Long handleInternally(JSONObject content) {
43      final long userId = content.getLong("id");
44      RelationshipType type = RelationshipType.fromKey(content.getInt("type"));
45  
46      // Technically this could be used to detect when another user has unblocked us,
47      // but it seems like functionality that may change so I'm not supporting it.
48      if (type == RelationshipType.NO_RELATIONSHIP) return null;
49  
50      // Make sure that we get the proper relationship, not just any one cached by this userId.
51      // Deals with possibly out of order RELATIONSHIP_REMOVE and RELATIONSHIP_ADD when blocking a
52      // Friend.
53      Relationship relationship = api.asClient().getRelationshipById(userId, type);
54      if (relationship == null) {
55        api.getEventCache()
56            .cache(EventCache.Type.RELATIONSHIP, userId, () -> handle(responseNumber, allContent));
57        EventCache.LOG.debug(
58            "Received a RELATIONSHIP_REMOVE for a relationship that was not yet cached! JSON: {}",
59            content);
60        return null;
61      }
62      api.asClient().getRelationshipMap().remove(userId);
63  
64      if (relationship.getType() == RelationshipType.FRIEND) {
65        // The user is not in a different guild that we share
66        if (api.getGuildMap()
67            .valueCollection()
68            .stream()
69            .noneMatch(g -> ((GuildImpl) g).getMembersMap().containsKey(userId))) {
70          UserImpl user = (UserImpl) api.getUserMap().remove(userId);
71          if (user.hasPrivateChannel()) {
72            PrivateChannelImpl priv = (PrivateChannelImpl) user.getPrivateChannel();
73            user.setFake(true);
74            priv.setFake(true);
75            api.getFakeUserMap().put(user.getIdLong(), user);
76            api.getFakePrivateChannelMap().put(priv.getIdLong(), priv);
77          } else {
78            // While the user might not have a private channel, if this is a client account then the
79            // user
80            // could be in a Group, and if so we need to change the User object to be fake and
81            // place it in the FakeUserMap
82            for (Group grp : api.asClient().getGroups()) {
83              if (grp.getNonFriendUsers().contains(user)) {
84                user.setFake(true);
85                api.getFakeUserMap().put(user.getIdLong(), user);
86                break;
87              }
88            }
89          }
90          api.getEventCache().clear(EventCache.Type.USER, userId);
91        }
92      } else {
93        // Checks that the user is fake, has no privateChannel,and is not in any other groups
94        // then we remove the fake user from the fake cache as it was only in this group
95        // Note: we getGroups() which gets all groups, however we already removed the user from the
96        // current group.
97        User user = relationship.getUser();
98        if (user.isFake()
99            && !user.hasPrivateChannel()
100           && api.asClient().getGroups().stream().noneMatch(g -> g.getUsers().contains(user))) {
101         api.getFakeUserMap().remove(userId);
102       }
103     }
104 
105     switch (type) {
106       case FRIEND:
107         api.getEventManager().handle(new FriendRemovedEvent(api, responseNumber, relationship));
108         break;
109       case BLOCKED:
110         api.getEventManager().handle(new UserUnblockedEvent(api, responseNumber, relationship));
111         break;
112       case INCOMING_FRIEND_REQUEST:
113         api.getEventManager()
114             .handle(new FriendRequestIgnoredEvent(api, responseNumber, relationship));
115         break;
116       case OUTGOING_FRIEND_REQUEST:
117         api.getEventManager()
118             .handle(new FriendRequestCanceledEvent(api, responseNumber, relationship));
119         break;
120       default:
121         WebSocketClient.LOG.warn(
122             "Received a RELATIONSHIP_REMOVE with an unknown RelationshipType! JSON: {}", content);
123         return null;
124     }
125     api.getEventCache().clear(EventCache.Type.RELATIONSHIP, userId);
126     return null;
127   }
128 }
