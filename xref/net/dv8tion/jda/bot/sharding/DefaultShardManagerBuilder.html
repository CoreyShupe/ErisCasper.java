97    /**
98     * Sets the {@link org.slf4j.MDC MDC} mappings provider to use in JDA. <br>
99     * If sharding is enabled JDA will automatically add a {@code jda.shard} context with the format
100    * {@code [SHARD_ID / TOTAL]} where {@code SHARD_ID} and {@code TOTAL} are the shard
101    * configuration. Additionally it will provide context for the id via {@code jda.shard.id} and the
102    * total via {@code jda.shard.total}.
103    *
104    * <p><b>The manager will call this with a shardId and it is recommended to provide a different
105    * context map for each shard!</b> <br>

107    * function is not null!

110    *     to reset
111    * @return The {@link net.dv8tion.jda.bot.sharding.DefaultShardManagerBuilder
112    *     DefaultShardManagerBuilder} instance. Useful for chaining.
113    * @see <a href="https://www.slf4j.org/api/org/slf4j/MDC.html" target="_blank">MDC Javadoc</a>
114    */
115   public DefaultShardManagerBuilder setContextMap(
116       IntFunction<ConcurrentMap<String, String>> provider) {
117     this.contextProvider = provider;
118     if (provider != null) this.enableContext = true;
119     return this;
120   }

681    * The login process runs in a different thread, so while this will return immediately, {@link

683    * {@link net.dv8tion.jda.bot.sharding.ShardManager ShardManager} methods have the chance to
684    * return incorrect information. <br>
685    * The main use of this method is to start the JDA connect process and do other things in parallel
686    * while startup is being performed like database connection or local resource loading.

689    * started.

691    * @throws LoginException If the provided token is invalid.
692    * @throws IllegalArgumentException If the provided token is empty or null.
693    * @return A {@link net.dv8tion.jda.bot.sharding.ShardManager ShardManager} instance that has
694    *     started the login process. It is unknown as to whether or not loading has finished when
695    *     this returns.
696    */
697   public ShardManager build() throws LoginException, IllegalArgumentException {
698     final DefaultShardManager manager =
699         new DefaultShardManager(
700             this.shardsTotal,
701             this.shards,
702             this.sessionController,
703             this.listeners,
704             this.token,
705             this.eventManager,
706             this.audioSendFactory,
707             this.gameProvider,
708             this.statusProvider,
709             this.httpClientBuilder,
710             this.wsFactory,
711             this.threadFactory,
712             this.maxReconnectDelay,
713             this.corePoolSize,
714             this.enableVoice,
715             this.enableShutdownHook,
716             this.enableBulkDeleteSplitting,
717             this.autoReconnect,
718             this.idleProvider,
719             this.retryOnTimeout,
720             this.useShutdownNow,
721             this.enableContext,
722             this.contextProvider);
